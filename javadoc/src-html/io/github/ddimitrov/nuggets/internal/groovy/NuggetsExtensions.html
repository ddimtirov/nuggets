017package io.github.ddimitrov.nuggets.internal.groovy;
018
019import groovy.lang.Closure;
020import groovy.lang.DelegatesTo;
021import io.github.ddimitrov.nuggets.ExceptionTransformerBuilder;
022import io.github.ddimitrov.nuggets.Extractors;
023import org.intellij.lang.annotations.Identifier;
024import org.jetbrains.annotations.NotNull;
025
026import java.util.function.Function;
027
028/** Extra API to make Groovy usage nicer */
029public class NuggetsExtensions {
030    private NuggetsExtensions() { }
031
032    /**
033     * Manipulate the exception contents on ad-hoc basis
034     * @param self the target of this extension method
035     * @param transformationSpec a spec closure that will be run on {@link ExceptionTransformerBuilder} delegate.
036     * @return the transformed exception - typically we would throw or log that.
037     */
038    public static Throwable transform(Throwable self, @DelegatesTo(ExceptionTransformerBuilder.class) Closure<?> transformationSpec) {
039        ExceptionTransformerBuilder builder = new ExceptionTransformerBuilder();
040        transformationSpec.setDelegate(builder);
041        transformationSpec.call();
042        Function<Throwable, Throwable> transformation = builder.build();
043        return transformation.apply(self);
044    }
045
046
047    /**
048     * <p>Extract the value of a static field regardless of its visibility. If there are multiple shadowed fields with the
049     * given name, return the value of the first found when traversing the inheritance hierarchy starting from the
050     * {@code type}, until {@code Object}. </p>
051     * <p>You can use this method to easily get {@code Unsafe} and cache it for further usage, like this:</p>
052     * <pre><code>
053     * public static final def UNSAFE = Unsafe.peekField("theUnsafe") as Unsafe
054     * </code></pre>
055     *
056     * @param self the class whose field we are peeking
057     * @param fieldName the name of the field to extract.
058     *
059     * @return the extracted value of the requested field.
060     * @throws IllegalArgumentException if the {@code fieldType} is a primitive type.
061     *
062     * @see Extractors#peekField(Object, Class, String, Class)
063     */
064    public static Object peekField(@NotNull Class<?> self, @NotNull @Identifier String fieldName) {
065        return Extractors.peekField(null, self, fieldName, Object.class);
066    }
067
068    /**
069     * <p>Extract the value of a field regardless of its visibility. If there are multiple shadowed fields with the
070     * given name, return the value of the first found when traversing the inheritance hierarchy starting from the
071     * {@code type}, until {@code Object}. </p>
072     *
073     * @param self the class whose field we are peeking
074     * @param fieldName the name of the field to extract.
075     * @param value the desired new value
076     *
077     * @see Extractors#pokeField(Object, Class, String, Object)
078     */
079    public static void pokeField(@NotNull Class<?> self, @NotNull @Identifier String fieldName, Object value) {
080        Extractors.pokeField(null, self, fieldName, value);
081    }
082
083}
