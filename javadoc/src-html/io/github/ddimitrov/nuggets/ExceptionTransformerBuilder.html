017package io.github.ddimitrov.nuggets;
018
019import org.jetbrains.annotations.Contract;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022
023import java.util.Arrays;
024import java.util.Collection;
025import java.util.List;
026import java.util.Objects;
027import java.util.function.Consumer;
028import java.util.function.Function;
029import java.util.function.Predicate;
030import java.util.regex.Pattern;

032/**
033 * <p>Builds a transformer function manipulating exception messages, stack traces and nesting structure
034 * (used internally by {@link Exceptions}).</p>
035 * <p> Most of the methods of this class return the current instance, so they can be chained. We call these
036 * <em>configuration methods</em> and their purpose is to define the behaviour of transforming function.
037 * Calling configuration method after the transforming function is built is an error.</p>
038 *
039 * <p>The {@link #build()} method creates a transforming function and returns it. The {@link #done()} method
040 * creates the transforming function and passes it to an action callback specified at construction time - this
041 * is useful when a service wants to allow the client to configure a managed transformer.
042 * We call these <em>builder methods</em>.</p>
043 *
044 * <p>As a simple rule, when we call methods multiple times, the configuration methods are accumulated, while the
045 * builder methods are idempotent. Notable exception is {@link #replaceMessage(Function)} that can be called only
046 * once.</p>
047 *
048 * <p>This class should be used by a single thread. The transforming function, that is returned from {@link #build()}
049 * is threadsafe only if all the function-classes passed to the configuration methods are threadsafe.</p>
050 *
051 * <p>Example:</p>
052 *
053 * <pre><code>
054 * Function&lt;Throwable, Throwable&gt; cleanup = new ExceptionTransformerBuilder()
055 *           .replaceStackTrace(ExceptionTransformerBuilder.FILTER_GROOVY_SYNTHETIC)
056 *           .filterStackFramesForClass(Pattern.compile("^(java.lang.reflect|sun.reflect).*"))
057 *           .filterStackFramesForClass(Pattern.compile("^(org.codehaus.groovy|org.gradle).*"))
058 *           .unwrapThese(InvocationTargetException.class)
059 *           .unwrapWhen(ex -&gt; ex.getMessage() == null &amp;&amp; ex.getClass().equals(RuntimeException.class))
060 *           .build();
061 * // somewhere in a top-level exception handler
062 *      ....
063 * } catch (Exception e) {
064 *     logger.error("something really bad happened, cleanup.apply(e));
065 * }
066 *
067 * </code></pre>
068 *
069 * @see Exceptions#transformOnRethrow()
070 * @see Exceptions#rethrowTransformed(Throwable, boolean)
071 */
072public class ExceptionTransformerBuilder {
073    private final @NotNull Function<Throwable, Throwable> transformFunction = this::transform;
074    private final @Nullable Consumer<Function<Throwable, Throwable>> buildCompleteAction;
075
076    private @Nullable Function<Throwable, String> messageMapper;
077    private @NotNull Predicate<Throwable> unwrap = it -> false;
078    private @NotNull Function<StackTraceElement[], StackTraceElement[]> stackMapper = Function.identity();
079    private @NotNull Function<@Nullable StackTraceElement, @Nullable StackTraceElement> frameMapper = Function.identity();
080    private boolean done;
081
082    /**
083     * <p>Create a builder for exception transformer function, allows to clean up and enrich a stack trace
084     * and get rid of insignificant stack frames, so troubleshooters can focus on what matters.</p>
085     *
086     * <p>To use, call the configuring methods by daisy-chaining them, and finally call {@link #build()}
087     * to get the transformer function. If you call a configuring method after that, you would get an
088     * {@code IllegalStateException.}</p>
089     */
090    public ExceptionTransformerBuilder() {
091        this(null);
092    }
093
094    /**
095     * <p>Create a configurer for exception transformer function used by a service component.
096     * In the typical use-case, the service will use this constructor and return the builder
097     * to the client, who will call the configuring methods by daisy-chaining them, and finally
098     * call {@link #done()}, which will build the result and invoke {@code #buildCompleteAction}.
099     * The {@code #buildCompleteAction} is specified by the service and is not visible to the client
100     * - typically it just assigns the passed-in transformer function to a field or a thread-local.</p>
101     *
102     * <p>To use, to publish the transformer function. If you call a configuring method after that,
103     * you would get an {@code IllegalStateException.}</p>
104     *
105     * @param buildCompleteAction specifies what to do with the function once built. Invoked when
106     *                            the client calls {@link #build()} or {@link #done()}
107     *
108     * @see ExceptionTransformerBuilder#ExceptionTransformerBuilder()
109     */
110    public ExceptionTransformerBuilder(@Nullable Consumer<Function<Throwable, Throwable>> buildCompleteAction) {
111        this.buildCompleteAction = buildCompleteAction;
112    }
113
114    /**
115     * <p>Creates the configured exception-transformer function and passes it to the
116     * {@code buildCompleteAction} if one was specified at construction time.
117     * Unlike {@link #done()} this method also returns the transformer function.</p>
118     *
119     * <p>Once this method has been called, any call to a method other than {@code #build()}
120     * or {@link #done()} will result in exception.</p>
121     *
122     * <p>Calling this method multiple times has no effect and it will always return the same value.</p>
123     * @return the configured exception-transformer function
124     */
125    public @NotNull Function<Throwable, Throwable> build() {
126        if (!done && buildCompleteAction != null) buildCompleteAction.accept(transformFunction);
127        done = true;
128        return transformFunction;
129    }
130
131    /**
132     * <p>Creates the configured exception-transformer function and passes it to the
133     * {@code buildCompleteAction} if one was specified at construction time.
134     * Unlike {@link #build()} this method does NOT return the transformer function,
135     * but always returns {@code null}, inferring the return type.</p>
136     *
137     * <p>Once this method has been called, any call to a method other than {@link #build()}
138     * or {@code #done()} will result in exception.</p>
139     *
140     * <p>Calling this method multiple times has no further effects.</p>
141     *
142     * @param <T> inferring the return type.
143     *
144     * @return the configured exception-transformer function
145     */
146    public @Nullable <T> T done() {
147        build();
148        return null;
149    }
150
151    /**
152     * If configured, will replace the {@link Throwable#getMessage()} of the thrown
153     * exception with the mapped value.
154     *
155     * @param messageMapper provide a new message for the passed-in exception.
156     * @return the builder instance for chaining
157     * @throws IllegalStateException if a messageMapper has already been specified or
158     *                               the transform function has already been built.
159     */
160    public @NotNull ExceptionTransformerBuilder replaceMessage(@NotNull Function<Throwable, String> messageMapper) {
161        if (done) throw new IllegalStateException("Can not change state once you have called build()!");
162        if (this.messageMapper != null) throw new IllegalStateException("ReplaceMessage mapper already specified");
163        this.messageMapper = messageMapper;
164        return this;
165    }
166
167    /**
168     * Specifies predicate to remove wrapped exceptions. Useful when using multiple-layers,
169     * where each layer wraps and rethrows an exception. If this method is called multiple
170     * times, the specified predicates are OR-ed together.
171     *
172     * @param unwrap if this predicate returns {@code true}, the tested exception will be removed
173     *               from the {@link Throwable#getCause()} nesting.
174     * @return the builder instance for chaining
175     * @throws IllegalStateException if the transform function has already been built.
176     */
177    public @NotNull ExceptionTransformerBuilder unwrapWhen(@NotNull Predicate<Throwable> unwrap) {
178        if (done) throw new IllegalStateException("Can not change state once you have called build()!");
179        this.unwrap = this.unwrap.or(unwrap);
180        return this;
181    }
182
183    /**
184     * Specifies to remove wrapped exceptions of certain classes. Useful when using multiple-layers,
185     * where each layer wraps and rethrows an exception. If this method is called multiple
186     * times, the specified conditions are OR-ed together.
187     *
188     * @param exceptionClasses throwable is removed only if its class matches exactly one
189     *                         of these specifief classes.
190     * @return the builder instance for chaining
191     * @throws IllegalStateException if the transform function has already been built.
192     */
193    @SafeVarargs
194    public final @NotNull ExceptionTransformerBuilder unwrapThese(@NotNull Class<? extends Throwable>... exceptionClasses) {
195        Collection<Class<? extends Throwable>> ecs = Arrays.asList(exceptionClasses);
196        return unwrapWhen(t -> ecs.contains(t.getClass()));
197    }
198
199    /**
200     * Transforms the full stack, allowing to remove frames, add new frames, or putting extra information in
201     * existing frames as follows:
202     * <ul>
203     *     <li>to remove a frame, just set the array cell to {@code null}</li>
204     *     <li>to add a new frame, allocate a bigger array, copy what you need and add your custom frames where needed</li>
205     *     <li>to decorate a frame, instantiate a new frame with the data of existing one and modify what you need</li>
206     * </ul>
207     *
208     * If configured multiple times, the stack mappers are applied in the specified order. That means that some of them
209     * may see {@code null} frames, leftovers from the previous.
210     *
211     * @param stackMapper the mapping function, as described above
212     * @return the builder instance for chaining
213     * @throws IllegalStateException if the transform function has already been built.
214     */
215    public @NotNull ExceptionTransformerBuilder replaceStackTrace(@NotNull Function<StackTraceElement[], @NotNull StackTraceElement[]> stackMapper) {
216        if (done) throw new IllegalStateException("Can not change state once you have called build()!");
217        this.stackMapper = this.stackMapper.andThen(stackMapper);
218        return this;
219    }
220
221    /**
222     * Transforms the stack frames one-by-one, allowing to remove frames, or putting extra information in existing
223     * frames as follows:
224     * <ul>
225     *     <li>to remove a frame, just return {@code null}</li>
226     *     <li>to decorate a frame, return a new frame, copying and modifying the data of existing one as needed</li>
227     *     <li>if you need to add a new frame, use {@link #replaceStackTrace(Function)}</li>
228     * </ul>
229     *
230     * If configured multiple times, the frame mappers are applied in the specified order.
231     *
232     * @param frameMapper the mapping function, as described above
233     * @return the builder instance for chaining
234     * @throws IllegalStateException if the transform function has already been built.
235     */
236    public @NotNull ExceptionTransformerBuilder replaceStackFrame(@NotNull Function<@Nullable StackTraceElement, @Nullable StackTraceElement> frameMapper) {
237        if (done) throw new IllegalStateException("Can not change state once you have called build()!");
238        this.frameMapper = this.frameMapper.andThen(frameMapper);
239        return this;
240    }
241
242    /**
243     * Removes stack frames not matching the predicate.
244     *
245     * If configured multiple times, the frame filters are applied in the specified order.
246     *
247     * @param frameFilter return {@code true} if we want to keep this frame; {@code false} otherwise.
248     * @return the builder instance for chaining
249     * @throws IllegalStateException if the transform function has already been built.
250     */
251    public @NotNull ExceptionTransformerBuilder filterStackFrame(@NotNull Predicate<StackTraceElement> frameFilter) {
252        return replaceStackFrame(frame -> frame!=null && frameFilter.test(frame) ? frame : null);
253    }
254
255    /**
256     * Removes stack frames whose {@link StackTraceElement#getClassName()} matches the specified pattern.
257     *
258     * @param classPattern classes of stack frames to remove.
259     * @return the builder instance for chaining
260     * @throws IllegalStateException if the transform function has already been built.
261     */
262    public @NotNull ExceptionTransformerBuilder filterStackFramesForClass(@NotNull Pattern classPattern) {
263        return filterStackFrame(frame -> !classPattern.matcher(frame.getClassName()).matches());
264    }
265
266    /**
267     * Removes stack frames whose {@link StackTraceElement#getClassName()} matches the specified prefix.
268     *
269     * @param classPrefix prefix of fully qualified classes of stack frames to remove.
270     * @return the builder instance for chaining
271     * @throws IllegalStateException if the transform function has already been built.
272     */
273    public @NotNull ExceptionTransformerBuilder filterStackFramesForClassPrefix(@NotNull String classPrefix) {
274        return filterStackFrame(frame -> !frame.getClassName().startsWith(classPrefix));
275    }
276
277    /**
278     * The actual transformation logic. Because it is an instance method and uses the class fields,
279     * it is fairly easy to debug and inspect.
280     *
281     * @param t the exception we want to transform.
282     * @return the transformed exception.
283     */
284    @Contract(value="null->null;!null->!null", pure=true)
285    @SuppressWarnings("ThrowableResultOfMethodCallIgnored")
286    private @Nullable Throwable transform(@Nullable Throwable t) {
287        if (t == null) return null;
288
289        Throwable cause = t.getCause();
290        if (cause!=null && cause != t) {
291            Throwable transformedCause = transform(cause);
292            Extractors.pokeField(t, Throwable.class, "cause", transformedCause);
293            if (unwrap.test(t)) return transformedCause;
294        }
295
296        @SuppressWarnings("unchecked")
297        List<Throwable> suppressedExceptions = Extractors.peekField(t, Throwable.class, "suppressedExceptions", List.class);
298        for (int i = 0; i < suppressedExceptions.size(); i++) {
299            suppressedExceptions.set(i, transform(suppressedExceptions.get(i)));
300        }
301        if (!suppressedExceptions.isEmpty()) suppressedExceptions.removeIf(Objects::isNull);
302
303        // map stack frames
304        StackTraceElement[] stack = t.getStackTrace();
305
306        // replace elements with null to remove, may return new (larger) array with synthetic frames
307        StackTraceElement[] transformedStack = stackMapper.apply(stack);
308
309        StackTraceElement[] filtered = Arrays.stream(transformedStack)
310                                             .map(frameMapper)
311                                             .filter(Objects::nonNull) // remove any nulls returned by frame mapper
312                                             .toArray(StackTraceElement[]::new);
313
314        Extractors.pokeField(t, Throwable.class, "stackTrace", filtered);
315
316        // map message
317        if (messageMapper!=null) {
318            Extractors.pokeField(t, Throwable.class, "detailMessage", messageMapper.apply(t));
319        }
320
321        return t;
322    }
