031/**
032 * <p><span class="badge green">Entry Point</span> Formats data as text-table, suitable for logging and printing. The recommended way is to
033 * use a builder-based DSL (though you may use the classes directly to achieve advanced scenarios).
034 * All following examples focus on the DSL:</p>
035 *
036 * <pre><code>
037 * String table = TextTable.withColumns("key", "value", "notes")
038 *                         .withData()
039 *                         .row("OP", "Austin Powers", "International Man of Mystery")
040 *                         .row("NP", "Nigel Powers", "NP complete")
041 *                         .buildTable()
042 *                         .format(10, new StringBuilder())
043 *                         .toString();
044 * </code></pre>
045 *
046 * <p>Which will be rendered as:</p>
047 *
048 * <pre><tt>
049 * +-----+---------------+------------------------------+
050 * | key | value         | notes                        |
051 * +-----+---------------+------------------------------+
052 * | OP  | Austin Powers | International Man of Mystery |
053 * | NP  | Nigel Powers  | NP complete                  |
054 * +-----+---------------+------------------------------+
055 * </tt></pre>
056 *
057 * <a>We may as well specify different visual style like this:</a>
058 *
059 * <pre><code>
060 * presetBox(Box.UNICODE_THIN); // will draw the table using nice thin box characters.
061 * </code></pre>
062 *
063 *
064 * <p>Which will change the rendering to a nice box using unicode's pseudo-graphic characters.
065 * (Blame Java's encoding handling for not being able to include it in the Javadoc easily.)</p>
066 *
067 *
068 * <p>Typically, we would want to define the table layout once and use it to format multiple
069 * data sets:</p>
070 *
071 * <pre><code>
072 * TextTable.LayoutBuilder layout = TextTable.withColumns("key", "value", "notes");
073 *
074 * layout.withData()
075 *       .row("OP", "Austin Powers", "International Man of Mystery")
076 *       .row("NP", "Nigel Powers", "NP complete")
077 *       .buildTable()
078 *       .format(10, System.err);
079 *
080 * layout.withData()
081 *       .row("foo", "Fubar", "beyond all repair")
082 *       .row("baz", "Bazqux", "")
083 *       .buildTable()
084 *       .format(10, System.err);
085 * </code></pre>
086 *
087 * <p>Or we may keep the table object around and format it to different outputs:</p>
088 *
089 * <pre><code>
090 * TextTable table = TextTable.withColumns("key", "value")
091 *                            .column("notes") // columns can be specified one-by-one
092 *                            .withData()
093 *                            .rows( // add rows in bulk
094 *                              new String[] { "OP", "Austin Powers", "International Man of Mystery" },
095 *                              new String[] { "NP", "Nigel Powers", "NP complete" },
096 *                            )
097 *                            .row()
098 *                            .buildTable();
099 *
100 * table.format(0, System.err);
101 * table.format(10, System.out);
102 * StringBuilder renderedTable = table.format(10, new StringBuilder());
103 * </code></pre>
104 *
105 * <p>By default each column is mandatory, left-aligned and rendered as string with 1 space
106 * padding. We can tweak all these per column with DSL like this:</p>
107 *
108 * <pre><code>
109 * TextTable.LayoutBuilder layout = TextTable
110 *      .withColumns()
111 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
112 *      .column("value", c -&gt; c.padding = 0) // remove padding
113 *      .column("notes", c -&gt; {
114 *            c.defaultValue="n/a";
115 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
116 *       });
117 * </code></pre>
118 *
119 * <p>If a value is an instance of {@code Supplier} or {@code Callable} (or id the value
120 * is {@code null} and the default value is instance of these interfaces), the functional
121 * interface will be called and the result will be rendered.</p>
122 *
123 * <pre><code>
124 * TextTable.LayoutBuilder layout = TextTable
125 *      .withColumns()
126 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
127 *      .column("value", c -&gt; c.padding = 0) // remove padding
128 *      .column("notes", c -&gt; {
129 *            c.defaultValue="n/a;
130 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
131 *       });
132 * </code></pre>
133 *
134 * <p>A formatter is allowed to change a column's metadata, which will be reset after each rendered value.
135 * Also we can access the columns after they are defined as follows:</p>
136 *
137 * <pre><code>
138 * TextTable.LayoutBuilder layout = TextTable.withColumns("Name", "Age", "Bio");
139 * DecimalFormat df = new DecimalFormat("#,###");
140 * layout.getAllColumns().each {c -&gt; { // for all columns
141 *   c.formatter = value -&gt; {
142 *       if (value instanceof Number) {
143 *           c.alignment = 1; // right-align
144 *           return df.format(((Number) value).longValue());
145 *       }
146 *       return String.valueOf(value).trim(); // will be left-aligned as default
147 *   }
148 *   c.defaultValue = new AtomicInteger::incrementAndGet;
149 * }}
150 * </code></pre>
151 *
152 * <p>There is no synchronization in any of the classes, but both the columns and table are practically
153 * immutable once defined (unless the formatters change the column fields). This means that both tables
154 * and layouts can be shared between threads. The builders on the other side are stateful and they are
155 * best confined to a single thread.</p>
156 */

273    /**
274     * <p>Render the table as text, based on the layout, per-column formatter, alignment and padding; and
275     * the {@code indent} parameter. If {@code pendingAutoformat} field is set to {@code true}, this
276     * method will also adjust the widths of the columns, so that the data will fit in the table.</p>
277     *
278     * @param indent the number of characters to put to the left of the start of the table.
279     * @param out the destination to which to append the text.
280     * @param <F> captures the type of the {@code out} parameter, so we can return it.
281     * @return the {@code out} parameter for chaining.
282     * @throws IOException passing through exceptions from {@link Appendable#append(char)}
283     *
284     * @see #pendingAutoformat
285     */
286    public <F extends Appendable> @NotNull F format(int indent, @NotNull F out) throws IOException {
287        if (pendingAutoformat) {
288            autoformatFromContent();
289            pendingAutoformat = false;
290        }
291
292        StringBuilder indentPad = pad(new StringBuilder(indent), indent, TextTable.indent);
293
294        // outer-frame: top
295        if (outerFrame) {
296            out.append(indentPad);
297            appendFrameHline(out, style.joints(0));
298        }
299
300        // header row
301        out.append(indentPad);
302        appendRow(out, null);
303
304        // inner-frame: header/values separator
305        out.append(indentPad);
306        appendFrameHline(out, style.joints(1));
307
308        // rows
309        int rowNum=0;
310        for (List<?> row : data) {
311            Separator sep = separatorBefore.get(rowNum++);
312            if (sep!=null) {
313                out.append(indentPad);
314                appendFrame(out, sep);
315            }
316            out.append(indentPad);
317            appendRow(out, row);
318        }
319
320        // outer-frame: bottom
321        if (outerFrame) {
322            out.append(indentPad);
323            appendFrameHline(out, style.joints(2));
324        }
325        return out;
326    }

328    /**
329     * Scan the table content and default column parameters, making sure
330     * that each column is at least as wide as the longest rendered value.
331     */
332    @SuppressWarnings("try")
333    private void autoformatFromContent() {
334        for (Column column : columns) {
335            int maxValueLength = data.stream()
336                                     .map(row -> row.get(column.index))
337                                     .map(value -> {
338                                         try (Column.Memento ignored = column.new Memento()) {
339                                             return column.format(value);
340                                         }
341                                     })
342                                     .map(String::length)
343                                     .max(Comparator.naturalOrder())
344                                     .orElse(0);
345
346            column.width = Math.max(maxValueLength, column.width);
347        }
348    }

350    private void appendFrame(@NotNull Appendable out, @NotNull Separator separator) throws IOException {
351        String horizontal = separator.horizontalGlyph != null ? separator.horizontalGlyph : style.horizontal();
352        String[] jointRow = separator.junctionGlyphs != null ? separator.junctionGlyphs : style.joints(2);
353
354
355        int leftBorderSize = outerFrame && jointRow[0] != null ? jointRow[0].length() : 0;
356        int rightBorderSize = outerFrame && jointRow[2] != null ? jointRow[2].length() : 0;
357        int jointSize = jointRow[1] != null ? jointRow[1].length() : 0;
358        int size = columns.stream().mapToInt(x -> x.width + x.padding*2).sum()
359                + leftBorderSize + jointSize * (columns.size()-1) + rightBorderSize;
360        StringBuilder sb = new StringBuilder(size+16);
361        appendHline(sb, jointRow, horizontal);
362        assert sb.length()==size;
363
364        if (separator.name!=null) {
365            int maxTitleSize = Math.max(0, size - 2 * separator.padding);
366            int alignGap = maxTitleSize - separator.name.length();
367            if(alignGap>0) {
368                int start = separator.padding + (int) (alignGap * separator.alignment);
369                sb.replace(start, start+separator.name.length(), separator.name);
370            } else {
371                String title;
372                if (maxTitleSize>9) {
373                    title = separator.name.substring(0, (maxTitleSize-3) / 2);
374                    title += "...";
375                    title += separator.name.substring(separator.name.length()-(maxTitleSize-title.length()));
376                } else {
377                    title = separator.name.substring(0, maxTitleSize);
378                }
379                assert title.length()==maxTitleSize;
380                sb.replace(separator.padding, size - separator.padding, title);
381            }
382            assert sb.length()==size;
383        }
384
385        out.append(sb).append(eol);
386    }

388    /**
389     * Appends formatted text line for horizontal table separator, including line terminator.
390     * Indentation is no appended as we want to reuse the rendering buffer
391     * (see the implementation of {@link #format(int, Appendable)}).
392     * @param out destination for the table row.
393     * @param jointRow array of 3 values for left, middle and center joints.
394     * @throws IOException if the {@param out} throws while appending.
395     * @see Style#joints(int)
396     */
397    private void appendFrameHline(@NotNull Appendable out, String[] jointRow) throws IOException {
398        String horizontal = style.horizontal();
399        if (horizontal==null) return;
400
401        appendHline(out, jointRow, horizontal);
402        out.append(eol);
403    }

405    /**
406     * Appends formatted text line for horizontal table separator, including line terminator.
407     * Indentation is no appended as we want to reuse the rendering buffer
408     * (see the implementation of {@link #format(int, Appendable)}).
409     * @param out destination for the table row.
410     * @param jointRow array of 3 values for left, middle and center joints.
411     * @param horizontal the glyph used to draw the hlines of this table
412     * @throws IOException if the {@param out} throws while appending.
413     * @see Style#joints(int)
414     */
415    private void appendHline(@NotNull Appendable out, String[] jointRow, @Nullable String horizontal) throws IOException {
416        for (int i = 0; i < columns.size(); i++) {
417            Column column = columns.get(i);
418
419            if (outerFrame && jointRow[0]!=null && i == 0) out.append(jointRow[0]);
420            pad(out, column.padding * 2 + column.width, horizontal !=null ? horizontal : " ");
421            if (i >= columns.size() - 1) {
422                String jointGlyph = jointRow[2];
423                if (outerFrame && jointGlyph != null) out.append(jointGlyph);
424            } else {
425                String jointGlyph = jointRow[1];
426                if (jointGlyph!=null) out.append(jointGlyph);
427            }
428        }
429    }
430
431    /**
432     * Appends formatted text line for single table row or header, including padding and
433     * line terminator. Indentation is no appended as we want to reuse the rendering buffer
434     * (see the implementation of {@link #format(int, Appendable)}).
435     * @param out destination for the table row.
436     * @param row the values to append. If {@code null} - append header with the column names.
437     * @throws IOException if the {@param out} throws while appending.
438     */
439    @SuppressWarnings("try")
440    private void appendRow(@NotNull Appendable out, @Nullable List<?> row) throws IOException {
441        if (outerFrame && style.vertical()!=null) out.append(style.vertical());
442        Column lastColumn = columns.get(columns.size() - 1);
443        for (Column column : columns) {
444            try (Column.Memento ignored = column.new Memento()) {
445                String formatted;
446                if (row == null) {
447                    formatted = column.name;
448                } else {
449                    Object rawValue = row.get(column.index);
450                    Object defaultedValue = rawValue == null ? column.defaultValue : rawValue;
451                    formatted = column.format(defaultedValue);
452                }
453
454                if (formatted.length() > column.width) {
455                    throw new IllegalStateException(
456                            "Formatted string '" + formatted + "'::length==" + formatted.length() +
457                                    " > column " + column + "::width==" + column.width);
458                }
459
460                if (formatted.isEmpty()) {
461                    // in case of multi-char padding string, don't restart in the middle
462                    pad(out, column.width + column.padding * 2, padding);
463                } else {
464                    int alignmentGap = column.width - formatted.length();
465                    int leftAlignmentPad = (int) (alignmentGap * column.alignment);
466                    int rightAlignmentPad = alignmentGap - leftAlignmentPad;
467                    pad(out, leftAlignmentPad + column.padding, padding);
468                    out.append(formatted);
469                    pad(out, rightAlignmentPad + column.padding, padding);
470                }
471                String rightBorderGlyph = column.rightBorder==null ? style.vertical() : column.rightBorder;
472                if ((column!=lastColumn || outerFrame) && rightBorderGlyph !=null) out.append(rightBorderGlyph);
473            }
474        }
475        out.append(eol);
476    }

478    /**
479     * <p>Specifies the characters used to draw the lines and corners of a {@link TextTable table}.
480     * As of now, the whole table is drawn with the same set of characters
481     * (we don't allow to customize per row or per cell).</p>
482     * <p>Typically you would want to use one of the {@link Box} values, though
483     * you may provide your own styles too.</p>
484     */
485    public interface Style {
486        /**
487         * Determines the characters used to draw the horizontal lines of a table.
488         * @return horizontal line string. If multiple characters, will be repeated
489         *         and trimmed as required to reach the desired size.
490         *         If {@code null}, horizontal separators will be skipped entirely.
491         * @see TextTable#style
492         * @see #presetBox(Style)
493         */
494        @Contract(pure = true)
495        @Nullable String horizontal();
496
497        /**
498         * Determines the characters used to draw the vertical lines of a table.
499         * @return vertical line string. If multiple characters, the column separators
500         *         will be using the whole string (make sure to have matching T-joints
501         *         and cross-joints).
502         *         If {@code null}, vertical separators will be skipped entirely.
503         * @see TextTable#style
504         * @see #presetBox(Style)
505         */
506        @Contract(pure = true)
507        @Nullable String vertical();
508
509        /**
510         * Determines the characters used to draw the joints and corners of a table.
511         * The joints are defined as a virtual 3x3 array, with values as follows:
512         * <pre>
513         * String[][] joints = {
514         *     { TOP_LEFT_CORNER    , TOP_T_JOINT    , TOP_RIGHT_CORNER    },
515         *     { LEFT_T_JOINT       , CROSS_JOINT    , RIGHT_T_JOINT       },
516         *     { BOTTOM_LEFT_CORNER , BOTTOM_T_JOINT , BOTTOM_RIGHT_CORNER },
517         * }
518         * </pre>
519         * @param row which row of the 9-box to return.
520         * @return array of 3 strings to be used as left corner, center and right-corner for the specified {@code row}.
521         *         See examples in {@link Box}.
522         * @see TextTable#style
523         * @see #presetBox(Style)
524         */
525        @Contract(pure = true)
526        @NotNull String[] joints(int row);
527    }

529    /**
530     * Config object, modelling a table internal horizontal separator
531     */
532    public static class Separator {
533        /**
534         * Text that should be rendered in the separator line.
535         * If you want start/end decorations (i.e. square brackets), make them part of the name.
536         */
537        public final @Nullable String name;
538
539        /**
540         * The glyph used to draw the hlines.
541         * @see Style#horizontal()
542         */
543        public @Nullable String horizontalGlyph;
544
545        /**
546         * The glyphs used to draw left, center and right joints.
547         * @see Style#joints(int)
548         */
549        public @Nullable String[] junctionGlyphs;
550
551        /**
552         * Title alignment - 0 for left; 0.5 for center; 1 for right
553         * @see Column#alignment
554         */
555        public double alignment = 0;
556
557        /**
558         * Title padding - distance from the frame to the title.
559         * @see Column#padding
560         */
561        public int padding = 3;
562
563        private Separator(@Nullable String name, @Nullable String horizontalGlyph, @Nullable String[] junctionGlyphs) {
564            this.name = name;
565            this.horizontalGlyph = horizontalGlyph;
566            this.junctionGlyphs = junctionGlyphs;
567        }
568
569        /**
570         * Factory method fro separator instances.
571         * @param name the title that should be rendered in the separator. If you want start/end
572         *             decorations, make them part of the name.
573         * @param horizontalGlyph the glyph used to draw the hlines.
574         * @param junctionGlyphs same as in {@link Style#joints(int)}
575         * @param config a closure allowing to set alignment, padding, etc.
576         * @return a new instance of {@code Separator}
577         */
578        static Separator of(@Nullable String name, @Nullable String horizontalGlyph, @Nullable String[] junctionGlyphs, @Nullable Consumer<Separator> config) {
579            Separator separator = new Separator(name, horizontalGlyph, junctionGlyphs);
580            if (config!=null) config.accept(separator);
581            return separator;
582        }
583    }

586     * <p>Preset styles, specifying the characters used to draw the lines and corners
587     * of a {@link TextTable table}. As of now, the whole table is drawn with the same
588     * set of characters (we don't allow to customize per row or per cell).</p>
589     * <p>You don't need to use this class if you prefer to provide your own styles.</p>
590     * @see TextTable#presetBox(Style)
591     */

593        /** No decorations - good start if you want to customize */
594        NO_BOX(null, null,
595                new String[]{"NOT_USED", "NOT_USED", "NOT_USED"},
596                new String[]{"NOT_USED", "NOT_USED", "NOT_USED"},
597                new String[]{"NOT_USED", "NOT_USED", "NOT_USED"}),
598
599        ASCII_HORIZONTAL("-", null,
600                new String[]{"-", "-", "-"},
601                new String[]{"-", "-", "-"},
602                new String[]{"-", "-", "-"}),
603        ASCII_HORIZONTAL_DOUBLE("=", null,
604                new String[]{"=", "=", "="},
605                new String[]{"=", "=", "="},
606                new String[]{"=", "=", "="}),
607        /** Plain ASCII - suitable for every log file */
608        ASCII("-", "|",
609                new String[]{"+", "+", "+"},
610                new String[]{"+", "+", "+"},
611                new String[]{"+", "+", "+"}),
612
613        /** Stylish and slim, spruce your stdio with UNICODE_THIN */
614        UNICODE_THIN("\u2500", "\u2502",
615                new String[]{"\u250c", "\u252c", "\u2510"},
616                new String[]{"\u251c", "\u253c", "\u2524"},
617                new String[]{"\u2514", "\u2534", "\u2518"}),
618
619        /** ...but how about drawing rectangles with rounded corners? Can we do that now, too? */
620        UNICODE_THIN_ROUNDED("\u2500", "\u2502",
621                new String[]{"\u256d", "\u252c", "\u256e"},
622                new String[]{"\u251c", "\u253c", "\u2524"},
623                new String[]{"\u2570", "\u2534", "\u256f"}),

628                new String[]{"\u2523", "\u254b", "\u252b"},

668    /**
669     * Defines the formatting attributes for a single column of a {@link TextTable table}.
670     */
671    public static class Column {
672        /**
673         * The column name.
674         */
675        public final @NotNull String name;
676
677        /**
678         * The index of the data row that should be displayed in this column.
679         * When you use a builder, this is managed automatically (auto-increment).
680         */
681        public final int index;
682
683        /**
684         * The column width (measured in characters)
685         */
686        public int width;
687
688        /**
689         * Optionally overrides the right border for the column. Can be changed per row.
690         * If the value is {@code null}, {@link TextTable#style style.vertical()} will be used.
691         */
692        public @Nullable String rightBorder = null;
693
694        /**
695         * Padding to be left between the value and the column separator (measured in characters)
696         */
697        public int padding = 1;
698
699        /**
700         * Alignment of the rendered value (0-left, 1-right, 0.5-center, 0.75-slightly rightish)
701         */
702        public double alignment = 0d;
703
704        /**
705         * Function to convert from row values to string. The formatter may change any of
706         * the mutable attributes of the column, impacting the rendering of the value
707         * (they will be reset for the next row).
708         */
709        public @NotNull Function<@Nullable Object, String> formatter = Column::defaultFormatter;
710
711        /**
712         * Used if the row value is {@code null}.
713         */
714        public @Nullable Object defaultValue;
715
716        /**
717         * <p>Memorizes the mutable fields of its enclosing class and restores them on
718         * {@link #restore()} or {@link #close()}.</p>
719         *
720         * <p>As this class implements {@code Autocloseable}, it can be used in a
721         * try-with-resources construct in RAII manner as follows:</p>
722         *
723         * <pre><code>
724         * \@SuppressWarnings("try") // to prevent the compiler from complaining
725         * void preserveColumnSettings() {
726         *     try(Column.Memento ignore = column.new Memento()) {
727         *         // do things that modify the fields `column`
728         *     }
729         *     // at this point all fields of `column`  will have their values from before the try statement
730         * }
731         * </code></pre>
732         */
733        protected class Memento implements AutoCloseable {
734            private int width = Column.this.width;
735            private int padding = Column.this.padding;
736            private double alignment = Column.this.alignment;
737            private @NotNull Function<@Nullable Object, String> formatter = Column.this.formatter;
738            private @Nullable String rightBorder = Column.this.rightBorder;
739            private @Nullable Object defaultValue = Column.this.defaultValue;
740
741            /**
742             * <p>Restore the mutable fields of the enclosing class to the values captured
743             * at the time of instantiation.</p>
744             *
745             * <p>To be used in a try-with-resources construct in RAII manner as follows:</p>
746             *
747             * <pre><code>
748             * \@SuppressWarnings("try") // to prevent the compiler from complaining
749             * void preserveColumnSettings() {
750             *     try(Column.Memento ignore = column.new Memento()) {
751             *         // do things that modify the fields `column`
752             *     }
753             *     // at this point all fields of `column`  will have their values from before the try statement
754             * }
755             * </code></pre>
756             */
757            @Override public void close() { restore(); }
758
759            /**
760             * Restore the mutable fields of the enclosing class to the values captured
761             * at the time of instantiation.
762             * @see #close()
763             */
764            protected void restore() {
765                Column.this.width = width;
766                Column.this.padding = padding;
767                Column.this.alignment = alignment;
768                Column.this.formatter = formatter;
769                Column.this.defaultValue = defaultValue;
770                Column.this.rightBorder = rightBorder;
771            }
772        }
773
774        /**
775         * Construct a new column with specified name, taking data from the specified index.
776         * @param name the name of this column (will be used in the table header and as key for mapped columns)
777         * @param index the index of the data value within a row
778         */
779        public Column(@NotNull String name, int index) {
780            this.name = name;
781            this.index = index;
782            width = name.length();
783        }
784
785        /**
786         * Convert a data value into string.
787         * @param value the input value (typically {@code row[column.index]})
788         * @return the formatted value (length should be less or equal to
789         *         the {@code column.width-column.padding})
790         */
791        @NotNull @Contract(pure = true)
792        protected String format(@Nullable Object value) {
793            return formatter.apply(value);
794        }
795
796        @Override
797        @Contract(pure = true)
798        public @NotNull String toString() {
799            return index + "-" + name;
800        }
801
802        @SuppressWarnings("unchecked")
803        @Contract(pure = true)
804        private static @NotNull String defaultFormatter(@Nullable Object value) {
805            if (value instanceof Callable) return Exceptions.rethrow((Callable) value).toString();
806            if (value instanceof Supplier) return ((Supplier) value).get().toString();
807            return String.valueOf(value);
808        }

811    /**
812     * <p>Provides a nice and compact DSL to build a set of columns for a {@link TextTable table}.</p>
813     *
814     * <p>A known limitation is that layouts built with this builder will be
815     * isomorphic with the data rows (i.e. you can't have a 3 column layout,
816     * displaying 1st, 15th and 8th value of the data row). While technically
817     * achievable, I don't see the point, so if you need it feel free to extend
818     * this builder or write your own.</p>
819     */
820    public static class LayoutBuilder {
821        private final @NotNull Map<String, Column> columns = new LinkedHashMap<>();
822
823        /**
824         * Adds a column to the right, with the desired name and default attributes,
825         * displaying the next data index.
826         * @param columnName the name of the new column
827         * @return reference to {@code this} for chaining.
828         * @see #column(String, Consumer)
829         */
830        @NotNull
831        public LayoutBuilder column(@NotNull String columnName) {
832            return column(columnName, null);
833        }
834
835        /**
836         * Adds a column to the right, with the desired name and configurable attributes,
837         * displaying the next data index.
838         * @param columnName the name of the new column
839         * @param config a function that can configure the column. If {@code null} or missing,
840         *               the default attributes will be used.
841         * @return reference to {@code this} for chaining.
842         * @see #column(String)
843         */
844        @NotNull
845        public LayoutBuilder column(@NotNull String columnName, @Nullable Consumer<Column> config) {
846            String name = columnName.trim();
847
848            if (columns.containsKey(name)) {
849                throw new IllegalArgumentException("Duplicate column '" + name + "'");
850            }
851
852            Column newColumn = new Column(name, columns.size());
853            columns.put(name, newColumn);
854
855            if (config != null) {
856                config.accept(newColumn);
857            }
858
859            return this;
860        }
861
862        /**
863         * Direct method to access all columns, can be used as lower level API to add and remove
864         * configured columns or to inspect and change their configuration.
865         * @return a map from column name to a column, ordered by their displayed order.
866         */
867        @NotNull @Contract(pure = true)
868        public Map<String, Column> getAllColumns() {
869            return columns;
870        }
871
872        /**
873         * <p>Create a strict data builder with the specified layout, that is
874         * used to build the data rows for the table. Each row should specify
875         * non-null values for all non-defaulted columns, and no extra data.</p>
876         *
877         * <p>The layout of the data builder is decoupled from this builder.
878         * This builder is still valid after creating a layout and can be used
879         * to create multiple layout objects.</p>
880         * @return the data builder.
881         * @see DataBuilder#buildTable()
882         */
883        @NotNull @Contract(pure = true)
884        public DataBuilder withData() {
885            return new DataBuilder(true, new ArrayList<>(columns.values()));
886        }
887
888        /**
889         * <p>Create a lenient data builder with the specified layout, that is
890         * used to build the data rows for the table. Each row should specify
891         * non-null values for all non-defaulted columns, but it may as well
892         * specify extra data.</p>
893         *
894         * <p>The layout of the data builder is decoupled from this builder.
895         * This builder is still valid after creating a layout and can be used
896         * to create multiple layout objects.</p>
897         *
898         * @return the data builder.
899         * @see DataBuilder#buildTable()
900         */
901        @NotNull @Contract(pure = true)
902        public DataBuilder withDataRelaxed() {
903            return new DataBuilder(false, new ArrayList<>(columns.values()));
904        }
905
906        @Override
907        @NotNull @Contract(pure = true)
908        public String toString() {
909            return String.format("TextTable.LayoutBuilder{columns=%s}", columns);
910        }
911    }

913    /**
914     * This builder can be used to build a dataset, that can be used
915     * together with the specified layout (a set of columns) to create
916     * a {@link TextTable table}.
917     */
918    public static class DataBuilder {
919        private final int maxColumnIndex;
920        private final boolean strict;
921        private final @NotNull List<@NotNull Column> columns;
922        private final @NotNull List<@NotNull List<?>> data = new ArrayList<>();
923        private final @NotNull Map<@NotNull Integer, @NotNull Separator> separatorBefore = new TreeMap<>();
924
925        /**
926         * Creates a builder for the specified layout.
927         * @param strict if {@code true}, rows can not specify extra data (unmapped to columns)
928         * @param columns the layout for hte table (defensively copied internally).
929         */
930        public DataBuilder(boolean strict, @NotNull List<@NotNull Column> columns) {
931            // cloning in the LayoutBuilder.withData() as well, but better safe than sorry
932            this.columns = new ArrayList<>(columns);
933            this.strict = strict;
934            this.maxColumnIndex = columns.stream()
935                                         .mapToInt(it -> it.index).max()
936                                         .orElseThrow(()-> new IllegalArgumentException("Empty layout"));
937        }
938
939        /**
940         * Adds a single row to the dataset
941         * @param row the row of values to add
942         * @return reference to {@code this} for chaining.
943         * @throws IllegalArgumentException if the row does not match in the layout
944         */
945        @NotNull
946        public DataBuilder row(@NotNull List<@Nullable ?> row) {
947            return appendRowChecked(row, row);
948        }
949
950        /**
951         * Adds a single row to the dataset
952         * @param row the row of values to add
953         * @return reference to {@code this} for chaining.
954         * @throws IllegalArgumentException if the row does not match in the layout
955         */
956        @NotNull
957        public DataBuilder row(Object... row) { // we want not null row array, but nullable array components
958            return appendRowChecked(Arrays.asList(row), Arrays.deepToString(row));
959        }
960
961        /**
962         * Adds a single row to the dataset
963         * @param row the row of values to add keyed by name
964         * @return reference to {@code this} for chaining.
965         * @throws IllegalArgumentException if the row does not match in the layout
966         */
967        @NotNull
968        public DataBuilder row(@NotNull Map<@NotNull String, @Nullable Object> row) {
969            Object[] rowArray = new Object[maxColumnIndex+1];
970            for (Column column : columns) {
971                if (!row.containsKey(column.name) && column.defaultValue == null) continue;
972                rowArray[column.index] = row.getOrDefault(column.name, column.defaultValue);
973            }
974            return appendRowChecked(Arrays.asList(rowArray), row + "->" + Arrays.toString(rowArray));
975        }
976
977        /**
978         * Adds multiple rows to the dataset
979         * @param rows a collection of rows to add
980         * @return reference to {@code this} for chaining.
981         * @throws IllegalArgumentException if the row does not match in the layout
982         */
983        @NotNull
984        public DataBuilder rows(@NotNull Collection<@NotNull List<@Nullable ?>> rows) {
985            rows.forEach(this::row);
986            return this;
987        }
988
989        /**
990         * Adds multiple rows to the dataset
991         * @param rows a collection of rows to add
992         * @return reference to {@code this} for chaining.
993         * @throws IllegalArgumentException if the row does not match in the layout
994         */
995        @NotNull
996        public DataBuilder rows(@NotNull Object[]... rows) {
997            for (Object[] row : rows) row(row);
998            return this;
999        }
1000
1001        /**
1002         * Add a plain row separator (use table style, no title)
1003         * @return reference to {@code this} for chaining.
1004         */
1005        public @NotNull DataBuilder separator() { return separator(null, null); }
1006
1007        /**
1008         * Add a row separator with optional title using the table inner joints style
1009         * @param name the separator title - it will be left-aligned and padded by default
1010         * @return reference to {@code this} for chaining.
1011         */
1012        public @NotNull DataBuilder separator(@Nullable String name) { return separator(name, null); }
1013
1014        /**
1015         * Add a row separator with no title and optional config closure
1016         * @param config a closure to configure a {@code Separator} object,
1017         *               if {@code null} - use the table inner joints style
1018         * @return reference to {@code this} for chaining.
1019         */
1020        public @NotNull DataBuilder separator(@Nullable Consumer<Separator> config) { return separator(null, config); }
1021
1022        /**
1023         * Add a row separator with optional title and config closure
1024         * @param name the separator title - it will be aligned and padded according to te config
1025         * @param config a closure to configure a {@code Separator} object,
1026         *               if {@code null} - use the table inner joints style
1027         * @return reference to {@code this} for chaining.
1028         */
1029        public @NotNull DataBuilder separator(@Nullable String name, @Nullable Consumer<Separator> config) {
1030            return separator(Separator.of(name, null, null, config));
1031        }
1032
1033        /**
1034         * Add a prepared row separator
1035         * @param separator the configured separator instance. The separator is not copied -
1036         *                  take care if you modify after adding.
1037         * @return reference to {@code this} for chaining.
1038         */
1039        public @NotNull DataBuilder separator(Separator separator) {
1040            separatorBefore.put(data.size(), separator);
1041            return this;
1042        }
1043
1044        private DataBuilder appendRowChecked(List<?> row, @Nullable Object rowDescription) {
1045            if (strict) {
1046                if (row.size() != maxColumnIndex + 1) {
1047                    throw new IllegalArgumentException(String.format(
1048                            "Data doesn't match the columns:%n%d COLUMNS: %s%n%d DATA: %s",
1049                            columns.size(), columns, row.size(), rowDescription
1050                    ));
1051                }
1052            }
1053            for (Column column : columns) {
1054                if (column.defaultValue != null) continue;
1055
1056                if (column.index>=row.size()) {
1057                    throw new IllegalArgumentException(String.format(
1058                            "Column index out of range: %s%n%d COLUMNS: %s%n%d DATA: %s",
1059                            column, columns.size(), columns, row.size(), rowDescription
1060                    ));
1061                }
1062
1063                if (row.get(column.index)==null) {
1064                    throw new IllegalArgumentException(String.format(
1065                            "Non default column is null: %s%n%d COLUMNS: %s%n%d DATA: %s",
1066                            column, columns.size(), columns, row.size(), rowDescription
1067                    ));
1068                }
1069            }
1070            data.add(row);
1071            return this;
1072        }
1073
1074        /**
1075         * Create a text table with the specified layout and data, that can be used to render
1076         * the data in tabular format to any appendable (stream or buffer). Both the columns
1077         * and data are defensively shallow-copied, which means that changes to the columns
1078         * and rows will visible in the table.
1079         * @return the text table.
1080         * @see TextTable#format(int, Appendable)
1081         */
1082        @NotNull @Contract(pure = true)
1083        public TextTable buildTable() {
1084            return new TextTable(new ArrayList<>(columns), new ArrayList<>(data), new TreeMap<>(separatorBefore));
1085        }
1086
1087        @Override
1088        public String toString() {
1089            return String.format(
1090                    "DataBuilder(%d columns, %d rows) { %s }",
1091                    columns.size(), data.size(), columns.stream().map(it -> it.name).collect(Collectors.joining(", "))
1092            );
1093        }
1094    }

1096    /**
1097     * <p>Adds {@code length} characters of padding to the {@code out} parameter.
1098     * The padding is filled by repeating the {@code pad} pattern as needed,
1099     * trimming the last occurrence to size.</p>
1100     * <p>For ecample, {@code pad(out, 8, "abc")} will result in <em>"abcabcab"</em>
1101     * (8 characters) appened to the {@code out} parameter.</p>
1102     * @param out where to put the padding.
1103     * @param length how many characters to pad.
1104     * @param pad what pattern to use for padding.
1105     * @param <T> captures the type of the {@code out} parameter, so we can return it.
1106     * @return the {@code out} parameter for chaining.
1107     */
1108    protected static <T extends Appendable> T pad(@NotNull T out, int length, @NotNull String pad) {
1109        try {
1110            int padLength = pad.length();
1111            for (int i = 0; i < length; /* incremented in loop */) {
1112                int charsToAppend = Math.min(padLength, length - i);
1113                out.append(pad, 0, charsToAppend);
1114                i += charsToAppend;
1115            }
1116            return out;
1117        } catch (IOException e) {
1118            return Exceptions.rethrow(e);
1119        }
1120    }
