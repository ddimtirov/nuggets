035    public static final String VALID_NAME_PATTERN = "\\S(?:.*\\S)?";
036    private static final java.util.regex.Pattern validName = java.util.regex.Pattern.compile(VALID_NAME_PATTERN);
037    private static void checkName(@NotNull String name) {
038        if (!validName.matcher(name).matches()) {
039            throw new IllegalArgumentException("Name should be non-empty and cannot start or end with whitespace!");
040        }
041    }

043    /**
044     * Decorates a supplier with a {@code toString()}, so that it can be printed and logged.
045     * The resulting supplier behaves in any other way like the {@code delegate}.
046     *
047     * @param name the string to be returned on {@code toString()}.
048     *             Needs to start and end with a letter, can not be empty.
049     * @param delegate the supplier to decorate.
050     * @param <T> the generic type of the decorated supplier return value.
051     * @return a supplier that behaves like {@code delegate}, but when printed prints as {@code name}
052     */
053    public static <T> @NotNull Supplier<T> sup(
054            @NotNull @Pattern(VALID_NAME_PATTERN) String name,
055            @NotNull Supplier<T> delegate
056    ) {
057        checkName(name);
058        class NamedSupplier implements Supplier<T> {
059            @Override public T get() { return delegate.get(); }
060            @Override public String toString() { return name; }
061        }
062        return new NamedSupplier();
063    }

202    /**
203     * <p>Tries multiple functions in the specified order and returns the first result that passes a predicate,
204     * optionally ignoring exceptions. Throws {@link DispatchException} if none of the fallback functions returned a
205     * good result, wrapping all fallback failures as {@link Throwable#getSuppressed() suppressed}.
206     * If {@code failFast} was specified and any of the fallbacks throws an exception, it is immediately rethrown.</p>
207     *
208     * <p><em><strong>Design notes:</strong> Ideally we would like this combinator to be a macro method, applicable to
209     * any functional interface that returns a value (non-void). We have decided to dispatch {@link Function Functions},
210     * as these show up as the most common functional interfaces in the wild.
211     *
212     * Alternative design is to dispatch {@link Supplier Suppliers}, and force the caller to tunnel the functional
213     * parameters using closures, but that makes the more common case uglier.
214     *
215     * Going to the other extreme, we could have dispatched {@link BiFunction BiFunctions}, and accommodated the
216     * {@code Function} use-case by requiring the client to adapt the result using partial application.</em></p>
217     *
218     * @param failFast if {@code true}, as soon as a function throws exception, we'll rethrow it and skip trying the
219     *                 remaining fallback functions. If {@code false}, when a function throws an exception we just
220     *                 continue with the remaining fallback functions. If we try all exceptions and didn't find a result
221     *                 all these exceptions will be rethrown as {@link Throwable#suppressedExceptions}
222     *
223     * @param approveResult a predicate that decides whether we shall return the result of a function or try the next
224     *                      one. If all functions are tried and none matched the predicate, this method will throw
225     *                      {@link DispatchException}
226     *
227     * @param functions a functions to be tried one after another until one returns a result that matches the
228     *                  {@code approveResult} predicate. Exceptions are swallowed if {@code failFast} is {@code true}
229     *                  and rethrown if {@code false}.
230     *
231     * @param <T> the function argument generic type
232     * @param <R> the function return generic type
233     *
234     * @return a function that dispatches through the {@code fallbacks}, based on the strategy determined by
235     *         {@code failFast} and {@code approveResult}
236     *
237     * @throws DispatchException if {@code failFast==false} or none of the results matched the predicate.
238     *         If {@code failFast==true} we would rethrow immediately if any of the {@code fallbacks} throws exception.
239     */
240    @SafeVarargs
241    public static <T, R> @NotNull Function<T, R> fallback(boolean failFast, @Nullable Predicate<@Nullable R> approveResult, @NotNull Function<@Nullable T, @Nullable R>... functions) {
242        if (functions.length==0) throw new IllegalArgumentException("Need to provide at least one function!");
243
244        if ((failFast || functions.length==1) && approveResult==null) {
245            // same as just rinning the first function: if it succeeds - return the result; if it fails - rethrow
246            return functions[0];
247        }
248
249        return t -> {
250            List<Throwable> suppressed = new ArrayList<>(functions.length);
251            for (Function<T, R> function : functions) {
252                try {
253                    R result = function.apply(t);
254                    if (approveResult != null && !approveResult.test(result)) {
255                        suppressed.add(new DispatchException.InvalidResult(result));
256                    } else {
257                        return result;
258                    }
259                } catch (Throwable e) {
260                    if (failFast) {
261                        return Exceptions.rethrow(e);
262                    }
263                    suppressed.add(e);
264                }
265            }
266
267            String verb = approveResult == null
268                    ? "successfully run any of"
269                    : failFast ? "approve result" : "approve successful result of";
270
271            return Exceptions.rethrow(new DispatchException("failed to " + verb + " " + functions.length + " fallback fuctions!", suppressed));
272        };
273    }

275    /**
276     * Creates an identity function, intercepting the function
277     * argument for inspection or manipulation. It can be used with
278     * {@link Function#compose(Function)} to intercept the argument
279     * of a function and with {@link Function#andThen(Function)} to
280     * intercept the result.
281     *
282     * @param c consumer operating on the function parameter
283     * @param <T> generic type for inferrence
284     * @return an identity function calling {@code c} on its argument/result
285     */
286    public static <T> @NotNull UnaryOperator<T> tap(@NotNull Consumer<T> c) {
287        return it -> { c.accept(it); return it; };
288    }

292     * the predicate argument for inspection or manipulation. It can

320    /**
321     * Wraps a supplier, without changing its logic, intercepting
322     * the supplier result for inspection or manipulation.
323     *
324     * @param s wrapped supplier
325     * @param c consumer operating on the supplier results
326     * @param <T> generic type for inferrence
327     * @return a supplier delegating to {@code s}, calling {@code c} on its result
328     */
329    public static <T> @NotNull Supplier<T> snoop(@NotNull Supplier<T> s, @NotNull Consumer<T> c) {
330        return () -> { T val = s.get(); c.accept(val); return val; };
331    }

334     * Wraps a predicate, without changing its logic, intercepting
335     * the predicate result for inspection.
336     *
337     * @param p wrapped predicate implementing some useful function

346    /**
347     * <p>An expressive shortcut for using a void function in context that requires to return something.</p>
348     *
349     * <p>Example:</p>
350     *
351     * <pre><code>
352     * return retval(foo, foo::validate)
353     * </code></pre>
354     *
355     * <p>This is just an alias for {@link Exceptions#rethrow(ThrowingRunnable, Object)} that places
356     * semantic emphasis on the returned value.</p>
357     *
358     * @param retval the value to return.
359     * @param r the void operation to perform.
360     * @param <T> generic param used for local type inference.
361     * @return {@code retval} if {@code r} did not throw exception.
362     */
363    @Contract("null,_->null;!null,_->!null")
364    public static <T> @Nullable T ret(@Nullable T retval, @NotNull ThrowingRunnable r) {
365        return Exceptions.rethrow(r, retval);
366    }
367
368    /**
369     * <p>An expressive shortcut for using a void function in context that requires to return something.</p>
370     *
371     * <p>Example:</p>
372     *
373     * <pre><code>
374     * qwe = fallback(true, null, retnul(foo::bar), foo::baz, ret("boo", foo::qux))
375     * </code></pre>
376     *
377     * <p>This is just another alias for {@link Exceptions#rethrow(ThrowingRunnable, Object)}.
378     * Unlike {@link #ret(Object, ThrowingRunnable) ret()}, that places semantic emphasis on the
379     * returned value, {@code retnul()} signifies that the return value is unimportant and we are
380     * using it purely for syntactic needs.</p>
381     *
382     * @param r the void operation to perform.
383     * @return {@code null} if {@code r} did not throw exception.
384     */
385    @Contract("_->null")
386    public static <T> @Nullable T retnul(@NotNull ThrowingRunnable r) {
387        return Exceptions.rethrow(r, null);
388    }
