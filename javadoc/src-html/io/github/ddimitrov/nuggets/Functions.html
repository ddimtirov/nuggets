034    public static final String VALID_NAME_PATTERN = "\\S(?:.*\\S)?";
035    private static final java.util.regex.Pattern validName = java.util.regex.Pattern.compile(VALID_NAME_PATTERN);
036    private static void checkName(@NotNull String name) {
037        if (!validName.matcher(name).matches()) {
038            throw new IllegalArgumentException("Name should be non-empty and cannot start or end with whitespace!");
039        }
040    }

042    /**
043     * Decorates a supplier with a {@code toString()}, so that it can be printed and logged.
044     * The resulting supplier behaves in any other way like the {@code delegate}.
045     *
046     * @param name the string to be returned on {@code toString()}.
047     *             Needs to start and end with a letter, can not be empty.
048     * @param delegate the supplier to decorate.
049     * @param <T> the generic type of the decorated supplier return value.
050     * @return a supplier that behaves like {@code delegate}, but when printed prints as {@code name}
051     */
052    public static <T> @NotNull Supplier<T> sup(
053            @NotNull @Pattern(VALID_NAME_PATTERN) String name,
054            @NotNull Supplier<T> delegate
055    ) {
056        checkName(name);
057        class NamedSupplier implements Supplier<T> {
058            @Override public T get() { return delegate.get(); }
059            @Override public String toString() { return name; }
060        }
061        return new NamedSupplier();
062    }

201    /**
202     * <p>Tries multiple functions in the specified order and returns the first result that passes a predicate,
203     * optionally ignoring exceptions. Throws {@link DispatchException} if none of the fallback functions returned a
204     * good result, wrapping all fallback failures as {@link Throwable#getSuppressed() suppressed}.
205     * If {@code failFast} was specified and any of the fallbacks throws an exception, it is immediately rethrown.</p>
206     *
207     * <p><em><strong>Design notes:</strong> Ideally we would like this combinator to be a macro method, applicable to
208     * any functional interface that returns a value (non-void). We have decided to dispatch {@link Function Functions},
209     * as these show up as the most common functional interfaces in the wild.
210     *
211     * Alternative design is to dispatch {@link Supplier Suppliers}, and force the caller to tunnel the functional
212     * parameters using closures, but that makes the more common case uglier.
213     *
214     * Going to the other extreme, we could have dispatched {@link BiFunction BiFunctions}, and accommodated the
215     * {@code Function} use-case by requiring the client to adapt the result using partial application.</em></p>
216     *
217     * @param failFast if {@code true}, as soon as a function throws exception, we'll rethrow it and skip trying the
218     *                 remaining fallback functions. If {@code false}, when a function throws an exception we just
219     *                 continue with the remaining fallback functions. If we try all exceptions and didn't find a result
220     *                 all these exceptions will be rethrown as {@link Throwable#suppressedExceptions}
221     *
222     * @param approveResult a predicate that decides whether we shall return the result of a function or try the next
223     *                      one. If all functions are tried and none matched the predicate, this method will throw
224     *                      {@link DispatchException}
225     *
226     * @param functions a functions to be tried one after another until one returns a result that matches the
227     *                  {@code approveResult} predicate. Exceptions are swallowed if {@code failFast} is {@code true}
228     *                  and rethrown if {@code false}.
229     *
230     * @param <T> the function argument generic type
231     * @param <R> the function return generic type
232     *
233     * @return a function that dispatches through the {@code fallbacks}, based on the strategy determined by
234     *         {@code failFast} and {@code approveResult}
235     *
236     * @throws DispatchException if {@code failFast==false} or none of the results matched the predicate.
237     *         If {@code failFast==true} we would rethrow immediately if any of the {@code fallbacks} throws exception.
238     */
239    @SafeVarargs
240    public static <T, R> @NotNull Function<T, R> fallback(boolean failFast, @Nullable Predicate<@Nullable R> approveResult, @NotNull Function<@Nullable T, @Nullable R>... functions) {
241        if (functions.length==0) throw new IllegalArgumentException("Need to provide at least one function!");
242
243        if ((failFast || functions.length==1) && approveResult==null) {
244            // same as just rinning the first function: if it succeeds - return the result; if it fails - rethrow
245            return functions[0];
246        }
247
248        return t -> {
249            List<Throwable> suppressed = new ArrayList<>(functions.length);
250            for (Function<T, R> function : functions) {
251                try {
252                    R result = function.apply(t);
253                    if (approveResult != null && !approveResult.test(result)) {
254                        suppressed.add(new DispatchException.InvalidResult(result));
255                    } else {
256                        return result;
257                    }
258                } catch (Throwable e) {
259                    if (failFast) {
260                        return Exceptions.rethrow(e);
261                    }
262                    suppressed.add(e);
263                }
264            }
265
266            String verb = approveResult == null
267                    ? "successfully run any of"
268                    : failFast ? "approve result" : "approve successful result of";
269
270            return Exceptions.rethrow(new DispatchException("failed to " + verb + " " + functions.length + " fallback fuctions!", suppressed));
271        };
272    }

274    /**
275     * Creates an identity function, intercepting the function
276     * argument for inspection or manipulation. It can be used with
277     * {@link Function#compose(Function)} to intercept the argument
278     * of a function and with {@link Function#andThen(Function)} to
279     * intercept the result.
280     *
281     * @param c consumer operating on the function parameter
282     * @param <T> generic type for inferrence
283     * @return an identity function calling {@code c} on its argument/result
284     */
285    public static <T> @NotNull UnaryOperator<T> tap(@NotNull Consumer<T> c) {
286        return it -> { c.accept(it); return it; };
287    }

289    /**
290     * Creates a fixed value {@code true} predicate, intercepting
291     * the predicate argument for inspection or manipulation. It can
292     * be used with {@link Predicate#and(Predicate)} - as first in
293     * the chain if we care about all arguments or as second if we
294     * only care about the succeeded.
295     *
296     * @param c consumer operating on the predicate parameter
297     * @param <T> generic type for inferrence
298     * @return a {@code true} predicate calling {@code c} on its argument/result
299     */
300    public static <T> @NotNull Predicate<T> yes(@NotNull Consumer<T> c) {
301        return it -> { c.accept(it); return true; };
302    }

319    /**
320     * Wraps a supplier, without changing its logic, intercepting
321     * the supplier result for inspection or manipulation.
322     *
323     * @param s wrapped supplier
324     * @param c consumer operating on the supplier results
325     * @param <T> generic type for inferrence
326     * @return a supplier delegating to {@code s}, calling {@code c} on its result
327     */
328    public static <T> @NotNull Supplier<T> snoop(@NotNull Supplier<T> s, @NotNull Consumer<T> c) {
329        return () -> { T val = s.get(); c.accept(val); return val; };
330    }

332    /**
333     * Wraps a predicate, without changing its logic, intercepting
334     * the predicate result for inspection.
335     *
336     * @param p wrapped predicate implementing some useful function
337     * @param c consumer operating on the predicate parameter
338     * @param <T> generic type for inferrence
339     * @return a predicate delegating to {@code p}, calling {@code c} on its result
340     */
341    public static <T> Predicate<T> snoop(Predicate<T> p, Consumer<Boolean> c) {
342        return it -> { boolean val = p.test(it); c.accept(val); return val; };
343    }
