017package io.github.ddimitrov.nuggets;
018
019import org.jetbrains.annotations.NotNull;
020import org.jetbrains.annotations.Nullable;
021
022import java.util.ArrayList;
023import java.util.List;
024import java.util.function.BiFunction;
025import java.util.function.Function;
026import java.util.function.Predicate;
027import java.util.function.Supplier;
028
029/**
030 * All methods are threadsafe.
031 */
032public final class Functions {
033    private Functions() {}
034
035    /**
036     * <p>Tries multiple functions in the specified order and returns the first result that passes a predicate,
037     * optionally ignoring exceptions. Throws {@link DispatchException} if none of the fallback functions returned a
038     * good result, wrapping all fallback failures as {@link Throwable#getSuppressed() suppressed}.
039     * If {@code failFast} was specified and any of the fallbacks throws an exception, it is immediately rethrown.</p>
040     *
041     * <p><em><strong>Design notes:</strong> Ideally we would like this combinator to be a macro method, applicable to
042     * any functional interface that returns a value (non-void). We have decided to dispatch {@link Function Functions},
043     * as these show up as the most common functional interfaces in the wild.
044     *
045     * Alternative design is to dispatch {@link Supplier Suppliers}, and force the caller to tunnel the functional
046     * parameters using closures, but that makes the more common case uglier.
047     *
048     * Going to the other extreme, we could have dispatched {@link BiFunction BiFunctions}, and accommodated the
049     * {@code Function} use-case by requiring the client to adapt the result using partial application.</em></p>
050     *
051     * @param failFast if {@code true}, as soon as a function throws exception, we'll rethrow it and skip trying the
052     *                 remaining fallback functions. If {@code false}, when a function throws an exception we just
053     *                 continue with the remaining fallback functions. If we try all exceptions and didn't find a result
054     *                 all these exceptions will be rethrown as {@link Throwable#suppressedExceptions}
055     *
056     * @param approveResult a predicate that decides whether we shall return the result of a function or try the next
057     *                      one. If all functions are tried and none matched the predicate, this method will throw
058     *                      {@link DispatchException}
059     *
060     * @param functions a functions to be tried one after another until one returns a result that matches the
061     *                  {@code approveResult} predicate. Exceptions are swallowed if {@code failFast} is {@code true}
062     *                  and rethrown if {@code false}.
063     *
064     * @param <T> the function argument generic type
065     * @param <R> the function return generic type
066     *
067     * @return a function that dispatches through the {@code fallbacks}, based on the strategy determined by
068     *         {@code failFast} and {@code approveResult}
069     *
070     * @throws DispatchException if {@code failFast==false} or none of the results matched the predicate.
071     *         If {@code failFast==true} we would rethrow immediately if any of the {@code fallbacks} throws exception.
072     */
073    @SafeVarargs
074    public static <T, R> @NotNull Function<T, R> fallback(boolean failFast, @Nullable Predicate<R> approveResult, @NotNull Function<T, R>... functions) {
075        if (functions.length==0) throw new IllegalArgumentException("Need to provide at least one function!");
076
077        if ((failFast || functions.length==1) && approveResult==null) {
078            // same as just rinning the first function: if it succeeds - return the result; if it fails - rethrow
079            return functions[0];
080        }
081
082        return t -> {
083            List<Throwable> suppressed = new ArrayList<>(functions.length);
084            for (Function<T, R> function : functions) {
085                try {
086                    R result = function.apply(t);
087                    if (approveResult != null && !approveResult.test(result)) {
088                        suppressed.add(new DispatchException.InvalidResult(result));
089                    } else {
090                        return result;
091                    }
092                } catch (Throwable e) {
093                    if (failFast) {
094                        return Exceptions.rethrow(e);
095                    }
096                    suppressed.add(e);
097                }
098            }
099
100            String verb = approveResult == null
101                    ? "successfully run any of"
102                    : failFast ? "approve result" : "approve successful result of";
103
104            return Exceptions.rethrow(new DispatchException("failed to " + verb + " " + functions.length + " fallback fuctions!", suppressed));
105        };
106    }
107}
