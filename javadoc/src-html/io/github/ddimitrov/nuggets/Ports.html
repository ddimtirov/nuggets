034/**
035 * <p><span class="badge green">Entry Point</span> Provides utilities for
036 * ad-hoc coordination of port usage for distributed systems in tests and
037 * lab environments.</p>
038 *
039 * <p> For example, when you do integration testing for a distributed system,
040 * and you want to run a couple of tests in parallel you need to make sure that
041 * none of the test environments shares ports and file paths with the others.</p>
042 *
043 * <p>To achieve that, you can use the {@code Ports} class to allocate the ports
044 * and tweak the app configuration in application-specific manner.
045 * See the user manual for details and more use-cases.</p>
046 *
047 * <p>This class has explicit lifecycle as follows:</p>
048 * <pre><code>
049 *     Ports p = new Ports(...);
050 *     // CONFIGURE: call withExporter(...), reservePort(id[, offset]), withPorts(...) as needed
051 *     p.freeze(...);
052 *     // USE: call getBasePort() and port(id)
053 *     p.close();
054 *     // DEAD: any method call would result in exception
055 * </code></pre>
056 */
057public class Ports implements AutoCloseable {
058
059    private final List<String> claimedOffsets = Collections.synchronizedList(new ArrayList<>());
060    private final Set<String> dynamicOffsets = new HashSet<>();
061    private final Set<Exporter> exporters = new HashSet<>();
062    private final @NotNull Registrar registrar;
063    private final @NotNull Supplier<Integer> dynamicPortFinder;
064    private volatile int basePort = -1;
065    private volatile boolean closed;
066
067    /**
068     * <p>Creates new ports allocator with the specified registrar
069     * and default dynamic port allocation strategy.</p>
070     *
071     * <p>The default port allocation strategy assigns offsets to the
072     * dynamic port ID's starting from the base port of the block and
073     * skipping all explicitly assigned offsets.</p>
074     *
075     * @param registrar defines a strategy for allocating a port block.
076     */
077    public Ports(@NotNull Registrar registrar) {
078        this(registrar, null);
079    }
080
081    /**
082     * <p>Creates new ports allocator with the specified registrar
083     * and default dynamic port allocation strategy.</p>
084     *
085     * @param registrar defines a strategy for allocating a port block.
086     * @param dynamicPortFinder defines a strategy for dynamically allocating
087     *        ports within a block. If {@code null}, a default strategy assigns
088     *        offsets to the dynamic port ID's starting from the {@link #basePort base port}
089     *        of the block and skipping all explicitly assigned offsets.
090     */
091    public Ports(@NotNull Registrar registrar, @Nullable Supplier<Integer> dynamicPortFinder) {
092        this.registrar = registrar;
093        this.dynamicPortFinder = dynamicPortFinder!=null
094                ? dynamicPortFinder
095                : () -> {
096                    int gapIdx = claimedOffsets.indexOf(null);
097                    return gapIdx>=0 ? gapIdx : claimedOffsets.size();
098                };
099    }

101    /**
102     * <p>Registers an {@link Exporter} that would be used to publish/validate
103     * the allocated ports. Can throw {@link PortVetoException} to reject a
104     * port and force the port allocator to use another block</p>
105     *
106     * <p>If more than one exporter is registered, exporters will be called in
107     * their registration order.</p>
108     *
109     *
110     * <p>If the implementation of this interface also implements
111     * {@link AutoCloseable}, the {@code close()} method will be called after
112     * all offsets have been exported and successfully validated, after
113     * the allocation is confirmed successful.</p>
114     *
115     * @param exporter a callback to publish or validate an allocated port.
116     * @return {@code this} instance for chaining.
117     *
118     * @see Exporter#batching(Consumer)
119     */
120    public @NotNull Ports withExporter(@NotNull Exporter exporter) {
121        if (basePort>=0) throw new IllegalStateException("Already frozen!");
122        if (closed) throw new IllegalStateException("Already closed!");
123        exporters.add(exporter);
124        return this;
125    }

127    /**
128     * Registers a port ID at desired offset within the block.
129     * Should be called as many times as needed before {@link #freeze(int)}.
130     *
131     * @param id the ID of the port we are registering.
132     *           Should be unique.
133     * @param offset the desired offset for the finally allocated port.
134     *               Should be unique.
135     *
136     * @return {@code this} instance for chaining.
137     *
138     * @throws IllegalArgumentException if any of the arguments is not unique
139     *         within the scope of {@code this} instance.
140     * @throws IllegalArgumentException if this port allocator has already been
141     *         {@link #freeze(int) frozen} or {@link #close() closed}
142     *
143     * @see #reservePort(String)
144     * @see #withPorts(int, Consumer)
145     */
146    public @NotNull Ports reservePort(@NotNull String id, int offset) {
147        if (basePort>=0) throw new IllegalStateException("Port allocation already finalized!");
148        if (closed) throw new IllegalStateException("Already closed!");
149
150        if (Exporter.BASE_PORT_ID.equals(id)) throw new IllegalArgumentException("Port ID can not be empty");
151        if (dynamicOffsets.contains(id)) throw new IllegalArgumentException("Port ID '" + id + "' already registered as dynamic (requested " + offset + ")");
152
153        int pos = claimedOffsets.indexOf(id);
154        if (pos>=0 && pos!=offset) throw new IllegalArgumentException("Port ID '" + id + "' already registered at offset: " + pos + " (requested " + offset + ")");
155
156        while (offset>=claimedOffsets.size()) claimedOffsets.add(null); // pad the list so we don't get an IOBE
157        String old = claimedOffsets.set(offset, requireNonNull(id));
158
159        if (old!=null && !old.equals(id)) {
160            claimedOffsets.set(offset, old); // EAFP to avoid TOCTOU issues
161            throw new IllegalArgumentException("Clashing port reservations for offset " + offset + ": old='" + old + "', new='" + id + "'");
162        }
163        return this;
164    }
165
166    /**
167     * Registers a port ID at automatically chosen offset within the block.
168     * Should be called as many times as needed before {@link #freeze(int)}.
169     *
170     * @param id the ID of the port we are registering. Should be unique.
171     *
172     * @return {@code this} instance for chaining.
173     *
174     * @throws IllegalArgumentException if any of the ID is not unique
175     *         within the scope of {@code this} instance.
176     * @throws IllegalArgumentException if this port allocator has already been
177     *         {@link #freeze(int) frozen} or {@link #close() closed}
178     *
179     * @see #reservePort(String, int)
180     * @see #withPorts(int, Consumer)
181     */
182    public @NotNull Ports reservePort(@NotNull String id) {
183        if (basePort>=0) throw new IllegalStateException("Port allocation already finalized!");
184        if (closed) throw new IllegalStateException("Already closed!");
185
186        if (Exporter.BASE_PORT_ID.equals(id)) throw new IllegalArgumentException("Port ID can not be empty");
187
188        int pos = claimedOffsets.indexOf(id);
189        if (pos>=0) throw new IllegalArgumentException("Port ID '" + id + "' already registered at offset " + pos + " (requested dynamic offset)");
190
191        dynamicOffsets.add(id);
192        return this;
193    }
194
195    /**
196     * Configures port IDs and freezes this allocator in one step. Uses a builder based DSL (see example).
197     * Should be called only once, IMPORTANT: internally it calls {@code freeze(basePortHint)}.
198     *
199     * <pre><code>
200     * p.withPorts(5000, register -&gt; {
201     *     register.id("foo");
202     *     register.id("bar").offset(1);
203     *     register.id("baz").offset(2);
204     *     register.id("qux");
205     * });
206     * assert p.getBasePort()==5000; // p is ready to use
207     * </code></pre>
208     *
209     * @param spec a closure configuring the builder (see example above).
210     * @param basePortHint the desired base port - the actual base port may
211     *                     differ if there is a conflict.
212     *
213     * @return {@code this} instance in {@link #freeze(int) frozen} state.
214     *
215     * @throws IllegalArgumentException if any of the ID's and offsets are conflicting.
216     * @throws IllegalArgumentException if this port allocator has already been
217     *         {@link #freeze(int) frozen} or {@link #close() closed}
218     *
219     * @see #reservePort(String)
220     * @see #reservePort(String, int)
221     * @see #freeze(int)
222     */
223    public @NotNull Ports withPorts(int basePortHint, @NotNull Consumer<SpecIdBuilder> spec) {
224        PortsSpecBuilder builder = new PortsSpecBuilder();
225        spec.accept(builder);
226        builder.flush();
227        return freeze(basePortHint);
228    }
229
230    /**
231     * Uses the {@code registrar} and {@code dynamicPortFinder} to allocate
232     * actual ports for each registered {@code portId} and transitions the
233     * internal state to USE.
234     * @param basePortHint the desired base port - the actual base port may
235     *                     differ if there is a conflict.
236     * @return this instance, on which now we can call {@link #port(String)}
237     *         and {@link #getBasePort()}
238     */
239    public @NotNull Ports freeze(int basePortHint) {
240        if (closed) throw new IllegalStateException("Already closed!");
241        if (this.basePort>=0) throw new IllegalStateException("Port range already decided!");
242
243        try {
244            int allocatedBasePort = registrar.lock(basePortHint);
245            if (allocatedBasePort<=0) {
246                throw new IllegalStateException("Failed to reserve a port range!");
247            }
248
249            for (String dynamicId : dynamicOffsets) {
250                int freePort = dynamicPortFinder.get();
251
252                while (freePort>=claimedOffsets.size()) claimedOffsets.add(null); // pad the list so we don't get an IOBE
253                String old = claimedOffsets.set(freePort, dynamicId);
254                if (old!=null) {
255                    claimedOffsets.set(freePort, old); // undo - better undo than check to avoid TOCTOU errors
256                    throw new IllegalArgumentException("The dynamic port finder caused conflict! Clashing port reservations for offset " + freePort + ": old='" + old + "', new='" + dynamicId + "'");
257                }
258            }
259
260            for (Exporter exporter : exporters) {
261                exporter.export(Exporter.BASE_PORT_ID, allocatedBasePort);
262                for (int portOffset = 0; portOffset < claimedOffsets.size(); portOffset++) {
263                    String id = claimedOffsets.get(portOffset);
264                    if (id!=null) {
265                        exporter.export(id, allocatedBasePort+portOffset);
266                    }
267                }
268            }
269            exporters.stream().filter(it -> it instanceof AutoCloseable)
270                     .forEach(it -> rethrow(((AutoCloseable) it)::close));
271            this.basePort = allocatedBasePort;
272            return this;
273        } catch (PortVetoException e) {
274            return freeze(e.port + 1);
275        }
276    }

278    /**
279     * The lowest port of the allocated block or {@code IllegalStateException}
280     * if {@link #freeze(int)} has not been called yet.
281     *
282     * @return the lowest port of the allocated block.
283     */
284    public int getBasePort() {
285        if (closed) throw new IllegalStateException("Already closed!");
286        if (basePort==-1) throw new IllegalStateException("Port range not allocated yet!");
287        return basePort;
288    }

290    /**
291     * Look up a port for ID.
292     * @param id the port ID, which we shall look up.
293     * @return the port corresponding to the {@code id}
294     */
295    public int port(@NotNull String id) {
296        if (basePort<=0) throw new IllegalStateException("Port allocation not finished yet! Perhaps somebody forgot to call freeze()?");
297        if (closed) throw new IllegalStateException("Already closed!");
298
299        int offset = claimedOffsets.indexOf(id);
300        return offset>=0
301                ? basePort + offset
302                : rethrow(new NoSuchElementException("No registered port for id: '" + id + "'"));
303    }

305    /**
306     * <p>Releases all resources and transitions the internal state, so that
307     * a call to any methods other than {@code close()} will result in
308     * {@code IllegalStateException}.</p>
309     *
310     * <p>This method is idempotent.</p>
311     */
312    @Override
313    public void close() {
314        if (basePort>=0) registrar.close();
315        basePort = 0;
316        closed = true;
317    }

329    /** Builder interface used to provide constrained, typesafe DSL. */
330    public interface SpecIdBuilder {
331
332        /**
333         * Specify a port ID. If followed by a call to
334         * {@link SpecOffsetBuilder#offset(int)},
335         * then this ID would be with explicit offset,
336         * otherwise it would be dynamic.
337         *
338         * @param id the port ID
339         * @return {@code this} instance for chaining.
340         */
341        @NotNull Ports.SpecOffsetBuilder id(@NotNull String id);
342    }

344    /**
345     * Typesafe builder implementation, used by extensions for specifying ports and offsets.
346     */
347    public class PortsSpecBuilder implements SpecIdBuilder, SpecOffsetBuilder {
348        private String id;
349        private Integer offset;
350
351        @Override
352        public @NotNull Ports.SpecOffsetBuilder id(@NotNull String id) {
353            flush();
354            this.id = id;
355            return this;
356        }
357
358        @Override
359        public void offset(int offset) {
360            this.offset = offset;
361            flush();
362        }
363
364        /**
365         * Needs to be called after finishing building the configuration,
366         * or the last configured port ID may be lost.
367         */
368        public void flush() {
369            if (offset != null && id == null) {
370                throw new IllegalArgumentException("Ambiguous port offset in spec: " + offset);
371            }
372            if (id == null) return;
373            if (offset == null) {
374                reservePort(id);
375            } else {
376                reservePort(id, offset);
377            }
378            offset = null;
379            id = null;
380        }
381    }

383    /**
384     * <p>A hook notified about allocated ports, and optionally validating them.
385     * Can be used to update configs, set system properties, validate port
386     * availability, etc.</p>
387     *
388     * <p>If the implementation of this interface also implements
389     * {@link AutoCloseable}, the {@code close()} method will be called after
390     * all offsets have been exported and successfully validated, after
391     * the allocation is confirmed successful.</p>
392     */
393    @FunctionalInterface
394    public interface Exporter {
395        /**
396         * The base port will be passed with empty-string ID.
397         * It is easier to check for empty string, but if you want to be explicit, you may use this constant.
398         */
399        String BASE_PORT_ID = "";
400
401        /**
402         * Called for each allocated port in the range, as well as for the base port.
403         * Note that in event of {@code PortVetoException}, you would receive the
404         * base port and port ID's more than once - the last value is the actual.
405         *
406         * @param id the port ID (empty string means base port)
407         * @param port the allocated port
408         * @throws PortVetoException to indicate that this block is bad and
409         *         request reallocation of different block.
410         */
411        void export(@NotNull String id, int port) throws PortVetoException;
412
413        /**
414         * <p>A convenient way to receive the allocated ports (without the base port)
415         * in one shot, rather than getting them one by one. Cannot do validation.</p>
416         *
417         * <p>If you need the base port maping during export, you may use the
418         * {@link BatchExporter} directly and pass {@code true} in the
419         * constructor.</p>
420         *
421         * @param publisher a consumer that does something with a map
422         *                  mapping portId to port.
423         *
424         * @return the exporter aggregating the port mappings and
425         *         on successful allocation calling the publishing
426         *         consumer.
427         *
428         * @see BatchExporter
429         */
430        static Exporter batching(Consumer<Map<String, Integer>> publisher) {
431           return new BatchExporter(false, publisher);
432        }
433    }

436     * A strategy for allocating a port block.
437     */
438    public interface Registrar extends AutoCloseable {
439        /**
440         * Returns the actual {@code basePort} of the newly allocated block,
441         * or throws exception.
442         *
443         * @param basePortHint the desired {@code basePort} to allocate port
444         *        from. If zero, that means that the registrar implementation
445         *        is free to choose.
446         *
447         * @return the base port of the newly allocated block.
448         */
449        int lock(int basePortHint);
450
451        /**
452         * Releases any resources related to this port block.
453         * After calling this method, the in the allocated block
454         * should not be considered reserved.
455         */
456        @Override void close();
457    }

459    /**
460     * A simple block-based allocation strategy, trying to bind a port at a
461     * fixed offset from {@code portBase} and on failure, retrying again
462     * at a new {@code portBase}, one block-size up.
463     */
464    public static class BlockRegistrar implements Registrar {
465        /**
466         * The address of the interface we want to use for locking.
467         */
468        public final @NotNull InetAddress bindAddress;
469
470        /**
471         * The desired number of ports in the block (subsequent port offsets
472         * should be in the [0..rangeSize-1] range).
473         */
474        public final int rangeSize;
475
476        /**
477         * The offset of the lock port, in the range [-1..rangeSize].
478         * Keep in mind that if you choose zero, you need to register
479         * the lock port, or it will clash with dynamically registered
480         * ports.
481         */
482        public final int lockOffset;
483
484        /**
485         * If {@code true} this registrar will put extra effort to make sure
486         * that the allocated {@code basePort==basePortHint+rangeSize*N}
487         * (where N is an integer number)
488         */
489        public boolean alignToBasePortHint;
490        private @Nullable ServerSocket locked;
491
492        /**
493         *
494         * @param bindAddress the address of the interface we want to use for locking.
495         * @param rangeSize the desired number of ports in the block
496         *        (subsequent port offsets should be in the [0..rangeSize-1] range).
497         * @param lockOffset the offset of the lock port, in the range [-1..rangeSize].
498         *        See {@link #lockOffset the field} for details.
499         */
500        public BlockRegistrar(@NotNull InetAddress bindAddress, int rangeSize, int lockOffset) {
501            if (lockOffset < -1) throw new IllegalArgumentException("Lock offset " + lockOffset + " should be >= -1");
502            if (lockOffset > rangeSize) throw new IllegalArgumentException("Lock offset " + lockOffset + " should be <= " + rangeSize + " (rangeSize)");
503            if (rangeSize<=0) throw new IllegalArgumentException("rangeSize: " + rangeSize);
504            this.bindAddress = bindAddress;
505            this.lockOffset = lockOffset;
506            this.rangeSize = rangeSize;
507        }
508
509        /**
510         * @param basePortHint {@inheritDoc} Negative value forces it to retry one
511         *        block up from the negated port.
512         */
513        @Override
514        public synchronized int lock(int basePortHint) {
515            if (Math.abs(basePortHint)>0xFFFF) {
516                throw new IllegalArgumentException("basePortHint: " + basePortHint);
517            }
518
519            int lastBasePort;
520            if (locked!=null) {
521                lastBasePort = locked.isBound() ? locked.getLocalPort() - lockOffset : -1;
522                rethrow(locked::close);
523            } else {
524                lastBasePort = -1;
525            }
526
527            int rangeStep = lockOffset<0 || lockOffset>=rangeSize
528                    ? rangeSize+1
529                    : rangeSize;
530
531            int basePort;
532            if (basePortHint>0) basePort=basePortHint;
533            else if (basePortHint<0) basePort = -basePortHint + rangeStep;
534            else if (basePortHint==0 && lastBasePort>0) basePort = lastBasePort + rangeStep;
535            else {
536                throw new IllegalArgumentException("basePortHint: " + basePortHint + ", lastBasePort: " + lastBasePort);
537            }
538
539            if (alignToBasePortHint && lastBasePort>0) {
540                int misalignment = Math.abs(lastBasePort - basePort) % rangeStep;
541                if (misalignment>0) {
542                    basePort += rangeStep - misalignment;
543                }
544            }
545
546            int lastPortInRange = basePort + rangeSize - 1;
547            if (basePort<=0 || lastPortInRange>0xFFFF) {
548                rethrow(new IOException("basePort: " + basePort));
549            }
550
551            int lockPort = basePort+lockOffset;
552            if (lockPort<=0 || lockPort>0xFFFF) {
553                rethrow(new IOException("lockPort: " + lockPort));
554            }
555            try {
556                locked = new ServerSocket();
557                locked.bind(new InetSocketAddress(bindAddress, lockPort));
558                return lockPort-lockOffset;
559            } catch (IOException e) {
560                return lock(-basePort);
561            }
562        }
563
564        @Override
565        public synchronized void close() {
566            if (locked !=null) rethrow(locked::close);
567        }
568    }

570    /**
571     * <p>A convenient way to receive all port mappings in one shot, rather
572     * than getting them one by one. Cannot do validation.</p>
573     *
574     * <p>This exporter internally aggregates all observed port mappings and
575     * calls the publishing consumer once on successful allocation.</p>
576     *
577     * @see Exporter#batching(Consumer)
578     */
579    private static class BatchExporter implements Exporter, AutoCloseable {
580        private final Map<String, Integer> batched = new HashMap<>();
581        private boolean withBasePort;
582        private final Consumer<Map<String, Integer>> publisher;
583
584        /**
585         * Creates an aggregating exporter, calling {@code publisher} once
586         * all mappings are finalized.
587         *
588         * @param withBasePort if {@code true}, the map passed to the
589         *        {@code publisher} includes the {@code basePort} mapped to
590         *        an empty string.
591         *
592         * @param publisher typically a closure that does something with the
593         *        ports (i.e. creates config file, calls management APIs, etc.)
594         */
595        public BatchExporter(boolean withBasePort, Consumer<Map<String, Integer>> publisher) {
596            this.withBasePort = withBasePort;
597            this.publisher = publisher;
598        }
599
600        @Override
601        public void export(@NotNull String id, int port) {
602            if (!withBasePort && id.isEmpty()) return;
603            batched.put(id, port);
604        }
605
606        @Override
607        public void close() {
608            publisher.accept(batched);
609        }
610    }
