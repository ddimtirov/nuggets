033/**
034 * <p><span class="badge green">Entry Point</span> Formats data as text-table, suitable for logging and printing. The recommended way is to
035 * use a builder-based DSL (though you may use the classes directly to achieve advanced scenarios).
036 * All following examples focus on the DSL:</p>
037 *
038 * <pre><code>
039 * String table = TextTable.withColumns("key", "value", "notes")
040 *                         .withData()
041 *                         .row("OP", "Austin Powers", "International Man of Mystery")
042 *                         .row("NP", "Nigel Powers", "NP complete")
043 *                         .buildTable()
044 *                         .format(10, new StringBuilder())
045 *                         .toString();
046 * </code></pre>
047 *
048 * <p>Which will be rendered as:</p>
049 *
050 * <pre><tt>
051 * +-----+---------------+------------------------------+
052 * | key | value         | notes                        |
053 * +-----+---------------+------------------------------+
054 * | OP  | Austin Powers | International Man of Mystery |
055 * | NP  | Nigel Powers  | NP complete                  |
056 * +-----+---------------+------------------------------+
057 * </tt></pre>
058 *
059 * <a>We may as well specify different visual style like this:</a>
060 *
061 * <pre><code>
062 * presetBox(Box.UNICODE_THIN); // will draw the table using nice thin box characters.
063 * </code></pre>
064 *
065 *
066 * <p>Which will change the rendering to a nice box using unicode's pseudo-graphic characters.
067 * (Blame Java's encoding handling for not being able to include it in the Javadoc easily.)</p>
068 *
069 *
070 * <p>Typically, we would want to define the table layout once and use it to format multiple
071 * data sets:</p>
072 *
073 * <pre><code>
074 * TextTable.LayoutBuilder layout = TextTable.withColumns("key", "value", "notes");
075 *
076 * layout.withData()
077 *       .row("OP", "Austin Powers", "International Man of Mystery")
078 *       .row("NP", "Nigel Powers", "NP complete")
079 *       .buildTable()
080 *       .format(10, System.err);
081 *
082 * layout.withData()
083 *       .row("foo", "Fubar", "beyond all repair")
084 *       .row("baz", "Bazqux", "")
085 *       .buildTable()
086 *       .format(10, System.err);
087 * </code></pre>
088 *
089 * <p>Or we may keep the table object around and format it to different outputs:</p>
090 *
091 * <pre><code>
092 * TextTable table = TextTable.withColumns("key", "value")
093 *                            .column("notes") // columns can be specified one-by-one
094 *                            .withData()
095 *                            .rows( // add rows in bulk
096 *                              new String[] { "OP", "Austin Powers", "International Man of Mystery" },
097 *                              new String[] { "NP", "Nigel Powers", "NP complete" },
098 *                            )
099 *                            .row()
100 *                            .buildTable();
101 *
102 * table.format(0, System.err);
103 * table.format(10, System.out);
104 * StringBuilder renderedTable = table.format(10, new StringBuilder());
105 * </code></pre>
106 *
107 * <p>By default each column is mandatory, left-aligned and rendered as string with 1 space
108 * padding. We can tweak all these per column with DSL like this:</p>
109 *
110 * <pre><code>
111 * TextTable.LayoutBuilder layout = TextTable
112 *      .withColumns()
113 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
114 *      .column("value", c -&gt; c.padding = 0) // remove padding
115 *      .column("notes", c -&gt; {
116 *            c.defaultValue="n/a";
117 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
118 *       });
119 * </code></pre>
120 *
121 * <p>If a value is an instance of {@code Supplier} or {@code Callable} (or id the value
122 * is {@code null} and the default value is instance of these interfaces), the functional
123 * interface will be called and the result will be rendered.</p>
124 *
125 * <pre><code>
126 * TextTable.LayoutBuilder layout = TextTable
127 *      .withColumns()
128 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
129 *      .column("value", c -&gt; c.padding = 0) // remove padding
130 *      .column("notes", c -&gt; {
131 *            c.defaultValue="n/a;
132 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
133 *       });
134 * </code></pre>
135 *
136 * <p>A formatter is allowed to change a column's metadata, which will be reset after each rendered value.
137 * Also we can access the columns after they are defined as follows:</p>
138 *
139 * <pre><code>
140 * TextTable.LayoutBuilder layout = TextTable.withColumns("Name", "Age", "Bio");
141 * DecimalFormat df = new DecimalFormat("#,###");
142 * layout.getAllColumns().each {c -&gt; { // for all columns
143 *   c.formatter = value -&gt; {
144 *       if (value instanceof Number) {
145 *           c.alignment = 1; // right-align
146 *           return df.format(((Number) value).longValue());
147 *       }
148 *       return String.valueOf(value).trim(); // will be left-aligned as default
149 *   }
150 *   c.defaultValue = new AtomicInteger::incrementAndGet;
151 * }}
152 * </code></pre>
153 *
154 * <p>There is no synchronization in any of the classes, but both the columns and table are practically
155 * immutable once defined (unless the formatters change the column fields). This means that both tables
156 * and layouts can be shared between threads. The builders on the other side are stateful and they are
157 * best confined to a single thread.</p>
158 */

259    /**
260     * <p>Constructor to create a table object given prepared columns and data.
261     * Typically one would use the builder API as follows:</p>
262     * <pre><code>
263     * TextTable table = TextTable.withColumns(
264     *                   "key"   , "value"
265     * ).withData().rows(
266     *    new String[] { "abc"   ,  123     },
267     *    new String[] { "foobar", "bazqux" },
268     * ).buildTable();
269     *
270     * table.format(10, System.out);   // print to stdout with indentation 10 chars
271     * logger.info(table.format(0, new StringBuilder()));   // log to a logger
272     * </code></pre>
273     *
274     * @param columns         list of columns describing the layout of the table
275     * @param data            collection of data rows
276     * @param separatorBefore map of row index to separator style
277     */
278    public TextTable(@NotNull List<@NotNull Column> columns, @NotNull List<@NotNull List<?>> data, @Nullable TreeMap<Integer, Separator> separatorBefore) {
279        this.columns = columns;
280        this.data = data;
281        this.separatorBefore = separatorBefore != null ? separatorBefore : Collections.emptyMap();
282
283        c2d = mapColumnIndexToDataIndex(columns);
284        List<SiblingAwareFormatter> safs = new ArrayList<>();
285        for (Column column : columns) {
286            if (column.formatter instanceof SiblingAwareFormatter) {
287                SiblingAwareFormatter saf = (SiblingAwareFormatter) column.formatter;
288                saf.lookup = new SiblingLookup(column.name);
289                safs.add(saf);
290            }
291        }
292        siblingAwareFormatters = safs;
293    }

295    /**
296     * <p>Render the table as text, based on the layout, per-column formatter, alignment and padding; and
297     * the {@code indent} parameter. If {@code pendingAutoformat} field is set to {@code true}, this
298     * method will also adjust the widths of the columns, so that the data will fit in the table.</p>
299     *
300     * @param indent the number of characters to put to the left of the start of the table.
301     * @param out the destination to which to append the text.
302     * @param <F> captures the type of the {@code out} parameter, so we can return it.
303     * @return the {@code out} parameter for chaining.
304     * @throws IOException passing through exceptions from {@link Appendable#append(char)}
305     *
306     * @see #pendingAutoformat
307     */
308    public <F extends Appendable> @NotNull F format(int indent, @NotNull F out) throws IOException {
309        return format(indent, out, outerFrame, true);
310    }

312    /**
313     * <p>Render the table as text, based on the layout, per-column formatter, alignment and padding; and
314     * the {@code indent} parameter. If {@code pendingAutoformat} field is set to {@code true}, this
315     * method will also adjust the widths of the columns, so that the data will fit in the table.</p>
316     *
317     * @param indent the number of characters to put to the left of the start of the table.
318     * @param out the destination to which to append the text.
319     * @param outerFrame if {@code false} will not render the outer frame
320     * @param includeHeader if {@code false} will not render the header and header-values separator
321     * @param <F> captures the type of the {@code out} parameter, so we can return it.
322     * @return the {@code out} parameter for chaining.
323     * @throws IOException passing through exceptions from {@link Appendable#append(char)}
324     *
325     * @see #pendingAutoformat
326     */
327    public <F extends Appendable> @NotNull F format(int indent, @NotNull F out, boolean outerFrame, boolean includeHeader) throws IOException {
328        if (pendingAutoformat) {
329            autoformatFromContent();
330            pendingAutoformat = false;
331        }
332
333        StringBuilder indentPad = pad(new StringBuilder(indent), indent, TextTable.indent);
334
335        // outer-frame: top
336        if (outerFrame) {
337            out.append(indentPad);
338            appendFrameHline(out, style.joints(0));
339        }
340
341        if (includeHeader) {
342            // header row
343            siblingAwareFormatters.forEach(saf -> saf.lookup.row=0);
344            out.append(indentPad);
345            appendRow(out, null, indentPad);
346
347            // inner-frame: header/values separator
348            out.append(indentPad);
349            appendFrameHline(out, style.joints(1));
350        }
351
352        // rows
353        int itemIdx=0;
354        for (int rowIdx = 0; rowIdx < data.size(); rowIdx++) {
355            for (SiblingAwareFormatter saf : siblingAwareFormatters) saf.lookup.row=rowIdx;
356            List<?> row = data.get(rowIdx);
357            Separator sep = separatorBefore.get(itemIdx++);
358            if (sep!=null) {
359                out.append(indentPad);
360                appendFrame(out, sep);
361            }
362            out.append(indentPad);
363            appendRow(out, row, indentPad);
364        }
365
366        // outer-frame: bottom
367        if (outerFrame) {
368            out.append(indentPad);
369            appendFrameHline(out, style.joints(2));
370        }
371        return out;
372    }

374    /**
375     * Scan the table content and default column parameters, making sure
376     * that each column is at least as wide as the longest rendered value.
377     */
378    @SuppressWarnings("try")
379    private void autoformatFromContent() {
380        for (Column column : columns) {
381            SiblingAwareFormatter saf = column.formatter instanceof SiblingAwareFormatter
382                    ? (SiblingAwareFormatter) column.formatter
383                    : null;
384
385            for (int rowIdx = 0; rowIdx < data.size(); rowIdx++) {
386                List<?> row = data.get(rowIdx);
387                if (saf!=null) saf.lookup.row = rowIdx;
388
389                int cellWidth;
390                try (Column.Memento ignored = column.new Memento()) {
391                    Object rawCell = column.virtual ? column.defaultValue : row.get(c2d[column.index]);
392                    String formattedCell = column.format(rawCell);
393                    cellWidth = Arrays.stream(formattedCell.split(eol))
394                                   .mapToInt(String::length)
395                                   .max().orElse(0);
396                }
397                column.width = Math.max(column.width, cellWidth);
398
399            }
400        }
401    }

403    private void appendFrame(@NotNull Appendable out, @NotNull Separator separator) throws IOException {
404        String horizontal = separator.horizontalGlyph != null ? separator.horizontalGlyph : style.horizontal();
405        String[] jointRow = separator.junctionGlyphs != null ? separator.junctionGlyphs : style.joints(2);
406
407
408        int leftBorderSize = outerFrame && jointRow[0] != null ? jointRow[0].length() : 0;
409        int rightBorderSize = outerFrame && jointRow[2] != null ? jointRow[2].length() : 0;
410        int jointSize = jointRow[1] != null ? jointRow[1].length() : 0;
411        int size = columns.stream().mapToInt(x -> x.width + x.padding*2).sum()
412                + leftBorderSize + jointSize * (columns.size()-1) + rightBorderSize;
413        StringBuilder sb = new StringBuilder(size+16);
414        appendHline(sb, jointRow, horizontal);
415        assert sb.length()==size;
416
417        if (separator.name!=null) {
418            int maxTitleSize = Math.max(0, size - 2 * separator.padding);
419            int alignGap = maxTitleSize - separator.name.length();
420            if(alignGap>0) {
421                int start = separator.padding + (int) (alignGap * separator.alignment);
422                sb.replace(start, start+separator.name.length(), separator.name);
423            } else {
424                String title;
425                if (maxTitleSize>9) {
426                    title = separator.name.substring(0, (maxTitleSize-3) / 2);
427                    title += "...";
428                    title += separator.name.substring(separator.name.length()-(maxTitleSize-title.length()));
429                } else {
430                    title = separator.name.substring(0, maxTitleSize);
431                }
432                assert title.length()==maxTitleSize;
433                sb.replace(separator.padding, size - separator.padding, title);
434            }
435            assert sb.length()==size;
436        }
437
438        out.append(sb).append(eol);
439    }

441    /**
442     * Appends formatted text line for horizontal table separator, including line terminator.
443     * Indentation is no appended as we want to reuse the rendering buffer
444     * (see the implementation of {@link #format(int, Appendable)}).
445     * @param out destination for the table row.
446     * @param jointRow array of 3 values for left, middle and center joints.
447     * @throws IOException if the {@param out} throws while appending.
448     * @see Style#joints(int)
449     */
450    private void appendFrameHline(@NotNull Appendable out, String[] jointRow) throws IOException {
451        String horizontal = style.horizontal();
452        if (horizontal==null) return;
453
454        appendHline(out, jointRow, horizontal);
455        out.append(eol);
456    }

458    /**
459     * Appends formatted text line for horizontal table separator, including line terminator.
460     * Indentation is no appended as we want to reuse the rendering buffer
461     * (see the implementation of {@link #format(int, Appendable)}).
462     * @param out destination for the table row.
463     * @param jointRow array of 3 values for left, middle and center joints.
464     * @param horizontal the glyph used to draw the hlines of this table
465     * @throws IOException if the {@param out} throws while appending.
466     * @see Style#joints(int)
467     */
468    private void appendHline(@NotNull Appendable out, String[] jointRow, @Nullable String horizontal) throws IOException {
469        for (int i = 0; i < columns.size(); i++) {
470            Column column = columns.get(i);
471
472            if (outerFrame && jointRow[0]!=null && i == 0) out.append(jointRow[0]);
473            if (column.hidden) continue;
474            pad(out, column.padding * 2 + column.width, horizontal !=null ? horizontal : " ");
475            if (i >= columns.size() - 1) {
476                String jointGlyph = jointRow[2];
477                if (outerFrame && jointGlyph != null) out.append(jointGlyph);
478            } else {
479                String jointGlyph = jointRow[1];
480                if (jointGlyph!=null) out.append(jointGlyph);
481            }
482        }
483    }

485    /**
486     * Appends formatted text snipped for single table row or header, including padding and
487     * line terminator. Indentation is not appended for the first line, but in case there are
488     * cell values rendering to multi-line, it will be appnded on the continuation lines (see
489     * the implementation of {@link #format(int, Appendable)}).
490     *
491     * @param out destination for the table row.
492     * @param row the values to append. If {@code null} - append header with the column names.
493     * @param continuationIndentPad the padding to use for
494     * @throws IOException if the {@param out} throws while appending.
495     */
496    @SuppressWarnings("try")
497    private void appendRow(@NotNull Appendable out, @Nullable List<?> row, StringBuilder continuationIndentPad) throws IOException {
498        Column lastColumn = columns.get(columns.size() - 1);
499
500        // rethink the approach if efficiency becomes an issue
501        // we can have a fast-path for single-line values and lazy allocation of arrays
502        int maxIndex = columns.stream().mapToInt(x->x.index).max().orElseThrow(() -> new IllegalStateException("No columns defined!"));
503        String[][] multilineValues = new String[maxIndex+1][];
504        Column.Memento[] multilineFormating = new Column.Memento[maxIndex+1];
505
506        for (Column column : columns) {
507            if (column.hidden) continue;
508
509            try (Column.Memento ignored = column.new Memento()) {
510                String formatted;
511                if (row == null) {
512                    formatted = column.name;
513                } else {
514                    Object rawValue = column.virtual ? column.defaultValue : row.get(c2d[column.index]);
515                    Object defaultedValue = rawValue == null ? column.defaultValue : rawValue;
516                    formatted = column.format(defaultedValue);
517                }
518
519                String[] lines = formatted.split(eol);
520                multilineFormating[column.index] = column.new Memento();
521                multilineValues[column.index] = lines;
522            }
523        }
524
525        int maxLines = Arrays.stream(multilineValues).filter(Objects::nonNull).mapToInt(x->x.length).max().orElse(0);
526        for (int lineIdx = 0; lineIdx < maxLines; lineIdx++) {
527            if (lineIdx>0) out.append(continuationIndentPad);
528            if (outerFrame && style.vertical()!=null) out.append(style.vertical());
529            for (Column column : columns) {
530                if (column.hidden) continue;
531
532                Column.Memento paragraphFormatting = multilineFormating[column.index];
533                String[] lines = multilineValues[column.index];
534                String formatted = lines!=null && lines.length>lineIdx ? lines[lineIdx] : "";
535                if (formatted.length() > column.width) {
536                    throw new IllegalStateException(
537                            "Formatted string '" + formatted + "'::length==" + formatted.length() +
538                                    " > column " + column + "::width==" + column.width);
539                }
540                appendValue(out, formatted,
541                        paragraphFormatting.width, paragraphFormatting.padding, paragraphFormatting.alignment,
542                        paragraphFormatting.rightBorder, column != lastColumn);
543            }
544            out.append(eol);
545        }
546    }

548    private void appendValue(@NotNull Appendable out, String formattedValue, int width, int padding, double alignment, @Nullable String rightBorder, boolean lastColumn) throws IOException {
549        if (formattedValue.isEmpty()) {
550            // in case of multi-char padding string, don't restart in the middle
551            pad(out, width + padding * 2, TextTable.padding);
552        } else {
553            int alignmentGap = width - formattedValue.length();
554            int leftAlignmentPad = (int) (alignmentGap * alignment);
555            int rightAlignmentPad = alignmentGap - leftAlignmentPad;
556            pad(out, leftAlignmentPad + padding, TextTable.padding);
557            out.append(formattedValue);
558            pad(out, rightAlignmentPad + padding, TextTable.padding);
559        }
560        String rightBorderGlyph = rightBorder ==null ? style.vertical() : rightBorder;
561        if ((lastColumn || outerFrame) && rightBorderGlyph !=null) out.append(rightBorderGlyph);
562    }

564    /**
565     * <p>Specifies the characters used to draw the lines and corners of a {@link TextTable table}.
566     * As of now, the whole table is drawn with the same set of characters
567     * (we don't allow to customize per row or per cell).</p>
568     * <p>Typically you would want to use one of the {@link Box} values, though
569     * you may provide your own styles too.</p>
570     */
571    public interface Style {
572        /**
573         * Determines the characters used to draw the horizontal lines of a table.
574         * @return horizontal line string. If multiple characters, will be repeated
575         *         and trimmed as required to reach the desired size.
576         *         If {@code null}, horizontal separators will be skipped entirely.
577         * @see TextTable#style
578         * @see #presetBox(Style)
579         */
580        @Contract(pure = true)
581        @Nullable String horizontal();
582
583        /**
584         * Determines the characters used to draw the vertical lines of a table.
585         * @return vertical line string. If multiple characters, the column separators
586         *         will be using the whole string (make sure to have matching T-joints
587         *         and cross-joints).
588         *         If {@code null}, vertical separators will be skipped entirely.
589         * @see TextTable#style
590         * @see #presetBox(Style)
591         */
592        @Contract(pure = true)
593        @Nullable String vertical();
594
595        /**
596         * Determines the characters used to draw the joints and corners of a table.
597         * The joints are defined as a virtual 3x3 array, with values as follows:
598         * <pre>
599         * String[][] joints = {
600         *     { TOP_LEFT_CORNER    , TOP_T_JOINT    , TOP_RIGHT_CORNER    },
601         *     { LEFT_T_JOINT       , CROSS_JOINT    , RIGHT_T_JOINT       },
602         *     { BOTTOM_LEFT_CORNER , BOTTOM_T_JOINT , BOTTOM_RIGHT_CORNER },
603         * }
604         * </pre>
605         * @param row which row of the 9-box to return.
606         * @return array of 3 strings to be used as left corner, center and right-corner for the specified {@code row}.

944     * Used by renderers to lookup values from the same row, coordinates of te currently rendered cell,
945     * as well as column aggregates.
946     */
947    // as the instance is per column we may also add memoization if it pops in profiling
948    public class SiblingLookup {
949        /**
950         * The name of the column for the cell being rendered.
951         */
952        public final @NotNull String columnName;
953
954        /**
955         * The index of the row for the cell being rendered.
956         */
957        public int row;
958
959        /**
960         * Create new lookup for a column - typically called by {@link TextTable}.
961         * We create a new instance for each column, to reduce the mutable state and allow
962         * for more efficient caching.
963         * @param columnName the current column name.
964         */
965        protected SiblingLookup(@NotNull String columnName) {
966            this.columnName = columnName;
967        }
968
969        /**
970         * Lookup aggregate value from the formatted values in a column.
971         * @param columnName the column to look up.
972         * @return a stream of the column's formatted values.
973         */
974        public @NotNull Stream<String> column(@NotNull String columnName) {
975            Column column = columnByName(columnName);
976            int originalRow = row;
977            try {
978                String[] c = new String[data.size()];
979                for (row = 0; row < c.length; row++) {
980                    Object cell = data.get(row).get(c2d[column.index]);
981                    c[row] = column.format(cell);
982                }
983                return Arrays.stream(c);
984            } finally {
985                row = originalRow;
986            }
987        }
988
989        /**
990         * Lookup aggregate value from the raw values in a column.
991         * @param columnName the column to look up.
992         * @return a stream of the column's raw values.
993         */
994        public @NotNull Stream<Object> columnRaw(@NotNull String columnName) {
995            Column column = columnByName(columnName);
996            return data.stream().map(r -> r.get(c2d[column.index]));
997        }
998
999        /**
1000         * Lookup formatted value from another column on the same row as the currently rendered
1001         * cell.
1002         *
1003         * @param columnName the column to look up.
1004         * @return formatted value
1005         */
1006        public @NotNull String sibling(@NotNull String columnName) {
1007            Column column = columnByName(columnName);
1008            Object value = column.virtual ? column.defaultValue : data.get(row).get(c2d[column.index]);
1009            if (column.formatter instanceof SiblingAwareFormatter) {
1010                SiblingAwareFormatter saf = (SiblingAwareFormatter) column.formatter;
1011                saf.lookup.row = row;
1012            }
1013            return column.format(value);
1014        }
1015
1016        /**
1017         * Lookup raw value from another column on the same row as the currently rendered
1018         * cell.
1019         *
1020         * @param columnName the column to look up.
1021         * @return raw value
1022         */
1023        public @Nullable Object siblingRaw(@NotNull String columnName) {
1024            Column column = columnByName(columnName);
1025            return data.get(row).get(c2d[column.index]);
1026        }
1027
1028        @NotNull
1029        private TextTable.@NotNull Column columnByName(@NotNull String columnName) {
1030            //noinspection ForLoopReplaceableByForEach - we want to save on the ArrayList iterator
1031            for (int i = 0; i < columns.size(); i++) {
1032                Column c = columns.get(i);
1033                if (Objects.equals(columnName, c.name)) return c;
1034            }
1035            throw new NoSuchElementException("Can't find a column named '" + columnName + "'");
1036        }

1256         *               if {@code null} - use the table inner joints style
1257         * @return reference to {@code this} for chaining.
1258         */
1259        public @NotNull DataBuilder separator(@Nullable String name, @Nullable Consumer<Separator> config) {
1260            return separator(Separator.of(name, null, null, config));
1261        }

1264         * Add a prepared row separator
1265         * @param separator the configured separator instance. The separator is not copied -
1266         *                  take care if you modify after adding.
1267         * @return reference to {@code this} for chaining.

1270            separatorBefore.put(data.size(), separator);

1274        private DataBuilder appendRowChecked(List<?> row, @Nullable Object rowDescription) {
1275            if (strict) {
1276                long virtualColumns = columns.stream().filter(c->c.virtual).count();
1277                if (row.size() != maxColumnIndex - virtualColumns + 1) {
1278                    throw new IllegalArgumentException(String.format(
1279                            "Data doesn't match the columns:%n%d COLUMNS: %s%n%d DATA: %s",
1280                            columns.size(), columns, row.size(), rowDescription
1281                    ));
1282                }
1283            }
1284            int virtualOffset = 0;
1285            for (Column column : columns) {
1286                if (column.virtual) {
1287                    virtualOffset++;
1288                    continue;
1289                }
1290                if (column.defaultValue != null) continue;
1291
1292                int colValueIndex = column.index - virtualOffset;
1293                if (colValueIndex >=row.size()) {
1294                    throw new IllegalArgumentException(String.format(
1295                            "Column index out of range: %s%n%d COLUMNS: %s%n%d DATA: %s",
1296                            column, columns.size(), columns, row.size(), rowDescription
1297                    ));
1298                }
1299
1300                if (row.get(colValueIndex)==null) {
1301                    throw new IllegalArgumentException(String.format(
1302                            "Non default column is null: %s%n%d COLUMNS: %s%n%d DATA: %s",
1303                            column, columns.size(), columns, row.size(), rowDescription
1304                    ));
1305                }
1306            }
1307            data.add(row);
1308            return this;
1309        }

1333    /**
1334     * <p>Adds {@code length} characters of padding to the {@code out} parameter.
1335     * The padding is filled by repeating the {@code pad} pattern as needed,
1336     * trimming the last occurrence to size.</p>
1337     * <p>For example, {@code pad(out, 8, "abc")} will result in <em>"abcabcab"</em>
1338     * (8 characters) appened to the {@code out} parameter.</p>
1339     * @param out where to put the padding.
1340     * @param length how many characters to pad.
1341     * @param pad what pattern to use for padding.
1342     * @param <T> captures the type of the {@code out} parameter, so we can return it.
1343     * @return the {@code out} parameter for chaining.
1344     */
1345    protected static <T extends Appendable> T pad(@NotNull T out, int length, @NotNull String pad) {
1346        try {
1347            int padLength = pad.length();
1348            for (int i = 0; i < length; /* incremented in loop */) {
1349                int charsToAppend = Math.min(padLength, length - i);
1350                out.append(pad, 0, charsToAppend);
1351                i += charsToAppend;
1352            }
1353            return out;
1354        } catch (IOException e) {
1355            return Exceptions.rethrow(e);
1356        }
1357    }

1359    /**
1360     * Maps the ingested data to the desired data layout.
1361     * @param columns the column spec;
1362     * @return array mapping column logical index to its place in the data layout.
1363     */
1364    protected static int[] mapColumnIndexToDataIndex(@NotNull List<@NotNull Column> columns) {
1365        int maxIndex = columns.stream().mapToInt(x->x.index).max().orElseThrow(() -> new IllegalStateException("No columns defined!"));
1366        int[] c2d = new int[maxIndex+1];
1367        Arrays.fill(c2d, -1);
1368
1369        int passedVirtual = 0;
1370        for (Column column : columns) {
1371            if (column.virtual) {
1372                passedVirtual++;
1373                c2d[column.index] = -2;
1374            } else {
1375                c2d[column.index] = column.index - passedVirtual;
1376            }
1377        }
1378        return c2d;
1379    }
