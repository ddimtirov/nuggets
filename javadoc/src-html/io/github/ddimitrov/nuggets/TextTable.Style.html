031/**
032 * <p>Formats data as text-table, suitable for logging and printing. The recommended way is to
033 * use a builder-based DSL (though you may use the classes directly to achieve advanced scenarios).
034 * All following examples focus on the DSL:</p>
035 *
036 * <pre><code>
037 * String table = TextTable.withColumns("key", "value", "notes")
038 *                         .withData()
039 *                         .row("OP", "Austin Powers", "International Man of Mystery")
040 *                         .row("NP", "Nigel Powers", "NP complete")
041 *                         .buildTable()
042 *                         .format(10, new StringBuilder())
043 *                         .toString();
044 * </code></pre>
045 *
046 * <p>Which will be rendered as:</p>
047 *
048 * <pre>
049 * +-----+---------------+------------------------------+
050 * | key | value         | notes                        |
051 * +-----+---------------+------------------------------+
052 * | OP  | Austin Powers | International Man of Mystery |
053 * | NP  | Nigel Powers  | NP complete                  |
054 * +-----+---------------+------------------------------+
055 * </pre>
056 *
057 * <a>We may as well specify different visual style like this:</a>
058 *
059 * <pre><code>
060 * presetBox(Box.UNICODE_THIN); // will draw the table using nice thin box characters.
061 * </code></pre>
062 *
063 *
064 * <p>Which will change the rendering to a nice box using unicode's pseudo-graphic characters.
065 * (Blame Java's encoding handling for not being able to include it in the Javadoc easily.)</p>
066 *
067 *
068 * <p>Typically, we would want to define the table layout once and use it to format multiple
069 * data sets:</p>
070 *
071 * <pre><code>
072 * TextTable.LayoutBuilder layout = TextTable.withColumns("key", "value", "notes");
073 *
074 * layout.withData()
075 *       .row("OP", "Austin Powers", "International Man of Mystery")
076 *       .row("NP", "Nigel Powers", "NP complete")
077 *       .buildTable()
078 *       .format(10, System.err);
079 *
080 * layout.withData()
081 *       .row("foo", "Fubar", "beyond all repair")
082 *       .row("baz", "Bazqux", "")
083 *       .buildTable()
084 *       .format(10, System.err);
085 * </code></pre>
086 *
087 * <p>Or we may keep the table object around and format it to different outputs:</p>
088 *
089 * <pre><code>
090 * TextTable table = TextTable.withColumns("key", "value")
091 *                            .column("notes") // columns can be specified one-by-one
092 *                            .withData()
093 *                            .rows( // add rows in bulk
094 *                              new String[] { "OP", "Austin Powers", "International Man of Mystery" },
095 *                              new String[] { "NP", "Nigel Powers", "NP complete" },
096 *                            )
097 *                            .row()
098 *                            .buildTable();
099 *
100 * table.format(0, System.err);
101 * table.format(10, System.out);
102 * StringBuilder renderedTable = table.format(10, new StringBuilder());
103 * </code></pre>
104 *
105 * <p>By default each column is mandatory, left-aligned and rendered as string with 1 space
106 * padding. We can tweak all these per column with DSL like this:</p>
107 *
108 * <pre><code>
109 * TextTable.LayoutBuilder layout = TextTable
110 *      .withColumns()
111 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
112 *      .column("value", c -&gt; c.padding = 0) // remove padding
113 *      .column("notes", c -&gt; {
114 *            c.defaultValue="n/a";
115 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
116 *       });
117 * </code></pre>
118 *
119 * <p>If a value is an instance od {@code Supplier} or {@code Callable} (or id the value
120 * is {@code null} and the default value is instance of these interfaces), the functional
121 * interface will be called and the result will be rendered.</p>
122 *
123 * <pre><code>
124 * TextTable.LayoutBuilder layout = TextTable
125 *      .withColumns()
126 *      .column("key"  , c -&gt; c.alignment=1) // right-aligned
127 *      .column("value", c -&gt; c.padding = 0) // remove padding
128 *      .column("notes", c -&gt; {
129 *            c.defaultValue="n/a;
130 *            c.formatter = val -&gt; val.toString().toLowerCase(); // custom formatter
131 *       });
132 * </code></pre>
133 *
134 * <p>A formatter is allowed to change a column's metadata, which will be reset after each rendered value.
135 * Also we can access the columns after they are defined as follows:</p>
136 *
137 * <pre><code>
138 * TextTable.LayoutBuilder layout = TextTable.withColumns("Name", "Age", "Bio");
139 * DecimalFormat df = new DecimalFormat("#,###");
140 * layout.getAllColumns().each {c -&gt; { // for all columns
141 *   c.formatter = value -&gt; {
142 *       if (value instanceof Number) {
143 *           c.alignment = 1; // right-align
144 *           return df.format(((Number) value).longValue());
145 *       }
146 *       return String.valueOf(value).trim(); // will be left-aligned as default
147 *   }
148 *   c.defaultValue = new AtomicInteger::incrementAndGet;
149 * }}
150 * </code></pre>
151 *
152 * <p>There is no synchronization in any of the classes, but both the columns and table are practically
153 * immutable once defined (unless the formatters change the column fields). This means that both tables
154 * and layouts can be shared between threads. The builders on the other side are stateful and they are
155 * best confined to a single thread.</p>
156 */
157public class TextTable {

240    /**
241     * <p>Constructor to create a table object given prepared columns and data.
242     * Typically one would use the builder API as follows:</p>
243     * <pre><code>
244     * TextTable table = TextTable.withColumns(
245     *                   "key"   , "value"
246     * ).withData().rows(
247     *    new String[] { "abc"   ,  123     },
248     *    new String[] { "foobar", "bazqux" },
249     * ).buildTable();
250     *
251     * table.format(10, System.out);   // print to stdout with indentation 10 chars
252     * logger.info(table.format(0, new StringBuilder()));   // log to a logger
253     * </code></pre>
254     *
255     * @param columns list of columns describing the layout of the table
256     * @param data collection of data rows
257     */
258    public TextTable(@NotNull List<@NotNull Column> columns, @NotNull List<@NotNull List<?>> data) {
259        this.columns = columns;
260        this.data = data;
261    }

263    /**
264     * <p>Render the table as text, based on the layout, per-column formatter, alignment and padding; and
265     * the {@code indent} parameter. If {@code pendingAutoformat} field is set to {@code true}, this
266     * method will also adjust the widths of the columns, so that the data will fit in the table.</p>
267     *
268     * @param indent the number of characters to put to the left of the start of the table.
269     * @param out the destination to which to append the text.
270     * @param <F> captures the type of the {@code out} parameter, so we can return it.
271     * @return the {@code out} parameter for chaining.
272     * @throws IOException passing through exceptions from {@link Appendable#append(char)}
273     *
274     * @see #pendingAutoformat
275     */
276    public <F extends Appendable> @NotNull F format(int indent, @NotNull F out) throws IOException {
277        if (pendingAutoformat) {
278            autoformatFromContent();
279            pendingAutoformat = false;
280        }
281
282        StringBuilder indentPad = pad(new StringBuilder(indent), indent, TextTable.indent);
283
284        out.append(indentPad);
285        appendSeparator(out, style.joints(0));
286
287        out.append(indentPad);
288        appendRow(out, null);
289
290        out.append(indentPad);
291        appendSeparator(out, style.joints(1));
292
293        for (List<?> row : data) {
294            out.append(indentPad);
295            appendRow(out, row);
296        }
297
298        out.append(indentPad);
299        appendSeparator(out, style.joints(2));
300        return out;
301    }
302
303    /**
304     * Scan the table content and default column parameters, making sure
305     * that each column is at least as wide as the longest rendered value.
306     */
307    @SuppressWarnings("try")
308    private void autoformatFromContent() {
309        for (Column column : columns) {
310            int maxValueLength = data.stream()
311                                     .map(row -> row.get(column.index))
312                                     .map(value -> {
313                                         try (Column.Memento ignored = column.new Memento()) {
314                                             return column.format(value);
315                                         }
316                                     })
317                                     .map(String::length)
318                                     .max(Comparator.naturalOrder())
319                                     .orElse(0);
320
321            column.width = Math.max(maxValueLength, column.width);
322        }
323    }

325    /**
326     * Appends formatted text line for horizontal table separator, including line terminator.
327     * Indentation is no appended as we want to reuse the rendering buffer
328     * (see the implementation of {@link #format(int, Appendable)}).
329     * @param out destination for the table row.
330     * @param jointRow array of 3 values for left, middle and center joints.
331     * @throws IOException if the {@param out} throws while appending.
332     * @see Style#joints(int)
333     */
334    private void appendSeparator(@NotNull Appendable out, @NotNull String[] jointRow) throws IOException {
335        for (int i = 0; i < columns.size(); i++) {
336            Column column = columns.get(i);
337            boolean lastColumn = i < columns.size() - 1;
338
339            if (i == 0) out.append(jointRow[0]);
340            pad(out, column.padding * 2 + column.width, style.horizontal());
341            out.append(jointRow[lastColumn ? 1 : 2]);
342        }
343        out.append(eol);
344    }
345
346    /**
347     * Appends formatted text line for single table row or header, including padding and
348     * line terminator. Indentation is no appended as we want to reuse the rendering buffer
349     * (see the implementation of {@link #format(int, Appendable)}).
350     * @param out destination for the table row.
351     * @param row the values to append. If {@code null} - append header with the column names.
352     * @throws IOException if the {@param out} throws while appending.
353     */
354    @SuppressWarnings("try")
355    private void appendRow(@NotNull Appendable out, @Nullable List<?> row) throws IOException {
356        out.append(style.vertical());
357        for (Column column : columns) {
358            try (Column.Memento ignored = column.new Memento()) {
359                String formatted;
360                if (row == null) {
361                    formatted = column.name;
362                } else {
363                    Object rawValue = row.get(column.index);
364                    Object defaultedValue = rawValue == null ? column.defaultValue : rawValue;
365                    formatted = column.format(defaultedValue);
366                }
367
368                if (formatted.length() > column.width) {
369                    throw new IllegalStateException(
370                            "Formatted string '" + formatted + "'::length==" + formatted.length() +
371                                    " > column " + column + "::width==" + column.width);
372                }
373
374                if (formatted.isEmpty()) {
375                    // in case of multi-char padding string, don't restart in the middle
376                    pad(out, column.width + column.padding * 2, padding);
377                } else {
378                    int alignmentGap = column.width - formatted.length();
379                    int leftAlignmentPad = (int) (alignmentGap * column.alignment);
380                    int rightAlignmentPad = alignmentGap - leftAlignmentPad;
381                    pad(out, leftAlignmentPad + column.padding, padding);
382                    out.append(formatted);
383                    pad(out, rightAlignmentPad + column.padding, padding);
384                }
385                out.append(style.vertical());
386            }
387        }
388        out.append(eol);
389    }

391    /**
392     * <p>Specifies the characters used to draw the lines and corners of a table.
393     * As of now, the whole table is drawn with the same set of characters
394     * (we don't allow to customize per row or per cell).</p>
395     * <p>Typically you would want to use one of the {@link Box} values, though
396     * you may provide your own styles too.</p>
397     */
398    public interface Style {
399        /**
400         * Determines the characters used to draw the horizontal lines of a table.
401         * @return horizontal line string. If multiple characters, will be repeated
402         *         and trimmed as required to reach the desired size.
403         * @see TextTable#style
404         * @see #presetBox(Style)
405         */
406        @Contract(pure = true)
407        @NotNull String horizontal();
408
409        /**
410         * Determines the characters used to draw the vertical lines of a table.
411         * @return vertical line string. If multiple characters, the column separators
412         *         will be using the whole string (make sure to have matching T-joints
413         *         and cross-joints).
414         * @see TextTable#style
415         * @see #presetBox(Style)
416         */
417        @Contract(pure = true)
418        @NotNull String vertical();
419
420        /**
421         * Determines the characters used to draw the joints and corners of a table.
422         * The joints are defined as a virtual 3x3 array, with values as follows:
423         * <pre>
424         * String[][] joints = {
425         *     { TOP_LEFT_CORNER    , TOP_T_JOINT    , TOP_RIGHT_CORNER    },
426         *     { LEFT_T_JOINT       , CROSS_JOINT    , RIGHT_T_JOINT       },
427         *     { BOTTOM_LEFT_CORNER , BOTTOM_T_JOINT , BOTTOM_RIGHT_CORNER },
428         * }
429         * </pre>
430         * @param row which row of the 9-box to return.
431         * @return array of 3 strings to be used as left corner, center and right-corner for the specified {@code row}.
432         *         See examples in {@link Box}.
433         * @see TextTable#style
434         * @see #presetBox(Style)
435         */
436        @Contract(pure = true)
437        @NotNull String[] joints(int row);
438    }
439
440    /**
441     * <p>Preset styles, specifying the characters used to draw the lines and corners
442     * of a table. As of now, the whole table is drawn with the same set of characters
443     * (we don't allow to customize per row or per cell).</p>
444     * <p>You don't need to use this class if you prefer to provide your own styles.</p>
445     * @see TextTable#presetBox(Style)
446     */
447    public enum Box implements Style {
448        /** Plain ASCII - suitable for every log file */
449        ASCII("-", "|",
450                new String[]{"+", "+", "+"},
451                new String[]{"+", "+", "+"},
452                new String[]{"+", "+", "+"}),
453
454        /** Stylish and slim, spruce your stdio with UNICODE_THIN */
455        UNICODE_THIN("\u2500", "\u2502",
456                new String[]{"\u250c", "\u252c", "\u2510"},
457                new String[]{"\u251c", "\u253c", "\u2524"},
458                new String[]{"\u2514", "\u2534", "\u2518"}),
459
460        /** ...but how about drawing rectangles with rounded corners? Can we do that now, too? */
461        UNICODE_THIN_ROUNDED("\u2500", "\u2502",
462                new String[]{"\u256d", "\u252c", "\u256e"},
463                new String[]{"\u251c", "\u253c", "\u2524"},
464                new String[]{"\u2570", "\u2534", "\u256f"}),
465
466        /** Sturdy and solid, nothing stands out quite like UNICODE_THICK */
467        UNICODE_THICK("\u2501", "\u2503",
468                new String[]{"\u250f", "\u2533", "\u2513"},
469                new String[]{"\u2523", "\u254b", "\u252b"},
470                new String[]{"\u2517", "\u253b", "\u251b"}),
471
472        /** Because sometimes two is better than one */
473        UNICODE_DOUBLE("\u2550", "\u2551",
474                new String[]{"\u2554", "\u2566", "\u2557"},
475                new String[]{"\u2560", "\u256c", "\u2563"},
476                new String[]{"\u255a", "\u2569", "\u255d"});
477
478        private final @NotNull String horizontal;
479        private final @NotNull String vertical;
480        private final @NotNull String[][] joints;
481
482        Box(@NotNull String horizontal, @NotNull String vertical, @NotNull String[]... joints) {
483            this.horizontal = horizontal;
484            this.vertical = vertical;
485            this.joints = joints;
486        }
487
488        @Override
489        @Contract(pure = true)
490        public @NotNull String horizontal() {
491            return horizontal;
492        }
493
494        @Override
495        @Contract(pure = true)
496        public @NotNull String vertical() {
497            return vertical;
498        }
499
500        @Override
501        @Contract(pure = true) // we are leaking the raw array - not worth creating garbage... please don't modify it!
502        public @NotNull String[] joints(int row) {
503            return joints[row];
504        }
505    }

507    /**
508     * Defines the formatting attributes for a single column.
509     */
510    public static class Column {
511        /**
512         * The column name.
513         */
514        public final @NotNull String name;
515
516        /**
517         * The index of the data row that should be displayed in this column.
518         * When you use a builder, this is managed automatically (auto-increment).
519         */
520        public final int index;
521
522        /**
523         * The column width (measured in characters)
524         */
525        public int width;
526
527        /**
528         * Padding to be left between the value and the column separator (measured in characters)
529         */
530        public int padding = 1;
531
532        /**
533         * Alignment of the rendered value (0-left, 1-right, 0.5-center, 0.75-slightly rightish)
534         */
535        public double alignment = 0d;
536
537        /**
538         * Function to convert from row values to string. The formatter may change any of
539         * the mutable attributes of the column, impacting the rendering of the value
540         * (they will be reset for the next row).
541         */
542        public @NotNull Function<@Nullable Object, String> formatter = Column::defaultFormatter;
543
544        /**
545         * Used if the row value is {@code null}.
546         */
547        public @Nullable Object defaultValue;
548
549        /**
550         * <p>Memorizes the mutable fields of its enclosing class and restores them on
551         * {@link #restore()} or {@link #close()}.</p>
552         *
553         * <p>As this class implements {@code Autocloseable}, it can be used in a
554         * try-with-resources construct in RAII manner as follows:</p>
555         *
556         * <pre><code>
557         * \@SuppressWarnings("try") // to prevent the compiler from complaining
558         * void preserveColumnSettings() {
559         *     try(Column.Memento ignore = column.new Memento()) {
560         *         // do things that modify the fields `column`
561         *     }
562         *     // at this point all fields of `column`  will have their values from before the try statement
563         * }
564         * </code></pre>
565         */
566        protected class Memento implements AutoCloseable {
567            private int width = Column.this.width;
568            private int padding = Column.this.padding;
569            private double alignment = Column.this.alignment;
570            private @NotNull Function<@Nullable Object, String> formatter = Column.this.formatter;
571            private @Nullable Object defaultValue = Column.this.defaultValue;
572
573            /**
574             * <p>Restore the mutable fields of the enclosing class to the values captured
575             * at the time of instantiation.</p>
576             *
577             * <p>To be used in a try-with-resources construct in RAII manner as follows:</p>
578             *
579             * <pre><code>
580             * \@SuppressWarnings("try") // to prevent the compiler from complaining
581             * void preserveColumnSettings() {
582             *     try(Column.Memento ignore = column.new Memento()) {
583             *         // do things that modify the fields `column`
584             *     }
585             *     // at this point all fields of `column`  will have their values from before the try statement
586             * }
587             * </code></pre>
588             */
589            @Override public void close() { restore(); }
590
591            /**
592             * Restore the mutable fields of the enclosing class to the values captured
593             * at the time of instantiation.
594             * @see #close()
595             */
596            protected void restore() {
597                Column.this.width = width;
598                Column.this.padding = padding;
599                Column.this.alignment = alignment;
600                Column.this.formatter = formatter;
601                Column.this.defaultValue = defaultValue;
602            }
603        }
604
605        /**
606         * Construct a new column with specified name, taking data from the specified index.
607         * @param name the name of this column (will be used in the table header and as key for mapped columns)
608         * @param index the index of the data value within a row
609         */
610        public Column(@NotNull String name, int index) {
611            this.name = name;
612            this.index = index;
613            width = name.length();
614        }
615
616        /**
617         * Convert a data value into string.
618         * @param value the input value (typically {@code row[column.index]})
619         * @return the formatted value (length should be less or equal to
620         *         the {@code column.width-column.padding})
621         */
622        @NotNull @Contract(pure = true)
623        protected String format(@Nullable Object value) {
624            return formatter.apply(value);
625        }
626
627        @Override
628        @Contract(pure = true)
629        public @NotNull String toString() {
630            return index + "-" + name;
631        }
632
633        @SuppressWarnings("unchecked")
634        @Contract(pure = true)
635        private static @NotNull String defaultFormatter(@Nullable Object value) {
636            // TODO: replace with multi-function in v0.2.0
637            if (value instanceof Callable) return Exceptions.rethrow((Callable) value).toString();
638            if (value instanceof Supplier) return ((Supplier) value).get().toString();
639            return String.valueOf(value);
640        }
641    }

643    /**
644     * <p>Provides a nice and compact DSL to build a set of columns.</p>
645     *
646     * <p>A known limitation is that layouts built with this builder will be
647     * isomorphic with the data rows (i.e. you can't have a 3 column layout,
648     * displaying 1st, 15th and 8th value of the data row). While technically
649     * achievable, I don't see the point, so if you need it feel free to extend
650     * this builder or write your own.</p>
651     */
652    public static class LayoutBuilder {
653        private final @NotNull Map<String, Column> columns = new LinkedHashMap<>();
654
655        /**
656         * Adds a column to the right, with the desired name and default attributes,
657         * displaying the next data index.
658         * @param columnName the name of the new column
659         * @return reference to {@code this} for chaining.
660         * @see #column(String, Consumer)
661         */
662        @NotNull
663        public LayoutBuilder column(@NotNull String columnName) {
664            return column(columnName, null);
665        }
666
667        /**
668         * Adds a column to the right, with the desired name and configurable attributes,
669         * displaying the next data index.
670         * @param columnName the name of the new column
671         * @param config a function that can configure the column. If {@code null} or missing,
672         *               the default attributes will be used.
673         * @return reference to {@code this} for chaining.
674         * @see #column(String, Consumer)
675         */
676        @NotNull
677        public LayoutBuilder column(@NotNull String columnName, @Nullable Consumer<Column> config) {
678            String name = columnName.trim();
679
680            if (columns.containsKey(name)) {
681                throw new IllegalArgumentException("Duplicate column '" + name + "'");
682            }
683
684            Column newColumn = new Column(name, columns.size());
685            columns.put(name, newColumn);
686
687            if (config != null) {
688                config.accept(newColumn);
689            }
690
691            return this;
692        }
693
694        /**
695         * Direct method to access all columns, can be used as lower level API to add and remove
696         * configured columns or to inspect and change their configuration.
697         * @return a map from column name to a column, ordered by their displayed order.
698         */
699        @NotNull @Contract(pure = true)
700        public Map<String, Column> getAllColumns() {
701            return columns;
702        }
703
704        /**
705         * <p>Create a strict data builder with the specified layout, that is
706         * used to build the data rows for the table. Each row should specify
707         * non-null values for all non-defaulted columns, and no extra data.</p>
708         *
709         * <p>The layout of the data builder is decoupled from this builder.
710         * This builder is still valid after creating a layout and can be used
711         * to create multiple layout objects.</p>
712         * @return the data builder.
713         * @see DataBuilder#buildTable()
714         */
715        @NotNull @Contract(pure = true)
716        public DataBuilder withData() {
717            return new DataBuilder(true, new ArrayList<>(columns.values()));
718        }
719
720        /**
721         * <p>Create a lenient data builder with the specified layout, that is
722         * used to build the data rows for the table. Each row should specify
723         * non-null values for all non-defaulted columns, but it may as well
724         * specify extra data.</p>
725         *
726         * <p>The layout of the data builder is decoupled from this builder.
727         * This builder is still valid after creating a layout and can be used
728         * to create multiple layout objects.</p>
729         *
730         * @return the data builder.
731         * @see DataBuilder#buildTable()
732         */
733        @NotNull @Contract(pure = true)
734        public DataBuilder withDataRelaxed() {
735            return new DataBuilder(false, new ArrayList<>(columns.values()));
736        }
737
738        @Override
739        @NotNull @Contract(pure = true)
740        public String toString() {
741            return String.format("TextTable.LayoutBuilder{columns=%s}", columns);
742        }
743    }
744
745    /**
746     * This builder can be used to build a dataset, that can be used
747     * together with the specified layout (a set of columns) to create
748     * a {@link TextTable table}.
749     */
750    public static class DataBuilder {
751        private final int maxColumnIndex;
752        private final boolean strict;
753        private final @NotNull List<@NotNull Column> columns;
754        private final @NotNull List<@NotNull List<?>> data = new ArrayList<>();
755
756        /**
757         * Creates a builder for the specified layout.
758         * @param strict if {@code true}, rows can not specify extra data (unmapped to columns)
759         * @param columns the layout for hte table (defensively copied internally).
760         */
761        public DataBuilder(boolean strict, @NotNull List<@NotNull Column> columns) {
762            // cloning in the LayoutBuilder.withData() as well, but better safe than sorry
763            this.columns = new ArrayList<>(columns);
764            this.strict = strict;
765            this.maxColumnIndex = columns.stream()
766                                         .mapToInt(it -> it.index).max()
767                                         .orElseThrow(()-> new IllegalArgumentException("Empty layout"));
768        }
769
770        /**
771         * Adds a single row to the dataset
772         * @param row the row of values to add
773         * @return reference to {@code this} for chaining.
774         * @throws IllegalArgumentException if the row does not match in the layout
775         */
776        @NotNull
777        public DataBuilder row(@NotNull List<@Nullable ?> row) {
778            return appendRowChecked(row, row);
779        }
780
781        /**
782         * Adds a single row to the dataset
783         * @param row the row of values to add
784         * @return reference to {@code this} for chaining.
785         * @throws IllegalArgumentException if the row does not match in the layout
786         */
787        @NotNull
788        public DataBuilder row(Object... row) { // we want not null row array, but nullable array components
789            return appendRowChecked(Arrays.asList(row), Arrays.deepToString(row));
790        }
791
792        /**
793         * Adds a single row to the dataset
794         * @param row the row of values to add keyed by name
795         * @return reference to {@code this} for chaining.
796         * @throws IllegalArgumentException if the row does not match in the layout
797         */
798        @NotNull
799        public DataBuilder row(@NotNull Map<@NotNull String, @Nullable Object> row) {
800            Object[] rowArray = new Object[maxColumnIndex+1];
801            for (Column column : columns) {
802                if (!row.containsKey(column.name) && column.defaultValue == null) continue;
803                rowArray[column.index] = row.getOrDefault(column.name, column.defaultValue);
804            }
805            return appendRowChecked(Arrays.asList(rowArray), row + "->" + Arrays.toString(rowArray));
806        }
807
808        /**
809         * Adds multiple rows to the dataset
810         * @param rows a collection of rows to add
811         * @return reference to {@code this} for chaining.
812         * @throws IllegalArgumentException if the row does not match in the layout
813         */
814        @NotNull
815        public DataBuilder rows(@NotNull Collection<@NotNull List<@Nullable ?>> rows) {
816            rows.forEach(this::row);
817            return this;
818        }
819
820        /**
821         * Adds multiple rows to the dataset
822         * @param rows a collection of rows to add
823         * @return reference to {@code this} for chaining.
824         * @throws IllegalArgumentException if the row does not match in the layout
825         */
826        @NotNull
827        public DataBuilder rows(@NotNull Object[]... rows) {
828            for (Object[] row : rows) row(row);
829            return this;
830        }
831
832        private DataBuilder appendRowChecked(List<?> row, @Nullable Object rowDescription) {
833            if (strict) {
834                if (row.size() != maxColumnIndex + 1) {
835                    throw new IllegalArgumentException(String.format(
836                            "Data doesn't match the columns:%n%d COLUMNS: %s%n%d DATA: %s",
837                            columns.size(), columns, row.size(), rowDescription
838                    ));
839                }
840            }
841            for (Column column : columns) {
842                if (column.defaultValue != null) continue;
843
844                if (column.index>=row.size()) {
845                    throw new IllegalArgumentException(String.format(
846                            "Column index out of range: %s%n%d COLUMNS: %s%n%d DATA: %s",
847                            column, columns.size(), columns, row.size(), rowDescription
848                    ));
849                }
850
851                if (row.get(column.index)==null) {
852                    throw new IllegalArgumentException(String.format(
853                            "Non default column is null: %s%n%d COLUMNS: %s%n%d DATA: %s",
854                            column, columns.size(), columns, row.size(), rowDescription
855                    ));
856                }
857            }
858            data.add(row);
859            return this;
860        }
861
862        /**
863         * Create a text table with the specified layout and data, that can be used to render
864         * the data in tabular format to any appendable (stream or buffer). Both the columns
865         * and data are defensively shallow-copied, which means that changes to the columns
866         * and rows will visible in the table.
867         * @return the text table.
868         * @see TextTable#format(int, Appendable)
869         */
870        @NotNull @Contract(pure = true)
871        public TextTable buildTable() {
872            return new TextTable(new ArrayList<>(columns), new ArrayList<>(data));
873        }
874
875        @Override
876        public String toString() {
877            return String.format(
878                    "DataBuilder(%d columns, %d rows) { %s }",
879                    columns.size(), data.size(), columns.stream().map(it -> it.name).collect(Collectors.joining(", "))
880            );
881        }
882    }
883
884    /**
885     * <p>Adds {@code length} characters of padding to the {@code out} parameter.
886     * The padding is filled by repeating the {@code pad} pattern as needed,
887     * trimming the last occurrence to size.</p>
888     * <p>For ecample, {@code pad(out, 8, "abc")} will result in <em>"abcabcab"</em>
889     * (8 characters) appened to the {@code out} parameter.</p>
890     * @param out where to put the padding.
891     * @param length how many characters to pad.
892     * @param pad what pattern to use for padding.
893     * @param <T> captures the type of the {@code out} parameter, so we can return it.
894     * @return the {@code out} parameter for chaining.
895     */
896    protected static <T extends Appendable> T pad(@NotNull T out, int length, @NotNull String pad) {
897        try {
898            int padLength = pad.length();
899            for (int i = 0; i < length; /* incremented in loop */) {
900                int charsToAppend = Math.min(padLength, length - i);
901                out.append(pad, 0, charsToAppend);
902                i += charsToAppend;
903            }
904            return out;
905        } catch (IOException e) {
906            return Exceptions.rethrow(e);
907        }
908    }
