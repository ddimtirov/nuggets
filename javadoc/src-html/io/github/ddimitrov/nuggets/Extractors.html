033/**
034 * <p>Provides utilities for <a href="#encapsulation">circumventing Java encapsulation</a>,
035 * as well as building blocks for <a href="#di">custom dependency injection</a>.
036 * All functions provided by this class are threadsafe.</p>
037 *
038 * <h2 id="encapsulation">Breaking Encapsulation</h2>
039 * <p>If you need to access non-public or final members of a class, you may use
040 * {@link #peekField(Object, Class, String, Class) peekField(target, targetType, fieldName, valueType)} and
041 * {@link #pokeField(Object, Class, String, Object)  pokeField(target, class, fieldName, value)}
042 * to bypass the Java encapsulation with minimal fuss.</p>
043 *
044 * <p> To access static members, just pass {@code null} for target. If you need to access instance
045 * fields, you may skip the {@code targetType} parameter and it will be assumed to be actual class
046 * of the {@code target} parameter.</p>
047 *
048 * <pre><code>
049 * // get access to Unsafe
050 * public static final Unsafe UNSAFE = Extractors.peekField(Unsafe.class, null, "theUnsafe", Unsafe.class);
051 *
052 * // swap the date format of existing object
053 * SimpleDateFormat sdf = Extractors.peekField(service, "dateFormat", SimpleDateFormat.class);
054 * Extractors.pokeField(sdf, "pattern", "yyyy/MM/dd");
055 * </code></pre>
056 *
057 * <p> In case you need to access  shadowed fields (i.e., field with the same name defined multiple times in
058 * the inheritance hierarchy, make sure you specify {@code targetType} to be a class where the field is
059 * declared (or at least visible according to the Java rules).</p>
060 *
061 * <p>Both {@code peek/pokeField()} are going through the full reflection API to resolve the
062 * field and assert security permissions for breaking the encapsulation. If you are repeatedly accessing the
063 * same fields, you can improve the performance at the expense of slightly more verbose code by using
064 * {@link #getAccessibleField(Class, String, boolean) getAccessibleField(type, fieldName, checkSuperclasses)}
065 * directly as follows:</p>
066 *
067 * <pre><code>
068 * // cache these
069 * private static final Field dateFormatField = Extractors.getAccessibleField(MyService.class, "dateFormat", true);
070 * private static final Field formatPatternField = Extractors.getAccessibleField(SimpleDateFormat.class, "pattern", true);
071 *
072 * // swap the date format of existing object (fast)
073 * try {
074 *     SimpleDateFormat sdf = (SimpleDateFormat) dateFormatField.get(service));
075 *     formatPatternField.set(sdf, "yyyy/MM/dd");
076 * } catch (IllegalAccessException e) {
077 *     Exceptions.rethrow(e);
078 * }
079 * </code></pre>
080 *
081 * <h2 id="di">Instantiation and injection utils</h2>
082 * <p>Sometimes, especially when we write a library, we may need to provide some functionality
083 * if certain library is present and provide a fall-back or even omit it, if it isn't.
084 * This is easy to do by using {@link #getClassIfPresent(String) getClassIfPresent(className)}: </p>
085 *
086 * <pre><code>
087 * templateResolver = Extractors.getClassIfPresent("groovy.lang.GroovyObject")
088 *     ? new GroovyTemplateResolver()
089 *     : new SimpleResolver("${", "}");
090 * </code></pre>
091 *
092 * <p>At times we would like some kind of DI functionality, but without bringing in a full-blown
093 * DI container, such as Guice, Dagger or Pico. This is when {@link #findInjectableConstructor(boolean, Constructor...)}
094 * and {@link #findInjectableMethods(Method...)} come handy. Once you find the things you want to invoke,
095 * you may easily assemble a signature using {@link #autowireByType(Executable, Function)} as follows:</p>
096 *
097 * <pre><code>
098 * public class SimpleDi {
099 *     private Map&lt;Class, Object&gt; instances = new HashMap&lt;&gt;();
100 * 
101 *     public &lt;T&gt; void bindSingleton(Class&lt;T&gt; type, T implementation)  {
102 *         instances.put(type, implementation);
103 *     }
104 * 
105 *     public &lt;T&gt; Collection&lt;T&gt; instancesOf(Class&lt;T&gt; type) {
106 *         return instances.values().stream()
107 *                         .filter(it -&gt; type.isAssignableFrom(it.getClass()))
108 *                         .map(type::cast)
109 *                         .collect(Collectors.toSet());
110 *     }
111 * 
112 *     public &lt;T&gt; T instantiate(Class&lt;T&gt; type) throws IllegalAccessException, InvocationTargetException, InstantiationException {
113 *         if (instances.containsKey(type)) { //noinspection unchecked
114 *             return (T) instances.get(type);
115 *         } else {
116 *             instances.put(type, null); // prevent instantiation loops
117 *         }
118 * 
119 *         Function&lt;Class&lt;?&gt;, Object&gt; paramResolver = t -&gt; Exceptions.rethrow(() -&gt; {
120 *             Object defaultValue = Extractors.defaultValue(t);
121 *             return defaultValue == null ? instantiate(t) : defaultValue;
122 *         });
123 * 
124 *         if ((type.getModifiers() &amp; (Modifier.ABSTRACT | Modifier.INTERFACE)) != 0) {
125 *             throw new InstantiationException("Only concrete classes supported!");
126 *         }
127 *         // instantiate object
128 *         Constructor&lt;T&gt; constructor = Extractors.findInjectableConstructor(false, type.getConstructors());
129 *         Object[] args = Extractors.autowireByType(constructor, paramResolver);
130 *         T instance = constructor.newInstance(args);
131 *         instances.put(type, instance);
132 * 
133 *         // call methods annotated with @Inject
134 *         for (Method method : Extractors.findInjectableMethods(type.getMethods())) {
135 *             method.invoke(instance, Extractors.autowireByType(method, paramResolver));
136 *         }
137 * 
138 *         return instance; // fully constructed and initialized
139 *     }
140 * } *
141 * ...
142 * // usage to bootstrap an app with plugin strategies and lifecycle management
143 * di = new SimpleDi();
144 * xyzzy = new XyzzyServiceImpl();
145 * di.bindSingleton(IXyzzyService.class, xyzzy);
146 * di.bindSingleton(IFooBarStrategy.class, new FooBarLocalImpl());
147 * app = di.instantiate(Application.class);
148 * di.instancesOf(XyzzyListener.class).forEach(it -&gt; xyzzy.addListener(it))
149 * app.run();
150 * di.instancesOf(Closeable.class).forEach(Closeable::close);
151 * app.shutdown();
152 * </code></pre>
153 */

181    /**
182     * <p>Extract the value of a field regardless of its visibility. If there are multiple shadowed fields with the
183     * given name, return the value of the first found when traversing the inheritance hierarchy starting from the
184     * {@code type}, until {@code Object}. </p>
185     * <p>You can use this method to easily get {@code Unsafe} and cache it for further usage, loke this:</p>
186     * <pre><code>
187     * public static final Unsafe UNSAFE = peekField(Unsafe.class, null, "theUnsafe", Unsafe.class);
188     * </code></pre>
189     *
190     * @param target the object containing the data. {@code null} if we are looking to extract a static field.
191     * @param type the type from which to start the search for the field. See {@link #getAccessibleField getAccessibleField(type, name, true)} for the precise meaning.
192     * @param fieldName the name of the field to extract.
193     * @param fieldType the expected type of the returned value. Needs to be a boxed (non-primitive) type.
194     *
195     * @param <T> inferring the return type.
196     * @return the extracted value of the requested field.
197     * @throws IllegalArgumentException if the {@code fieldType} is a primitive type.
198     *
199     * @see #getAccessibleField(Class, String, boolean)
200     * @see #peekField(Object, String, Class)
201     */
202    @Contract(pure=true)
203    public static <T> T peekField(
204            @Nullable Object target,
205            @NotNull Class<?> type,
206            @NotNull String fieldName,
207            @NotNull Class<T> fieldType
208    ) {
209
210        try {
211            if (fieldType.isPrimitive()) {
212                throw new IllegalArgumentException("Due to limitations of the Java Reflection API, you need to always use boxed types");
213            }
214            Field field = getAccessibleField(type, fieldName, true);
215            return fieldType.cast(field.get(target));
216        } catch (IllegalAccessException e) {
217            return rethrow(e);
218        }
219    }

223     * If there are multiple shadowed fields with the given name, set the value of the first found when
224     * traversing the inheritance hierarchy starting from the {@code type}, until {@code Object}. </p>
225     * <p>If you are looking to set a static field, or would like to set the value of a shadowed field, use the
226     * overloaded {@link #pokeField(Object, Class, String, Object)}</p>
227     *
228     * @param target the object on which to set the field. {@code null} if we are looking to set a static field.
229     * @param fieldName the name of the field to set.
230     * @param value the expected type of the value to set.
231     *
232     * @see #getAccessibleField(Class, String, boolean)
233     * @see #peekField(Object, String, Class)
234     * @see #pokeField(Object, Class, String, Object)

236    public static void pokeField(@NotNull Object target, @NotNull String fieldName, @Nullable Object value) {
237        pokeField(target, target.getClass(), fieldName, value);

240    /**
241     * <p>Extract the value of a field regardless of its visibility or {@code final} modifier. If there are
242     * multiple shadowed fields with the given name, set the value of the first found when traversing the
243     * inheritance hierarchy starting from the {@code type}, until {@code Object}. </p>
244     *
245     * @param target the object on which to set the field. {@code null} if we are looking to set a static field.
246     * @param type the type from which to start the search for the field. See {@link #getAccessibleField getAccessibleField(type, name, true)} for the precise meaning.
247     * @param fieldName the name of the field to set.
248     * @param value the expected type of the value to set.
249     *
250     * @see #getAccessibleField(Class, String, boolean)
251     * @see #peekField(Object, Class, String, Class)
252     * @see #pokeField(Object, String, Object)
253     */
254    public static void pokeField(
255            @Nullable Object target,
256            @NotNull Class<?> type,
257            @NotNull String fieldName,
258            @Nullable Object value
259    ) {
260        try {
261            Field field = getAccessibleField(type, fieldName, true);
262            field.set(target, value);
263        } catch (IllegalAccessException e) {
264            rethrow(e); // do not use #rethrow(Throwable, String) because we would go into mutual recursion
265        }
266    }

269    /**
270     * <p>Gets a {@link Field} instance that can be used to get and set values, regardless of the field visibility, final or
271     * static modifiers. Subject to security policy restrictions (if you don't know what that is, don't worry - you are
272     * probably safe).</p>
273     * <p>Sometimes there would be multiple fields with the same name declared in parent and sub-classes. In such cases
274     * make sure you specify the type of the exact class you want to get the field from.</p>
275     *
276     * @param type the type or a subtype of the class that holds the field.
277     * @param fieldName the name of the field we want.
278     * @param checkSuperclasses if {@code false}, will check only {@code type}; if {@code true},
279     *                          will check {@code type} and it's super classes all the way up
280     *                          to {@code Object}.
281     * @return the writable field instance
282     * @throws NoSuchFieldException if the field could not be found in the searched classes.
283     *
284     * @see #peekField(Object, String, Class)
285     * @see #pokeField(Object, String, Object)
286     * @see Field#setAccessible(boolean)
287     */
288    @Contract(pure=true) @SuppressWarnings("JavaDoc")
289    public static @NotNull Field getAccessibleField(@NotNull Class<?> type, @NotNull String fieldName, boolean checkSuperclasses) {
290        try {
291            Field field = type.getDeclaredField(fieldName);
292            field.setAccessible(true);
293
294            int modifiers = field.getModifiers();
295            if (!Modifier.isStatic(modifiers) || !Modifier.isFinal(modifiers)) {
296                return field;
297            }
298
299            // the rest of the method deals with static final fields - should be a rare use case
300            // http://zarnekow.blogspot.jp/2013/01/java-hacks-changing-final-fields.html
301            // http://www.cliffc.org/blog/2011/10/17/writing-to-final-fields-via-reflection
302            // http://www.cliffc.org/blog/2011/10/27/final-fields-part-2
303            Field modifiersField = field.getClass().getDeclaredField("modifiers");
304            boolean modifiersWasAccessible = modifiersField.isAccessible();
305            try {
306                modifiersField.setAccessible(true);
307                modifiersField.setInt(field, modifiers & ~Modifier.FINAL);
308                field.set(null, field.get(null)); // force the creation of accessor
309            } finally {
310                modifiersField.setInt(field, modifiers);
311                modifiersField.setAccessible(modifiersWasAccessible);
312            }
313
314            return field;
315        } catch (IllegalAccessException e) {
316            return rethrow(e);
317        }catch (NoSuchFieldException e) {
318            Class<?> superclass = type.getSuperclass();
319            return !checkSuperclasses || Object.class.equals(superclass)
320                    ? rethrow(e)
321                    : getAccessibleField(superclass, fieldName, true);
322        }
323    }

325    /**
326     * Converts Java boxed classes ({@code java.lang.Void}, {@code java.lang.Integer},
327     * {@code java.lang.Character}, etc.) to their primitive peers.
328     * If the {@code type} is already primitive, it is returned as is.
329     * Array classes are returned as-is (that may change in the future).
330     *
331     * @param type a class to convert
332     * @return the corresponding primitive class
333     */
334    @Contract(pure=true)
335    public static @NotNull Class<?> unboxClass(@NotNull Class<?> type) {
336        if (type.isPrimitive()) return type;
337        try {
338            // How to find the primitive type
339            // http://stackoverflow.com/questions/180097/dynamically-find-the-class-that-represents-a-primitive-java-type
340            return (Class) type.getField("TYPE").get(null);
341        } catch (NoSuchFieldException|IllegalAccessException e) {
342            return type; // unreachable - all primitives have TYPE
343        }
344    }

346    /**
347     * Converts primitive Java classes ({@code void.class}, {@code char.class},
348     * {@code int.class}) to their boxed ({@code java.lang.Void}, {@code java.lang.Character},
349     * {@code java.lang.Integer}, etc.)
350     * If the {@code type} is not a primitive primitive, it will be returned as is.
351     * Array classes are returned as-is (that may change in the future).
352     *
353     * @param type a class to convert
354     * @return the corresponding primitive class
355     */
356    @Contract(pure=true)
357    public static @NotNull Class<?> boxClass(@NotNull Class<?> type) {
358        if (!type.isPrimitive()) return type;
359
360        if (boolean.class.equals(type)) return Boolean.class;
361        if (byte.class.equals(type)) return Byte.class;
362        if (char.class.equals(type)) return Character.class;
363        if (double.class.equals(type)) return Double.class;
364        if (float.class.equals(type)) return Float.class;
365        if (int.class.equals(type)) return Integer.class;
366        if (long.class.equals(type)) return Long.class;
367        if (short.class.equals(type)) return Short.class;
368        if (void.class.equals(type)) return Void.class;
369
370        return type; // unreachable - all primitives are enumerated above
371    }

375     * Useful for providing optional functionality if certain library is in classpath.

393    /**
394     * Provides default value for each type, such as {@code false}, zero, empty array,
395     * no-args constructed instance, or {@code null} if there is no no-arg constructor
396     * or it threw exception.
397     *
398     * @param c type
399     * @return default value
400     */
401    @Contract(pure=true)
402    public static <T> @Nullable T defaultValue(@NotNull Class<T> c) {
403        try {
404            Constructor<?> constructor = c.getConstructor();
405            if (constructor !=null) return c.cast(constructor.newInstance());
406        } catch (Exception ignored) { }
407
408        if (c.isPrimitive()) {
409            @SuppressWarnings("unchecked")
410            Class<T> boxedEquivalent = (Class<T>) boxClass(c);
411            c = boxedEquivalent;
412        }
413
414        if (c.isArray())                            return c.cast(Array.newInstance(c.getComponentType(), 0));
415        if (BigDecimal.class.isAssignableFrom(c))   return c.cast(BigDecimal.ZERO);
416        if (BigInteger.class.isAssignableFrom(c))   return c.cast(BigInteger.ZERO);
417        if (Boolean.class.equals(c))                return c.cast(Boolean.FALSE);
418        if (Byte.class.equals(c))                   return c.cast((byte) 0);
419        if (Character.class.equals(c))              return c.cast('\0');
420        if (Double.class.equals(c))                 return c.cast(0d);
421        if (Float.class.equals(c))                  return c.cast(0f);
422        if (Integer.class.equals(c))                return c.cast(0);
423        if (Long.class.equals(c))                   return c.cast(0L);
424        if (Short.class.equals(c))                  return c.cast((short) 0);
425
426        if (Map.class.isAssignableFrom(c))          return c.cast(new LinkedHashMap<>());
427        if (Set.class.isAssignableFrom(c))          return c.cast(new LinkedHashSet<>());
428        if (List.class.isAssignableFrom(c))         return c.cast(new ArrayList<>());
429        if (Collection.class.isAssignableFrom(c))   return c.cast(new ArrayList<>());
430
431        return null; // void, classes without default constructor, etc.
432    }

434    /**
435     * <p>Finds the best constructor for dependency injection instantiation, based on number of arguments
436     * and {@code @Injected} annotation. If at least one candidate is annotated with {@code Injected}
437     * or the {@code requireAnnotation==true}, then only annotated candidates will be considered.
438     * Otherwise, return the candidates with largest number of formal arguments.</p>
439     * <br>
440     * <p>Example: find the most suitable constructor for DI (annotated with {@code Inject}, or largest number of params):</p>
441     * <pre><code>
442     * Constructor&lt;FooBar&gt; fbc = Extractors.findInjectableConstructor(false, FooBar.class.getConstructors());
443     * FooBar foobar = fbc.newInstance(Extractors.autowireByType(fbc, Extractors::defaultValue));
444     * </code></pre>
445     *
446     * @param requireAnnotation if {@code true}, consider only candidates annotated with {@code @Inject}
447     * @param candidates list of constructors to consider. In practice we expect an array of {@code Constructor<T>},
448     *                   though the actual arg is declared as {@code Constructor<T>} so it can be used without
449     *                   warnings with the return value of {@link Class#getConstructors()}.
450     *
451     * @param <T> inferring the return type.
452     * @return the most suitable constructor (most params, or annotated with {@code @Inject})
453     *
454     * @see #autowireByType(Executable, Function)
455     * @see #findInjectableMethods(Method...)
456     */
457    @Contract(pure=true)
458    public static <T> @NotNull Constructor<T> findInjectableConstructor(boolean requireAnnotation, @NotNull Constructor<?>... candidates) {
459        if (candidates.length==0) {
460            return rethrow(new NoSuchMethodException("Need at least one candidate for injectable constructor!"));
461        }
462
463        Constructor<?>[] injectable = findInjectableCandidates(requireAnnotation, null, candidates);
464        if (injectable.length>1) {
465            return rethrow(new NoSuchMethodException("Ambiguous injectable constructor: " + Arrays.toString(injectable)));
466        }
467
468        if (injectable.length<1) {
469            return rethrow(new NoSuchMethodException("No injectable constructor found in: " + Arrays.toString(candidates)));
470        }
471
472        @SuppressWarnings("unchecked")  // cast Constructor<?> to Constructor<T>
473        Constructor<T> injectableConstructor = (Constructor<T>) injectable[0];
474        return injectableConstructor;
475    }

478     * <p>Finds candidates for method dependency injection, based on {@code @Injected} annotation.</p>
479     *
480     * <p>Example: Find all methods annotated with {@code Inject} and invoke them:</p>
481     * <pre><code>
482     * for (Executable m : Extractors.findInjectableCandidates(foobar.getClass().geMethods())) {
483     *      m.invoke(foobar, Extractors.autowireByType(m, Extractors::defaultValue))
484     * }
485     * </code></pre>
486     *
487     * @param candidates list of methods to consider.
488     * @return all candidates annotated with {@code @Inject})
489     *
490     * @see #findInjectableConstructor
491     * @see #findInjectableCandidates(boolean, Predicate, Executable[])
492     * @see #autowireByType(Executable, Function)
493     */
494    @Contract(pure=true)
495    public static @NotNull Method[] findInjectableMethods(@NotNull Method... candidates) {
496        return findInjectableCandidates(true, it -> it.getParameterCount()>0, candidates);
497    }

500    /**
501     * <p>Finds constructors or methods for dependency injection, based on the number of arguments
502     * and {@code @Injected} annotation. If at least one candidate is annotated with {@code Injected}
503     * or the {@code requireAnnotation==true}, then only annotated candidates will be considered.
504     * Otherwise, return the candidates with largest number of formal arguments.</p>
505     * <br>
506     * <p>Example: find the most suitable constructor for DI (annotated with {@code Inject}, or largest number of params):</p>
507     * <pre><code>
508     * Constructor&lt;FooBar&gt; fbc = Extractors.findInjectableCandidates(false, FooBar.class.getConstructors());
509     * FooBar foobar = fbc.newInstance(Extractors.autowireByType(fbc, Extractors::defaultValue));
510     * </code></pre>
511     *
512     * @param requireAnnotation if {@code true}, consider only candidates annotated with {@code @Inject}
513     * @param filter     if not null and returns false for a candidate, the candidate is excluded
514     * @param candidates list of constructors to consider. In practice we expect an array of {@code Constructor<T>},
515     *                   though the actual arg is declared as {@code Constructor<T>} so it can be used without
516     *                   warnings with the return value of {@link Class#getConstructors()}.
517     *
518     * @param <T> inferring the constructor or method type.
519     * @return the most suitable constructor (most params, or annotated with {@code @Inject})
520     *
521     * @see #findInjectableConstructor(boolean, Constructor[])
522     * @see #findInjectableMethods(Method...)
523     * @see #autowireByType(Executable, Function)
524     */
525    @SafeVarargs
526    @Contract(pure=true)
527    public static <T extends Executable> @NotNull T[] findInjectableCandidates(boolean requireAnnotation, @Nullable Predicate<T> filter, @NotNull T... candidates) {
528        if (requireAnnotation&&INJECT_ANNOTATION==null) return Arrays.copyOfRange(candidates, 0, 0);
529
530        T[] selected = Arrays.copyOf(candidates, candidates.length);
531        int found = 0;
532        boolean annotated = requireAnnotation;
533
534        for (T candidate : candidates) {
535            if (filter!=null && !filter.test(candidate)) continue;
536
537            boolean selectedAnnotated = INJECT_ANNOTATION != null && candidate.getAnnotation(INJECT_ANNOTATION) != null;
538            if (selectedAnnotated) {
539                if (!annotated) {
540                    found = 0;
541                    annotated = true;
542                }
543                selected[found++] = candidate;
544            } else if (!annotated) {
545                int selectedArity = found==0 ? -1 : selected[found - 1].getParameterTypes().length;
546                int candidateArity = candidate.getParameterTypes().length;
547                if (selectedArity > candidateArity) continue;
548                if (selectedArity < candidateArity) found=0;
549                selected[found++] = candidate;
550            }
551        }
552
553        return Arrays.copyOf(selected, found);
554    }

556    /**
557     * Provide arguments to invoke a method or constructor, based on signature types and resolver function.
558     * Can be used to call arguments with null params, or implement simple DI layer.
559     *
560     * @param ctorOrMethod the constructor or method to call.
561     * @param paramResolver mapping function from type to implementation value can be as simple as {@code Map::get}.
562     * @return the signature array.
563     */
564    @Contract(pure=true)
565    public static @NotNull Object[] autowireByType(
566            @NotNull Executable ctorOrMethod,
567            @NotNull Function<Class<?>, Object> paramResolver
568    ) {
569        Class<?>[] parameterTypes = ctorOrMethod.getParameterTypes();
570        Object[] params = new Object[parameterTypes.length];
571        int i = 0;
572        for (Class<?> type : parameterTypes) {
573            params[i++] = paramResolver.apply(type);
574        }
575        return params;
576    }
