033/**
034 * <p><span class="badge green">Entry Point</span> Provides utilities for
035 * <a href="#encapsulation">circumventing Java encapsulation</a>,
036 * as well as building blocks for <a href="#di">custom dependency injection</a>.
037 * All functions provided by this class are threadsafe.</p>
038 *
039 * <h2 id="encapsulation">Breaking Encapsulation</h2>
040 * <p>If you need to access non-public or final members of a class, you may use
041 * {@link #peekField(Object, Class, String, Class) peekField(target, targetType, fieldName, valueType)} and
042 * {@link #pokeField(Object, Class, String, Object)  pokeField(target, class, fieldName, value)}
043 * to bypass the Java encapsulation with minimal fuss.</p>
044 *
045 * <p> To access static members, just pass {@code null} for target. If you need to access instance
046 * fields, you may skip the {@code targetType} parameter and it will be assumed to be actual class
047 * of the {@code target} parameter.</p>
048 *
049 * <pre><code>
050 * // get access to Unsafe
051 * public static final Unsafe UNSAFE = Extractors.peekField(Unsafe.class, null, "theUnsafe", Unsafe.class);
052 *
053 * // swap the date format of existing object
054 * SimpleDateFormat sdf = Extractors.peekField(service, "dateFormat", SimpleDateFormat.class);
055 * Extractors.pokeField(sdf, "pattern", "yyyy/MM/dd");
056 * </code></pre>
057 *
058 * <p> In case you need to access  shadowed fields (i.e., field with the same name defined multiple times in
059 * the inheritance hierarchy, make sure you specify {@code targetType} to be a class where the field is
060 * declared (or at least visible according to the Java rules).</p>
061 *
062 * <p>Both {@code peek/pokeField()} are going through the full reflection API to resolve the
063 * field and assert security permissions for breaking the encapsulation. If you are repeatedly accessing the
064 * same fields, you can improve the performance at the expense of slightly more verbose code by using
065 * {@link #getAccessibleField(Class, String, boolean) getAccessibleField(type, fieldName, checkSuperclasses)}
066 * directly as follows:</p>
067 *
068 * <pre><code>
069 * // cache these
070 * private static final Field dateFormatField = Extractors.getAccessibleField(MyService.class, "dateFormat", true);
071 * private static final Field formatPatternField = Extractors.getAccessibleField(SimpleDateFormat.class, "pattern", true);
072 *
073 * // swap the date format of existing object (fast)
074 * try {
075 *     SimpleDateFormat sdf = (SimpleDateFormat) dateFormatField.get(service));
076 *     formatPatternField.set(sdf, "yyyy/MM/dd");
077 * } catch (IllegalAccessException e) {
078 *     Exceptions.rethrow(e);
079 * }
080 * </code></pre>
081 *
082 * <h2 id="di">Instantiation and injection utils</h2>
083 * <p>Sometimes, especially when we write a library, we may need to provide some functionality
084 * if certain library is present and provide a fall-back or even omit it, if it isn't.
085 * This is easy to do by using {@link #getClassIfPresent(String) getClassIfPresent(className)}: </p>
086 *
087 * <pre><code>
088 * templateResolver = Extractors.getClassIfPresent("groovy.lang.GroovyObject")
089 *     ? new GroovyTemplateResolver()
090 *     : new SimpleResolver("${", "}");
091 * </code></pre>
092 *
093 * <p>At times we would like some kind of DI functionality, but without bringing in a full-blown
094 * DI container, such as Guice, Dagger or Pico. This is when {@link #findInjectableConstructor(boolean, Constructor...)}
095 * and {@link #findInjectableMethods(Method...)} come handy. Once you find the things you want to invoke,
096 * you may easily assemble a signature using {@link #autowireByType(Executable, Function)} as follows:</p>
097 *
098 * <pre><code>
099 * public class SimpleDi {
100 *     private Map&lt;Class, Object&gt; instances = new HashMap&lt;&gt;();
101 * 
102 *     public &lt;T&gt; void bindSingleton(Class&lt;T&gt; type, T implementation)  {
103 *         instances.put(type, implementation);
104 *     }
105 * 
106 *     public &lt;T&gt; Collection&lt;T&gt; instancesOf(Class&lt;T&gt; type) {
107 *         return instances.values().stream()
108 *                         .filter(it -&gt; type.isAssignableFrom(it.getClass()))
109 *                         .map(type::cast)
110 *                         .collect(Collectors.toSet());
111 *     }
112 * 
113 *     public &lt;T&gt; T instantiate(Class&lt;T&gt; type) throws IllegalAccessException, InvocationTargetException, InstantiationException {
114 *         if (instances.containsKey(type)) { //noinspection unchecked
115 *             return (T) instances.get(type);
116 *         } else {
117 *             instances.put(type, null); // prevent instantiation loops
118 *         }
119 * 
120 *         Function&lt;Class&lt;?&gt;, Object&gt; paramResolver = t -&gt; Exceptions.rethrow(() -&gt; {
121 *             Object defaultValue = Extractors.defaultValue(t);
122 *             return defaultValue == null ? instantiate(t) : defaultValue;
123 *         });
124 * 
125 *         if ((type.getModifiers() &amp; (Modifier.ABSTRACT | Modifier.INTERFACE)) != 0) {
126 *             throw new InstantiationException("Only concrete classes supported!");
127 *         }
128 *         // instantiate object
129 *         Constructor&lt;T&gt; constructor = Extractors.findInjectableConstructor(false, type.getConstructors());
130 *         Object[] args = Extractors.autowireByType(constructor, paramResolver);
131 *         T instance = constructor.newInstance(args);
132 *         instances.put(type, instance);
133 * 
134 *         // call methods annotated with @Inject
135 *         for (Method method : Extractors.findInjectableMethods(type.getMethods())) {
136 *             method.invoke(instance, Extractors.autowireByType(method, paramResolver));
137 *         }
138 * 
139 *         return instance; // fully constructed and initialized
140 *     }
141 * } *
142 * ...
143 * // usage to bootstrap an app with plugin strategies and lifecycle management
144 * di = new SimpleDi();
145 * xyzzy = new XyzzyServiceImpl();
146 * di.bindSingleton(IXyzzyService.class, xyzzy);
147 * di.bindSingleton(IFooBarStrategy.class, new FooBarLocalImpl());
148 * app = di.instantiate(Application.class);
149 * di.instancesOf(XyzzyListener.class).forEach(it -&gt; xyzzy.addListener(it))
150 * app.run();
151 * di.instancesOf(Closeable.class).forEach(Closeable::close);
152 * app.shutdown();
153 * </code></pre>
154 */

182    /**
183     * <p>Extract the value of a field regardless of its visibility. If there are multiple shadowed fields with the
184     * given name, return the value of the first found when traversing the inheritance hierarchy starting from the
185     * {@code type}, until {@code Object}. </p>
186     *
187     * @param target the object containing the data. {@code null} if we are looking to extract a static field.
188     * @param type the type from which to start the search for the field. See {@link #getAccessibleField getAccessibleField(type, name, true)} for the precise meaning.
189     * @param fieldName the name of the field to extract.
190     * @param fieldType the expected type of the returned value. Needs to be a boxed (non-primitive) type.
191     *
192     * @param <T> inferring the return type.
193     * @return the extracted value of the requested field.
194     * @throws IllegalArgumentException if the {@code fieldType} is a primitive type.
195     *
196     * @see #getAccessibleField(Class, String, boolean)
197     * @see #peekField(Object, String, Class)
198     */
199    @Contract(pure=true)
200    public static <T> T peekField(
201            @Nullable Object target,
202            @NotNull Class<?> type,
203            @Identifier @NotNull String fieldName,
204            @NotNull Class<T> fieldType
205    ) {
206
207        try {
208            if (fieldType.isPrimitive()) {
209                throw new IllegalArgumentException("Due to limitations of the Java Reflection API, you need to always use boxed types");
210            }
211            Field field = getAccessibleField(type, fieldName, true);
212            return fieldType.cast(field.get(target));
213        } catch (IllegalAccessException e) { return doSneakyThrow(e); }
214    }

220     * <p>If you are looking to set a static field, or would like to set the value of a shadowed field, use the

222     *

226     *

236     * <p>Modify the value of a field regardless of its visibility or {@code final} modifier. If there are
237     * multiple shadowed fields with the given name, set the value of the first found when traversing the
238     * inheritance hierarchy starting from the {@code type}, until {@code Object}. </p>
239     *
240     * @param target the object on which to set the field. {@code null} if we are looking to set a static field.
241     * @param type the type from which to start the search for the field. See {@link #getAccessibleField getAccessibleField(type, name, true)} for the precise meaning.
242     * @param fieldName the name of the field to set.
243     * @param value the new value to set.
244     *
245     * @see #getAccessibleField(Class, String, boolean)
246     * @see #peekField(Object, Class, String, Class)
247     * @see #pokeField(Object, String, Object)
248     */
249    public static void pokeField(
250            @Nullable Object target,
251            @NotNull Class<?> type,
252            @Identifier @NotNull String fieldName,
253            @Nullable Object value
254    ) {
255        try {
256            Field field = getAccessibleField(type, fieldName, true);
257            field.set(target, value);
258        } catch (IllegalAccessException e) { doSneakyThrow(e); }

262    /**
263     * <p>Gets a {@link Field} instance that can be used to get and set values, regardless of the field visibility, final or
264     * static modifiers. Subject to security policy restrictions (if you don't know what that is, don't worry - you are
265     * probably safe).</p>
266     * <p>Sometimes there would be multiple fields with the same name declared in parent and sub-classes. In such cases
267     * make sure you specify the type of the exact class you want to get the field from.</p>
268     *
269     * @param type the type or a subtype of the class that holds the field.
270     * @param fieldName the name of the field we want.
271     * @param checkSuperclasses if {@code false}, will check only {@code type}; if {@code true},
272     *                          will check {@code type} and it's super classes all the way up
273     *                          to {@code Object}.
274     * @return the writable field instance
275     * @throws NoSuchFieldException if the field could not be found in the searched classes.
276     *
277     * @see #peekField(Object, String, Class)
278     * @see #pokeField(Object, String, Object)
279     * @see Field#setAccessible(boolean)
280     */
281    @Contract(pure=true) @SuppressWarnings("JavaDoc")
282    public static @NotNull Field getAccessibleField(@NotNull Class<?> type, @Identifier @NotNull String fieldName, boolean checkSuperclasses) {
283        try {
284            Field field = type.getDeclaredField(fieldName);
285            field.setAccessible(true);
286
287            int modifiers = field.getModifiers();
288            if (!Modifier.isStatic(modifiers) || !Modifier.isFinal(modifiers)) {
289                return field;
290            }
291
292            // the rest of the method deals with static final fields - should be a rare use case
293            // http://zarnekow.blogspot.jp/2013/01/java-hacks-changing-final-fields.html
294            // http://www.cliffc.org/blog/2011/10/17/writing-to-final-fields-via-reflection
295            // http://www.cliffc.org/blog/2011/10/27/final-fields-part-2
296            Field modifiersField = field.getClass().getDeclaredField("modifiers");
297            boolean modifiersWasAccessible = modifiersField.isAccessible();
298            try {
299                modifiersField.setAccessible(true);
300                modifiersField.setInt(field, modifiers & ~Modifier.FINAL);
301                field.set(null, field.get(null)); // force the creation of accessor
302            } finally {
303                modifiersField.setInt(field, modifiers);
304                modifiersField.setAccessible(modifiersWasAccessible);
305            }
306
307            return field;
308        } catch (NoSuchFieldException e) {
309            Class<?> superclass = type.getSuperclass();
310            return !checkSuperclasses || Object.class.equals(superclass)
311                    ? doSneakyThrow(e)
312                    : getAccessibleField(superclass, fieldName, true);
313        } catch (IllegalAccessException e) { return doSneakyThrow(e); }
314    }

316    /**
317     * Converts Java boxed classes ({@code java.lang.Void}, {@code java.lang.Integer},
318     * {@code java.lang.Character}, etc.) to their primitive peers.
319     * If the {@code type} is already primitive, it is returned as is.
320     * Array classes are returned as-is (that may change in the future).
321     *
322     * @param type a class to convert
323     * @return the corresponding primitive class
324     */
325    @Contract(pure=true)
326    public static @NotNull Class<?> unboxClass(@NotNull Class<?> type) {
327        if (type.isPrimitive()) return type;
328        try {
329            // How to find the primitive type
330            // http://stackoverflow.com/questions/180097/dynamically-find-the-class-that-represents-a-primitive-java-type
331            return (Class) type.getField("TYPE").get(null);
332        } catch (NoSuchFieldException|IllegalAccessException e) {
333            return type; // unreachable - all primitives have TYPE
334        }
335    }

337    /**
338     * Converts primitive Java classes ({@code void.class}, {@code char.class},
339     * {@code int.class}) to their boxed ({@code java.lang.Void}, {@code java.lang.Character},
340     * {@code java.lang.Integer}, etc.)
341     * If the {@code type} is not a primitive primitive, it will be returned as is.
342     * Array classes are returned as-is (that may change in the future).
343     *
344     * @param type a class to convert
345     * @return the corresponding primitive class
346     */
347    @Contract(pure=true)
348    public static @NotNull Class<?> boxClass(@NotNull Class<?> type) {
349        if (!type.isPrimitive()) return type;
350
351        if (boolean.class.equals(type)) return Boolean.class;
352        if (byte.class.equals(type)) return Byte.class;
353        if (char.class.equals(type)) return Character.class;
354        if (double.class.equals(type)) return Double.class;
355        if (float.class.equals(type)) return Float.class;
356        if (int.class.equals(type)) return Integer.class;
357        if (long.class.equals(type)) return Long.class;
358        if (short.class.equals(type)) return Short.class;
359        if (void.class.equals(type)) return Void.class;
360
361        return type; // unreachable - all primitives are enumerated above
362    }

384    /**
385     * Provides default value for each type, such as {@code false}, zero, empty array,
386     * no-args constructed instance, or {@code null} if there is no no-arg constructor
387     * or it threw exception.
388     *
389     * @param c type
390     * @param <T> inferring the return type.
391     * @return default value
392     */
393    @Contract(pure=true)
394    public static <T> @Nullable T defaultValue(@NotNull Class<T> c) {
395        try {
396            Constructor<?> constructor = c.getConstructor();
397            if (constructor !=null) return c.cast(constructor.newInstance());
398        } catch (Exception ignored) { }
399
400        if (c.isPrimitive()) {
401            @SuppressWarnings("unchecked")
402            Class<T> boxedEquivalent = (Class<T>) boxClass(c);
403            c = boxedEquivalent;
404        }
405
406        if (c.isArray())                            return c.cast(Array.newInstance(c.getComponentType(), 0));
407        if (BigDecimal.class.isAssignableFrom(c))   return c.cast(BigDecimal.ZERO);
408        if (BigInteger.class.isAssignableFrom(c))   return c.cast(BigInteger.ZERO);
409        if (Boolean.class.equals(c))                return c.cast(Boolean.FALSE);
410        if (Byte.class.equals(c))                   return c.cast((byte) 0);
411        if (Character.class.equals(c))              return c.cast('\0');
412        if (Double.class.equals(c))                 return c.cast(0d);
413        if (Float.class.equals(c))                  return c.cast(0f);
414        if (Integer.class.equals(c))                return c.cast(0);
415        if (Long.class.equals(c))                   return c.cast(0L);
416        if (Short.class.equals(c))                  return c.cast((short) 0);
417
418        if (Map.class.isAssignableFrom(c))          return c.cast(new LinkedHashMap<>());
419        if (Set.class.isAssignableFrom(c))          return c.cast(new LinkedHashSet<>());
420        if (List.class.isAssignableFrom(c))         return c.cast(new ArrayList<>());
421        if (Collection.class.isAssignableFrom(c))   return c.cast(new ArrayList<>());
422
423        return null; // void, classes without default constructor, etc.
424    }

426    /**
427     * <p>Finds the best constructor for dependency injection instantiation, based on number of arguments
428     * and {@code @Injected} annotation. If at least one candidate is annotated with {@code Injected}
429     * or the {@code requireAnnotation==true}, then only annotated candidates will be considered.
430     * Otherwise, return the candidates with largest number of formal arguments.</p>
431     * <br>
432     * <p>Example: find the most suitable constructor for DI (annotated with {@code Inject}, or largest number of params):</p>
433     * <pre><code>
434     * Constructor&lt;FooBar&gt; fbc = Extractors.findInjectableConstructor(false, FooBar.class.getConstructors());
435     * FooBar foobar = fbc.newInstance(Extractors.autowireByType(fbc, Extractors::defaultValue));
436     * </code></pre>
437     *
438     * @param requireAnnotation if {@code true}, consider only candidates annotated with {@code @Inject}
439     * @param candidates list of constructors to consider. In practice we expect an array of {@code Constructor<T>},
440     *                   though the actual arg is declared as {@code Constructor<T>} so it can be used without
441     *                   warnings with the return value of {@link Class#getConstructors()}.
442     *
443     * @param <T> inferring the return type.
444     * @return the most suitable constructor (most params, or annotated with {@code @Inject})
445     *
446     * @see #autowireByType(Executable, Function)
447     * @see #findInjectableMethods(Method...)
448     */
449    @Contract(pure=true)
450    public static <T> @NotNull Constructor<T> findInjectableConstructor(boolean requireAnnotation, @NotNull Constructor<?>... candidates) {
451        if (candidates.length==0) {
452            return doSneakyThrow(new NoSuchMethodException("Need at least one candidate for injectable constructor!"));
453        }
454
455        Constructor<?>[] injectable = findInjectableCandidates(requireAnnotation, null, candidates);
456        if (injectable.length>1) {
457            return doSneakyThrow(new NoSuchMethodException("Ambiguous injectable constructor: " + Arrays.toString(injectable)));
458        }
459
460        if (injectable.length<1) {
461            return doSneakyThrow(new NoSuchMethodException("No injectable constructor found in: " + Arrays.toString(candidates)));
462        }
463
464        @SuppressWarnings("unchecked")  // cast Constructor<?> to Constructor<T>
465        Constructor<T> injectableConstructor = (Constructor<T>) injectable[0];
466        return injectableConstructor;
467    }

470     * <p>Finds candidates for method dependency injection, based on {@code @Injected} annotation.</p>
471     *
472     * <p>Example: Find all methods annotated with {@code Inject} and invoke them:</p>
473     * <pre><code>
474     * for (Executable m : Extractors.findInjectableCandidates(foobar.getClass().geMethods())) {
475     *      m.invoke(foobar, Extractors.autowireByType(m, Extractors::defaultValue))
476     * }
477     * </code></pre>
478     *
479     * @param candidates list of methods to consider.
480     * @return all candidates annotated with {@code @Inject})
481     *
482     * @see #findInjectableConstructor
483     * @see #findInjectableCandidates(boolean, Predicate, Executable[])
484     * @see #autowireByType(Executable, Function)
485     */
486    @Contract(pure=true)
487    public static @NotNull Method[] findInjectableMethods(@NotNull Method... candidates) {
488        return findInjectableCandidates(true, it -> it.getParameterCount()>0, candidates);
489    }

492    /**
493     * <p>Finds constructors or methods for dependency injection, based on the number of arguments
494     * and {@code @Injected} annotation. If at least one candidate is annotated with {@code Injected}
495     * or the {@code requireAnnotation==true}, then only annotated candidates will be considered.
496     * Otherwise, return the candidates with largest number of formal arguments.</p>
497     * <br>
498     * <p>Example: find the most suitable constructor for DI (annotated with {@code Inject}, or largest number of params):</p>
499     * <pre><code>
500     * Constructor&lt;FooBar&gt; fbc = Extractors.findInjectableCandidates(false, FooBar.class.getConstructors());
501     * FooBar foobar = fbc.newInstance(Extractors.autowireByType(fbc, Extractors::defaultValue));
502     * </code></pre>
503     *
504     * @param requireAnnotation if {@code true}, consider only candidates annotated with {@code @Inject}
505     * @param filter     if not null and returns false for a candidate, the candidate is excluded
506     * @param candidates list of constructors to consider. In practice we expect an array of {@code Constructor<T>},
507     *                   though the actual arg is declared as {@code Constructor<T>} so it can be used without
508     *                   warnings with the return value of {@link Class#getConstructors()}.
509     *
510     * @param <T> inferring the constructor or method type.
511     * @return the most suitable constructor (most params, or annotated with {@code @Inject})
512     *
513     * @see #findInjectableConstructor(boolean, Constructor[])
514     * @see #findInjectableMethods(Method...)
515     * @see #autowireByType(Executable, Function)
516     */
517    @SafeVarargs
518    @Contract(pure=true)
519    public static <T extends Executable> @NotNull T[] findInjectableCandidates(boolean requireAnnotation, @Nullable Predicate<T> filter, @NotNull T... candidates) {
520        if (requireAnnotation&&INJECT_ANNOTATION==null) return Arrays.copyOfRange(candidates, 0, 0);
521
522        T[] selected = Arrays.copyOf(candidates, candidates.length);
523        int found = 0;
524        boolean annotated = requireAnnotation;
525
526        for (T candidate : candidates) {
527            if (filter!=null && !filter.test(candidate)) continue;
528
529            boolean selectedAnnotated = INJECT_ANNOTATION != null && candidate.getAnnotation(INJECT_ANNOTATION) != null;
530            if (selectedAnnotated) {
531                if (!annotated) {
532                    found = 0;
533                    annotated = true;
534                }
535                selected[found++] = candidate;
536            } else if (!annotated) {
537                int selectedArity = found==0 ? -1 : selected[found - 1].getParameterTypes().length;
538                int candidateArity = candidate.getParameterTypes().length;
539                if (selectedArity > candidateArity) continue;
540                if (selectedArity < candidateArity) found=0;
541                selected[found++] = candidate;
542            }
543        }
544
545        return Arrays.copyOf(selected, found);
546    }

549     * Provide arguments to invoke a method or constructor, based on signature types and resolver function.
550     * Can be used to call arguments with null params, or implement simple DI layer.
551     *
552     * @param ctorOrMethod the constructor or method to call.
553     * @param paramResolver mapping function from type to implementation value can be as simple as {@code Map::get}.
554     * @return the signature array.
555     */
556    @Contract(pure=true)
557    public static @NotNull Object[] autowireByType(
558            @NotNull Executable ctorOrMethod,
559            @NotNull Function<Class<?>, Object> paramResolver
560    ) {
561        Class<?>[] parameterTypes = ctorOrMethod.getParameterTypes();
562        Object[] params = new Object[parameterTypes.length];
563        int i = 0;
564        for (Class<?> type : parameterTypes) {
565            params[i++] = paramResolver.apply(type);
566        }
567        return params;
568    }

575    /**
576     * <p>Super-type linearization is an algorithm used to obtain an ordered
577     * sequence of all super-types in the presence of multiple inheritance.
578     * The list should provide the following guarantees:</p>
579     *
580     * <ol>
581     * <li>All classes in the result should be assignable from {@code type}
582     *     (i.e. {@code it} is supertype of {@code type)}, and the result
583     *     should contain all supertypes of {@code type}</li>
584     * <li>The linearization result of any supertype should result in a subset
585     *     of the subtype linearization, preserving the same relative order.</li>
586     * <li>The ordering is by "specificity".</li>
587     * </ol>
588     *
589     * <p>An informal definition of specificity ordering is <em>"given two types -
590     * {@code A} and {@code B}, more specific is the type you would expect to match
591     * if you had overloads with both types and an argument of type {@code type}"</em>.
592     * It is a concept easy to understand intuitively, and
593     * <a href="https://blogs.oracle.com/mcimadamore/entry/testing_overload_resolution">surprisingly tricky</a>
594     * around the edge cases.</p>
595     *
596     * <p>In practice I currently use the following algorithm, which is a simple and
597     * good enough approximation to the JLS rules (see the referenced link):</p>
598     * <ol>
599     * <li>Superclasses come first, in the order of inheritance, except for
600     *     Object which comes last. (That is, if {@code type} is not an interface)</li>
601     * <li>In the middle are all interfaces in breadth-first, left-to-right order.
602     *     On multiple inheritances from the same interface, the first wins.</li>
603     * </ol>
604     * @param type the type we are linearizing
605     * @return a sequence of all supertypes of the type, ordered by specificity
606     * @see <a href="http://docs.oracle.com/javase/specs/jls/se8/html/jls-15.html#jls-15.12.2.5">JLS 15.12.2.5 - specificity rules for method resolution</a>
607     * @see <a href="https://en.wikipedia.org/wiki/C3_linearization">C3 alorithm for linearization</a>
608     */
609    public static Iterable<? extends Class<?>> linearized(@NotNull Class<?> type) {
610        LinkedHashSet<Class<?>> linearized = new LinkedHashSet<>(); // both ordering and uniqueness are important
611
612        // step 1: put super-classes first, up to but excluding Object (classes take precedence to interfaces)
613            for (Class<?> c = type; c != Object.class; c = c.getSuperclass()) {
614                linearized.add(c);
615                if (type.isInterface()) break; // interfaces are handled in the next step (still we need to put the seed)
616            }
617
618        // step 2: collect all interfaces in left-to-right breadth-first fashion
619        Collection<Class<?>> prev = new ArrayList<>(linearized); // what we added in the previous round
620        Collection<Class<?>> curr = new ArrayList<>();           // what we are going to add on this round
621        do {
622            for (Class<?> t : prev) {   // add the super interfaces of all types found in the prev round
623                curr.addAll(Arrays.asList(t.getInterfaces()));
624            }
625
626            curr.removeAll(linearized); // remove the types already in the resulting set
627            linearized.addAll(curr);    // add the new types to the resulting set
628
629            // prepare for the curr iteration: swap prev<->curr, clear the curr
630            Collection<Class<?>> temp = curr;
631            curr = prev;
632            prev = temp;
633            curr.clear();
634        } while(!prev.isEmpty());
635
636        // step 3: add Object last (or it would be more specific than all of then interfaces, which is counter-intuitive)
637        if (!type.isInterface()) {
638            linearized.add(Object.class);
639        }
640
641        return linearized;
642    }

644    /**
645     * <p>Walks through all accessible members (fields, methods or constructors)
646     * of a given type and its supertypes, regardless of their visibility or
647     * modifiers. Members of derived classes are processed before the members of
648     * base classes.</p>
649     * <p>Each member is made {@link AccessibleObject#setAccessible(boolean) accessible}
650     * before being passed to the processor, so you can  invoke operations
651     * on it directly.</p>
652     * @param <T> the inferred type of the accessible member.
653     * @param extractor a function extracting a list of accessible members
654     *                  from a class. Typically one of
655     *                  {@link Class#getDeclaredConstructors()},
656     *                  {@link Class#getDeclaredFields()},
657     *                  {@link Class#getDeclaredMethods()}
658     * @param startType the class from which we should start processing
659     *                  working up the inheritance hierarchy towards
660     *                  the {@code stopClass}
661     * @param stopType the class where we should stop processing
662     *                  (if {@code null}, we will stop at {@code Object})
663     * @param processor typically lambda that will process each accessible
664     *                  member. If needed to access {@code o} - do it through
665     *                  the closure.
666     */
667    public static <T extends AccessibleObject> void eachAccessible(
668            @NotNull Function<Class<?>, T[]> extractor,
669            Class<?> startType, @Nullable Class<Object> stopType,
670            @NotNull Extractors.AccessibleMemberProcessor<@NotNull T> processor
671    ) {
672        Class<?> type = startType;
673        try {
674            Class<Object> stop = stopType == null ? Object.class : stopType;
675            while (type!=null && type !=stop) {
676                for (T it : extractor.apply(type)) {
677                    it.setAccessible(true);
678                    processor.process(it);
679                }
680                type = type.getSuperclass();
681            }
682        } catch (Exception e) {
683            Exceptions.rethrow(e);
684        }
685    }
686
687    @FunctionalInterface
688    public interface AccessibleMemberProcessor<T extends AccessibleObject> {
689        void process(T subject) throws Exception;
690    }
