017package io.github.ddimitrov.nuggets;
018
019import org.intellij.lang.annotations.Identifier;
020import org.jetbrains.annotations.Contract;
021import org.jetbrains.annotations.NotNull;
022import org.jetbrains.annotations.Nullable;
023
024import java.util.Objects;
025import java.util.function.Consumer;
026import java.util.function.Function;
027
028/**
029 * <p><span class="badge green">Entry Point</span> Provides a concise and expressive way
030 * for performing multiple reflective access operations, building on the {@code Extractors}
031 * class. As a reflection proxy has a final association with a single object, its API provides
032 * the (currently unused) opportunity for optimizations, such as memoizing methods annotated
033 * with {@code Contract(@pure=true)} and caching accessors and method/var handles.</p>
034 *
035 */
036public class ReflectionProxy {
037    private final @NotNull Object delegate;
038    private final @NotNull Class<?> type;
039
040    /**
041     * Constructor for derived classes (normally you should use {@link #wrap(Object)}).
042     * @param delegate the object whose members we would access reflectively.
043     * @param type the type used for member resolution in case of shadowed members.
044     */
045    protected ReflectionProxy(@NotNull Object delegate, @NotNull Class<?> type) {
046        this.delegate = type.cast(delegate);
047        this.type = type;
048    }
049
050    /**
051     * Wraps an object in reflection proxy.
052     * @param o an object to  be wrapped.
053     * @return a reflection proxy for {@code o};
054     *         if {@code o} is {@code null}, {@code null} will be returned;
055     *         if {@code o} is a reflection proxy it will be returned without further wrapping
056     */
057    @Contract(value = "null->null;!null->!null", pure = true)
058    public static @Nullable ReflectionProxy wrap(@Nullable Object o) {
059        return o==null || o instanceof ReflectionProxy
060                ? (ReflectionProxy) o
061                : new ReflectionProxy(o, o.getClass());
062    }
063
064    /**
065     * Returns the wrapped object and casts it to the specified type.
066     *
067     * @param type the expected type
068     * @param <V> a type inference generic parameter
069     * @return the wrapped object
070     */
071    @SuppressWarnings("unchecked")
072    @Contract(pure = true)
073    public <V> @NotNull V unwrap(@NotNull Class<V> type) {
074        return (V) Extractors.boxClass(type).cast(delegate);
075    }
076
077    /**
078     * Returns the wrapped object
079     *
080     * @return the wrapped object
081     */
082    public @NotNull Object unwrap() {
083        return delegate;
084    }
085
086    /**
087     * Changes the {@code type}, used to resolve shadowed members.
088     * @param type the new type, to be used to resolve shadowed members.
089     * @return an adapted reflection proxy instance
090     */
091    @Contract(pure = true)
092    public @NotNull ReflectionProxy resolvingAtType(@NotNull Class<?> type) {
093        return new ReflectionProxy(delegate, type);
094    }
095
096    /**
097     * <p>Invokes a (potentially private) method on the wrapped object and returns the result
098     * wrapped in a new reflection proxy.</p>
099     *
100     * <p>Known deficiency - if a static method is shadowed by an instance method in the same
101     * type, the instance method will be preferred and there is no way to invoke the static
102     * one using this class.</p>
103     *
104     * @param name the name of the method to invoke
105     * @param args the actual arguments for the method
106     * @return the {@link #wrap(Object) wrapped} method return value
107     */
108    public /*@Nullable*/ ReflectionProxy invoke(@NotNull @Identifier String name, @Nullable Object... args) {
109        return wrap(Extractors.invokeMethod(delegate, type, name, Object.class, args));
110    }
111
112    /**
113     * <p>Gets a field and returns the value wrapped in a new reflection proxy.</p>
114     *
115     * <p>Known deficiency - if a static field is shadowed by an instance field in the same
116     * type, the instance field will be preferred and there is no way to access the static
117     * one using this class.</p>
118     *
119     * @param name the name of the field to get
120     * @return the {@link #wrap(Object) wrapped} value
121     */
122    public /*@Nullable*/ ReflectionProxy get(@NotNull @Identifier String name) {
123        return wrap(Extractors.peekField(delegate, type, name, Object.class));
124    }
125
126    /**
127     * <p>Sets a field.</p>
128     *
129     * <p>Known deficiency - if a static field is shadowed by an instance field in the same
130     * type, the instance field will be preferred and there is no way to access the static
131     * one using this class.</p>
132     *
133     * @param name the name of the field to set
134     * @param value the value to assign
135     */
136    public void set(@NotNull @Identifier String name, @Nullable Object value) {
137        Extractors.pokeField(delegate, type, name, value);
138    }
139
140    /**
141     * Transform this reflection proxy to another reflection proxy - useful for chaining when
142     * the desired operation is not get or invoke.
143     *
144     * @param c function taking a reflection proxy as argument and returning object,
145     *          which will be {@link #wrap(Object) wrapped} automaticlaly as needed.
146     *
147     * @return the <em>wrapped</em> result of {@code c}
148     */
149    public /*@Nullable*/ ReflectionProxy map(@NotNull Function<ReflectionProxy, Object> c) {
150        return wrap(c.apply(this));
151    }
152
153    /**
154     * Do something with this reflection proxy in the middle of a method-call chain.
155     * @param c the operations to perform on this reflection proxy.
156     * @return this instance
157     */
158    public @NotNull ReflectionProxy tap(@NotNull Consumer<ReflectionProxy> c) {
159        c.accept(this);
160        return this;
161    }
162
163    @Override
164    public int hashCode() {
165        return Objects.hash(delegate);
166    }
167
168    @Override
169    public boolean equals(Object o) {
170        if (this == o) return true;
171        if (!(o instanceof ReflectionProxy)) return false;
172        ReflectionProxy that = (ReflectionProxy) o;
173        return Objects.equals(delegate, that.delegate) && Objects.equals(type, that.type);
174    }
175
176    @Override
177    public String toString() {
178        StringBuilder sb = new StringBuilder().append("rpx*").append(delegate).append(": ").append(delegate.getClass().getSimpleName());
179        if (!delegate.getClass().equals(type)) sb.append(" proxied as ").append(type.getSimpleName());
180        return sb.toString();
181    }
182}
