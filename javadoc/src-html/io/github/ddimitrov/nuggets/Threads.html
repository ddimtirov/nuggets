017package io.github.ddimitrov.nuggets;
018
019import org.jetbrains.annotations.Contract;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022
023import java.util.*;
024import java.util.stream.Collectors;
025
026/**
027 * <p><span class="badge green">Entry Point</span> Introspects the
028 * running threads and provides access to their runnable delegates,
029 * can take a snapshot list of all running threads.
030 * Useful for whitebox testing as well as ensuring orderly shutdown.
031 * </p>
032 * <p>An instance of this class corresponds to a {@link ThreadGroup}
033 * instance, providing access to its contained threads.</p>
034 */
035public class Threads {
036    private final ThreadGroup threadGroup;
037
038    /**
039     * Creates a new instance, introspecting the threads in {@code threadGroup}
040     * @param threadGroup the thread group to introspect.
041     *
042     * @see #getSiblingThreads()
043     * @see #getAllThreads()
044     */
045    public Threads(@NotNull ThreadGroup threadGroup) {
046        this.threadGroup = threadGroup;
047    }
048
049    /**
050     * Creates an instance introspecting all threads in the JVM.
051     *
052     * @return an instance introspecting the top-level thread group
053     *
054     * @see #Threads(ThreadGroup)
055     * @see #getSiblingThreads()
056     * @see #getParentGroupThreads()
057     */
058    @Contract(pure = true)
059    public static @NotNull Threads getAllThreads() {
060        ThreadGroup threadGroup = Thread.currentThread().getThreadGroup();
061        while (threadGroup.getParent()!=null) threadGroup = threadGroup.getParent();
062        return new Threads(threadGroup);
063    }
064
065    /**
066     * Creates an instance introspecting all threads in the same
067     * thread group with the current thread.
068     *
069     * @return an instance introspecting the current thread group
070     *
071     * @see #Threads(ThreadGroup)
072     * @see #getAllThreads()
073     * @see #getParentGroupThreads()
074     */
075    @Contract(pure = true)
076    public static @NotNull Threads getSiblingThreads() {
077        return new Threads(Thread.currentThread().getThreadGroup());
078    }
079
080    /**
081     * Creates an instance introspecting all threads in the parent
082     * thread group of the thread group wrapped by this instance.
083     *
084     * @return an instance introspecting the parent thread group
085     *         or {@code null} if this is the top group
086     *
087     * @see #Threads(ThreadGroup)
088     * @see #getAllThreads()
089     * @see #getSiblingThreads()
090     */
091    @Contract(pure = true)
092    public @Nullable Threads getParentGroupThreads() {
093        ThreadGroup parent = threadGroup.getParent();
094        return parent==null ? null : new Threads(parent);
095    }
096
097    /**
098     * <p>Returns a best effort list of all thread groups contained by
099     * the wrapped thread group.</p>
100     *
101     * <p>Note that as thread groups are created and die asynchronously,
102     * there is a natural race condition. We guarantee that this method
103     * will not fail and each element of the data has been valid at some
104     * point in time, but possibly not at the same point in time.</p>
105     *
106     * @param recurse {@code true} to return thread groups contained in
107     *                child thread groups, {@code false} to return only
108     *                direct children
109     * @return list of thread groups.
110     *
111     * @see #Threads(ThreadGroup)
112     * @see #getParentGroupThreads()
113     */
114    public @NotNull List<ThreadGroup> containedGroups(boolean recurse) {
115        ThreadGroup[] all = null;
116        int actual = 0;
117        while(all==null || all.length!=actual) {
118            all = new ThreadGroup[threadGroup.activeGroupCount()];
119            actual = threadGroup.enumerate(all);
120        }
121
122        List<ThreadGroup> groups = new ArrayList<>(Arrays.asList(all));
123        groups.removeIf(Objects::isNull);
124        if (!recurse) groups.removeIf(it -> {
125            ThreadGroup parentGroup = it.getParent();
126            return parentGroup==null || !parentGroup.equals(threadGroup);
127        });
128        return groups;
129    }
130
131    /**
132     * <p>Returns a best effort list of all threads contained by the
133     * wrapped thread group.</p>
134     *
135     * <p>Note that as threads are created and die asynchronously, there
136     * is a natural race condition. We guarantee that this method will
137     * not fail and each element of the data has been valid at some
138     * point in time, but possibly not at the same point in time.</p>
139     *
140     * @param recurse {@code true} to return threads contained in child
141     *                thread groups, {@code false} to return only threads
142     *                having this thread-group as a parent.
143     * @return list of thread groups.
144     *
145     * @see #getAll()
146     */
147    public @NotNull List<Thread> containedThreads(boolean recurse) {
148        Thread[] all = null;
149        int actual = 0;
150        while(all==null || all.length!=actual) {
151            all = new Thread[threadGroup.activeCount()];
152            actual = threadGroup.enumerate(all);
153        }
154
155        List<Thread> threads = new ArrayList<>(Arrays.asList(all));
156        threads.removeIf(Objects::isNull);
157        if (!recurse) threads.removeIf(it -> {
158            ThreadGroup parentGroup = it.getThreadGroup();
159            return parentGroup==null || !parentGroup.equals(threadGroup);
160        });
161        return threads;
162    }
163
164    /**
165     * <p>Returns a best effort list of all runnables contained by the
166     * wrapped thread group.</p>
167     *
168     * <p>Note that as threads are created and die asynchronously, there
169     * is a natural race condition. We guarantee that this method will
170     * not fail and each element of the data has been valid at some
171     * point in time, but possibly not at the same point in time.</p>
172     *
173     * @return list of thread runnables.
174     *
175     * @see #containedThreads(boolean)
176     * @see #getByName()
177     * @see #getWithUniqueName()
178     */
179    public @NotNull List<Runnable> getAll() {
180        return containedThreads(true).stream().map(Threads::extractRunnable).collect(Collectors.toList());
181    }
182
183    /**
184     * <p>Returns a best effort multi-map of thread-names to runnables
185     * for all threads contained by the wrapped thread group. If there
186     * are more than one threads with the same name, their runnables are
187     * put in a list.</p>
188     *
189     * <p>Example:</p>
190     * <pre><code>
191     *     threads.getByName().get("SEDA-PROCESSOR").forEach(SedaProcessor::stop);
192     *     assert threads.getByName().get("ComponentX").get(0).getFoo() == 42;
193     * </code></pre>
194     *
195     * <p>Note that as threads are created and die asynchronously, there
196     * is a natural race condition. We guarantee that this method will
197     * not fail and each element of the data has been valid at some
198     * point in time, but possibly not at the same point in time.</p>
199     *
200     * @return list of thread runnables.
201     *
202     * @see #containedThreads(boolean)
203     * @see #getAll()
204     * @see #getWithUniqueName()
205     */
206    public @NotNull Map<String, List<Runnable>> getByName() {
207        return containedThreads(true).stream().collect(Collectors.groupingBy(
208                Thread::getName,
209                Collectors.mapping(Threads::extractRunnable, Collectors.toList())
210        ));
211    }
212
213    /**
214     * <p>Returns a best effort map of thread-names to runnables for all
215     * uniquely named threads contained by the wrapped thread group. If
216     * there are more than one threads with the same name, their runnables
217     * are ignored.</p>
218     *
219     * <p>Example:</p>
220     * <pre><code>
221     *     assert threads.getByName().get("ComponentX").getFoo() == 42;
222     * </code></pre>
223     *
224     * <p>Note that as threads are created and die asynchronously, there
225     * is a natural race condition. We guarantee that this method will
226     * not fail and each element of the data has been valid at some
227     * point in time, but possibly not at the same point in time.</p>
228     *
229     * @return list of thread runnables.
230     *
231     * @see #containedThreads(boolean)
232     * @see #getAll()
233     * @see #getByName()
234     */
235    public @NotNull Map<String, Runnable> getWithUniqueName() {
236        Map<String, Runnable> runnables = new HashMap<>();
237        getByName().forEach((name, runnablesForName) -> {
238            if (runnablesForName.size()!=1) return; // ambiguous or missing
239            runnables.put(name, runnablesForName.get(0));
240        });
241        return runnables;
242    }
243
244    /**
245     * Extracts the runnable that the thread was created with, or
246     * returns the thread itself if it was created with {@code null}
247     * runnable.
248     *
249     * @param thread the thread we are interested in
250     * @return the runnable whose {@code run()} method implements the
251     *         logic for this thread.
252     */
253    @Contract(pure = true)
254    public static @NotNull Runnable extractRunnable(@NotNull Thread thread) {
255        Runnable threadTarget = Extractors.peekField(thread,  Thread.class, "target", Runnable.class);
256        return threadTarget == null ? thread : threadTarget;
257    }
258}
