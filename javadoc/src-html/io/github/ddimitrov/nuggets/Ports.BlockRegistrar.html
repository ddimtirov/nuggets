033/**
034 * <p><span class="badge green">Entry Point</span> Provides utilities for
035 * ad-hoc coordination of port usage for distributed systems in tests and
036 * lab environments.</p>
037 *
038 * <p> For example, when you do integration testing for a distributed system,
039 * and you want to run a couple of tests in parallel you need to make sure that
040 * none of the test environments shares ports and file paths with the others.</p>
041 *
042 * <p>To achieve that, you can use the {@code Ports} class to allocate the ports
043 * and tweak the app configuration in application-specific manner.
044 * See the user manual for details and more use-cases.</p>
045 *
046 * <p>This class has explicit lifecycle as follows:</p>
047 * <pre><code>
048 *     Ports p = new Ports(...);
049 *     // CONFIGURE: call withExporter(...), reservePort(id[, offset]), withPorts(...) as needed
050 *     p.freeze(...);
051 *     // USE: call getBasePort() and port(id)
052 *     p.close();
053 *     // DEAD: any method call would result in exception
054 * </code></pre>
055 */
056public class Ports implements AutoCloseable {
057
058    private final List<String> claimedOffsets = Collections.synchronizedList(new ArrayList<>());
059    private final Set<String> dynamicOffsets = new HashSet<>();
060    private final Set<Exporter> exporters = new HashSet<>();
061    private final @NotNull Registrar registrar;
062    private final @NotNull Supplier<Integer> dynamicPortFinder;
063    private volatile int basePort = -1;
064    private volatile boolean closed;
065
066    /**
067     * <p>Creates new ports allocator with the specified registrar
068     * and default dynamic port allocation strategy.</p>
069     *
070     * <p>The default port allocation strategy assigns offsets to the
071     * dynamic port ID's starting from the base port of the block and
072     * skipping all explicitly assigned offsets.</p>
073     *
074     * @param registrar defines a strategy for allocating a port block.
075     */
076    public Ports(@NotNull Registrar registrar) {
077        this(registrar, null);
078    }
079
080    /**
081     * <p>Creates new ports allocator with the specified registrar
082     * and default dynamic port allocation strategy.</p>
083     *
084     * @param registrar defines a strategy for allocating a port block.
085     * @param dynamicPortFinder defines a strategy for dynamically allocating
086     *        ports within a block. If {@code null}, a default strategy assigns
087     *        offsets to the dynamic port ID's starting from the {@link #basePort base port}
088     *        of the block and skipping all explicitly assigned offsets.
089     */
090    public Ports(@NotNull Registrar registrar, @Nullable Supplier<Integer> dynamicPortFinder) {
091        this.registrar = registrar;
092        this.dynamicPortFinder = dynamicPortFinder!=null
093                ? dynamicPortFinder
094                : () -> {
095                    int gapIdx = claimedOffsets.indexOf(null);
096                    return gapIdx>=0 ? gapIdx : claimedOffsets.size();
097                };
098    }

100    /**
101     * <p>Registers an {@link Exporter} that would be used to publish/validate
102     * the allocated ports. Can throw {@link PortVetoException} to reject a
103     * port and force the port allocator to use another block</p>
104     *
105     * <p>If more than one exporter is registered, exporters will be called in
106     * their registration order.</p>
107     *
108     *
109     * <p>If the implementation of this interface also implements
110     * {@link AutoCloseable}, the {@code close()} method will be called after
111     * all offsets have been exported and successfully validated, after
112     * the allocation is confirmed successful.</p>
113     *
114     * @param exporter a callback to publish or validate an allocated port.
115     * @return {@code this} instance for chaining.
116     *
117     * @see Exporter#batching(Consumer)
118     */
119    public @NotNull Ports withExporter(@NotNull Exporter exporter) {
120        if (basePort>=0) throw new IllegalStateException("Already frozen!");
121        if (closed) throw new IllegalStateException("Already closed!");
122        exporters.add(exporter);
123        return this;
124    }

126    /**
127     * Registers a port ID at desired offset within the block.
128     * Should be called as many times as needed before {@link #freeze(int)}.
129     *
130     * @param id the ID of the port we are registering.
131     *           Should be unique.
132     * @param offset the desired offset for the finally allocated port.
133     *               Should be unique.
134     *
135     * @return {@code this} instance for chaining.
136     *
137     * @throws IllegalArgumentException if any of the arguments is not unique
138     *         within the scope of {@code this} instance.
139     * @throws IllegalArgumentException if this port allocator has already been
140     *         {@link #freeze(int) frozen} or {@link #close() closed}
141     *
142     * @see #reservePort(String)
143     * @see #withPorts(int, Consumer)
144     */
145    public @NotNull Ports reservePort(@NotNull String id, int offset) {
146        if (basePort>=0) throw new IllegalStateException("Port allocation already finalized!");
147        if (closed) throw new IllegalStateException("Already closed!");
148
149        if (Exporter.BASE_PORT_ID.equals(id)) throw new IllegalArgumentException("Port ID can not be empty");
150        if (dynamicOffsets.contains(id)) throw new IllegalArgumentException("Port ID '" + id + "' already registered as dynamic (requested " + offset + ")");
151
152        int pos = claimedOffsets.indexOf(id);
153        if (pos>=0 && pos!=offset) throw new IllegalArgumentException("Port ID '" + id + "' already registered at offset: " + pos + " (requested " + offset + ")");
154
155        while (offset>=claimedOffsets.size()) claimedOffsets.add(null); // pad the list so we don't get an IOBE
156        String old = claimedOffsets.set(offset, requireNonNull(id));
157
158        if (old!=null && !old.equals(id)) {
159            claimedOffsets.set(offset, old); // EAFP to avoid TOCTOU issues
160            throw new IllegalArgumentException("Clashing port reservations for offset " + offset + ": old='" + old + "', new='" + id + "'");
161        }
162        return this;
163    }
164
165    /**
166     * Registers a port ID at automatically chosen offset within the block.
167     * Should be called as many times as needed before {@link #freeze(int)}.
168     *
169     * @param id the ID of the port we are registering. Should be unique.
170     *
171     * @return {@code this} instance for chaining.
172     *
173     * @throws IllegalArgumentException if any of the ID is not unique
174     *         within the scope of {@code this} instance.
175     * @throws IllegalArgumentException if this port allocator has already been
176     *         {@link #freeze(int) frozen} or {@link #close() closed}
177     *
178     * @see #reservePort(String, int)
179     * @see #withPorts(int, Consumer)
180     */
181    public @NotNull Ports reservePort(@NotNull String id) {
182        if (basePort>=0) throw new IllegalStateException("Port allocation already finalized!");
183        if (closed) throw new IllegalStateException("Already closed!");
184
185        if (Exporter.BASE_PORT_ID.equals(id)) throw new IllegalArgumentException("Port ID can not be empty");
186
187        int pos = claimedOffsets.indexOf(id);
188        if (pos>=0) throw new IllegalArgumentException("Port ID '" + id + "' already registered at offset " + pos + " (requested dynamic offset)");
189
190        dynamicOffsets.add(id);
191        return this;
192    }
193
194    /**
195     * Configures port IDs and freezes this allocator in one step. Uses a builder based DSL (see example).
196     * Should be called only once, IMPORTANT: internally it calls {@code freeze(basePortHint)}.
197     *
198     * <pre><code>
199     * p.withPorts(5000, register -&gt; {
200     *     register.id("foo");
201     *     register.id("bar").offset(1);
202     *     register.id("baz").offset(2);
203     *     register.id("qux");
204     * });
205     * assert p.getBasePort()==5000; // p is ready to use
206     * </code></pre>
207     *
208     * @param spec a closure configuring the builder (see example above).
209     * @param basePortHint the desired base port - the actual base port may
210     *                     differ if there is a conflict.
211     *
212     * @return {@code this} instance in {@link #freeze(int) frozen} state.
213     *
214     * @throws IllegalArgumentException if any of the ID's and offsets are conflicting.
215     * @throws IllegalArgumentException if this port allocator has already been
216     *         {@link #freeze(int) frozen} or {@link #close() closed}
217     *
218     * @see #reservePort(String)
219     * @see #reservePort(String, int)
220     * @see #freeze(int)
221     */
222    public @NotNull Ports withPorts(int basePortHint, @NotNull Consumer<SpecIdBuilder> spec) {
223        PortsSpecBuilder builder = new PortsSpecBuilder();
224        spec.accept(builder);
225        builder.flush();
226        return freeze(basePortHint);
227    }
228
229    /**
230     * Uses the {@code registrar} and {@code dynamicPortFinder} to allocate
231     * actual ports for each registered {@code portId} and transitions the
232     * internal state to USE.
233     * @param basePortHint the desired base port - the actual base port may
234     *                     differ if there is a conflict.
235     * @return this instance, on which now we can call {@link #port(String)}
236     *         and {@link #getBasePort()}
237     */
238    public @NotNull Ports freeze(int basePortHint) {
239        if (closed) throw new IllegalStateException("Already closed!");
240        if (this.basePort>=0) throw new IllegalStateException("Port range already decided!");
241
242        try {
243            int allocatedBasePort = registrar.lock(basePortHint);
244            if (allocatedBasePort<=0) {
245                throw new IllegalStateException("Failed to reserve a port range!");
246            }
247
248            for (String dynamicId : dynamicOffsets) {
249                int freePort = dynamicPortFinder.get();
250
251                while (freePort>=claimedOffsets.size()) claimedOffsets.add(null); // pad the list so we don't get an IOBE
252                String old = claimedOffsets.set(freePort, dynamicId);
253                if (old!=null) {
254                    claimedOffsets.set(freePort, old); // undo - better undo than check to avoid TOCTOU errors
255                    throw new IllegalArgumentException("The dynamic port finder caused conflict! Clashing port reservations for offset " + freePort + ": old='" + old + "', new='" + dynamicId + "'");
256                }
257            }
258
259            for (Exporter exporter : exporters) {
260                exporter.export(Exporter.BASE_PORT_ID, allocatedBasePort);
261                for (int portOffset = 0; portOffset < claimedOffsets.size(); portOffset++) {
262                    String id = claimedOffsets.get(portOffset);
263                    if (id!=null) {
264                        exporter.export(id, allocatedBasePort+portOffset);
265                    }
266                }
267            }
268            exporters.stream().filter(it -> it instanceof AutoCloseable)
269                     .forEach(it -> rethrow(((AutoCloseable) it)::close));
270            this.basePort = allocatedBasePort;
271            return this;
272        } catch (PortVetoException e) {
273            return freeze(e.port + 1);
274        }
275    }
276
277    /**
278     * The lowest port of the allocated block or {@code IllegalStateException}
279     * if {@link #freeze(int)} has not been called yet.
280     *
281     * @return the lowest port of the allocated block.
282     */
283    public int getBasePort() {
284        if (closed) throw new IllegalStateException("Already closed!");
285        if (basePort==-1) throw new IllegalStateException("Port range not allocated yet!");
286        return basePort;
287    }

289    /**
290     * Look up a port for ID.
291     * @param id the port ID, which we shall look up.
292     * @return the port corresponding to the {@code id}
293     */
294    public int port(@NotNull String id) {
295        if (basePort<=0) throw new IllegalStateException("Port allocation not finished yet! Perhaps somebody forgot to call freeze()?");
296        if (closed) throw new IllegalStateException("Already closed!");
297
298        int offset = claimedOffsets.indexOf(id);
299        return offset>=0
300                ? basePort + offset
301                : rethrow(new NoSuchElementException("No registered port for id: '" + id + "'"));
302    }

304    /**
305     * <p>Releases all resources and transitions the internal state, so that
306     * a call to any methods other than {@code close()} will result in
307     * {@code IllegalStateException}.</p>
308     *
309     * <p>This method is idempotent.</p>
310     */
311    @Override
312    public void close() {
313        if (basePort>=0) registrar.close();
314        basePort = 0;
315        closed = true;
316    }

328    /** Builder interface used to provide constrained, typesafe DSL. */
329    public interface SpecIdBuilder {
330
331        /**
332         * Specify a port ID. If followed by a call to
333         * {@link SpecOffsetBuilder#offset(int)},
334         * then this ID would be with explicit offset,
335         * otherwise it would be dynamic.

337         * @param id the port ID
338         * @return {@code this} instance for chaining.
339         */
340        @NotNull Ports.SpecOffsetBuilder id(@NotNull String id);
341    }

343    /**
344     * Typesafe builder implementation, used by extensions for specifying ports and offsets.
345     */
346    public class PortsSpecBuilder implements SpecIdBuilder, SpecOffsetBuilder {
347        private String id;
348        private Integer offset;
349
350        @Override
351        public @NotNull Ports.SpecOffsetBuilder id(@NotNull String id) {
352            flush();
353            this.id = id;
354            return this;
355        }
356
357        @Override
358        public void offset(int offset) {
359            this.offset = offset;
360            flush();
361        }
362
363        /**
364         * Needs to be called after finishing building the configuration,
365         * or the last configured port ID may be lost.
366         */
367        public void flush() {
368            if (offset != null && id == null) {
369                throw new IllegalArgumentException("Ambiguous port offset in spec: " + offset);
370            }
371            if (id == null) return;
372            if (offset == null) {
373                reservePort(id);
374            } else {
375                reservePort(id, offset);
376            }
377            offset = null;
378            id = null;
379        }
380    }

382    /**
383     * <p>A hook notified about allocated ports, and optionally validating them.
384     * Can be used to update configs, set system properties, validate port
385     * availability, etc.</p>
386     *
387     * <p>If the implementation of this interface also implements
388     * {@link AutoCloseable}, the {@code close()} method will be called after
389     * all offsets have been exported and successfully validated, after
390     * the allocation is confirmed successful.</p>
391     */
392    @FunctionalInterface
393    public interface Exporter {
394        /**
395         * The base port will be passed with empty-string ID.
396         * It is easier to check for empty string, but if you want to be explicit, you may use this constant.
397         */
398        String BASE_PORT_ID = "";
399
400        /**
401         * Called for each allocated port in the range, as well as for the base port.
402         * Note that in event of {@code PortVetoException}, you would receive the
403         * base port and port ID's more than once - the last value is the actual.
404         *
405         * @param id the port ID (empty string means base port)
406         * @param port the allocated port
407         * @throws PortVetoException to indicate that this block is bad and
408         *         request reallocation of different block.
409         */
410        void export(@NotNull String id, int port) throws PortVetoException;
411
412        /**
413         * <p>A convenient way to receive the allocated ports (without the base port)
414         * in one shot, rather than getting them one by one. Cannot do validation.</p>
415         *
416         * <p>If you need the base port maping during export, you may use the
417         * {@link BatchExporter} directly and pass {@code true} in the
418         * constructor.</p>
419         *
420         * @param publisher a consumer that does something with a map
421         *                  mapping portId to port.
422         *
423         * @return the exporter aggregating the port mappings and
424         *         on successful allocation calling the publishing
425         *         consumer.
426         *
427         * @see BatchExporter
428         */
429        static Exporter batching(Consumer<Map<String, Integer>> publisher) {
430           return new BatchExporter(false, publisher);
431        }
432    }

469    public interface Registrar extends AutoCloseable {
470        /**
471         * Returns the actual {@code basePort} of the newly allocated block,
472         * or throws exception.
473         *
474         * @param basePortHint the desired {@code basePort} to allocate port
475         *        from. If zero, that means that the registrar implementation
476         *        is free to choose.

478         * @return the base port of the newly allocated block.
479         */
480        int lock(int basePortHint);
481
482        /**
483         * Releases any resources related to this port block.
484         * After calling this method, the in the allocated block
485         * should not be considered reserved.
486         */
487        @Override void close();
488    }

490    /**
491     * A simple block-based allocation strategy, trying to bind a port at a
492     * fixed offset from {@code portBase} and on failure, retrying again
493     * at a new {@code portBase}, one block-size up.
494     */
495    public static class BlockRegistrar implements Registrar {
496        /**
497         * The address of the interface we want to use for locking.
498         */
499        public final @NotNull InetAddress bindAddress;
500
501        /**
502         * The desired number of ports in the block (subsequent port offsets
503         * should be in the [0..rangeSize-1] range).
504         */
505        public final int rangeSize;
506
507        /**
508         * The offset of the lock port, in the range [-1..rangeSize].
509         * Keep in mind that if you choose zero, you need to register
510         * the lock port, or it will clash with dynamically registered
511         * ports.
512         */
513        public final int lockOffset;
514
515        /**
516         * If {@code true} this registrar will put extra effort to make sure
517         * that the allocated {@code basePort==basePortHint+rangeSize*N}
518         * (where N is an integer number)
519         */
520        public boolean alignToBasePortHint;
521        private @Nullable ServerSocket locked;
522
523        /**
524         *
525         * @param bindAddress the address of the interface we want to use for locking.
526         * @param rangeSize the desired number of ports in the block
527         *        (subsequent port offsets should be in the [0..rangeSize-1] range).
528         * @param lockOffset the offset of the lock port, in the range [-1..rangeSize].
529         *        See {@link #lockOffset the field} for details.
530         */
531        public BlockRegistrar(@NotNull InetAddress bindAddress, int rangeSize, int lockOffset) {
532            if (lockOffset < -1) throw new IllegalArgumentException("Lock offset " + lockOffset + " should be >= -1");
533            if (lockOffset > rangeSize) throw new IllegalArgumentException("Lock offset " + lockOffset + " should be <= " + rangeSize + " (rangeSize)");
534            if (rangeSize<=0) throw new IllegalArgumentException("rangeSize: " + rangeSize);
535            this.bindAddress = bindAddress;
536            this.lockOffset = lockOffset;
537            this.rangeSize = rangeSize;
538        }
539
540        /**
541         * @param basePortHint {@inheritDoc} Negative value forces it to retry one
542         *        block up from the negated port.
543         */
544        @Override
545        public synchronized int lock(int basePortHint) {
546            if (Math.abs(basePortHint)>0xFFFF) {
547                throw new IllegalArgumentException("basePortHint: " + basePortHint);
548            }
549
550            int lastBasePort;
551            if (locked!=null) {
552                lastBasePort = locked.isBound() ? locked.getLocalPort() - lockOffset : -1;
553                rethrow(locked::close);
554            } else {
555                lastBasePort = -1;
556            }
557
558            int rangeStep = lockOffset<0 || lockOffset>=rangeSize
559                    ? rangeSize+1
560                    : rangeSize;
561
562            int basePort;
563            if (basePortHint>0) basePort=basePortHint;
564            else if (basePortHint<0) basePort = -basePortHint + rangeStep;
565            else if (basePortHint==0 && lastBasePort>0) basePort = lastBasePort + rangeStep;
566            else {
567                throw new IllegalArgumentException("basePortHint: " + basePortHint + ", lastBasePort: " + lastBasePort);
568            }
569
570            if (alignToBasePortHint && lastBasePort>0) {
571                int misalignment = Math.abs(lastBasePort - basePort) % rangeStep;
572                if (misalignment>0) {
573                    basePort += rangeStep - misalignment;
574                }
575            }
576
577            int lastPortInRange = basePort + rangeSize - 1;
578            if (basePort<=0 || lastPortInRange>0xFFFF) {
579                rethrow(new IOException("basePort: " + basePort));
580            }
581
582            int lockPort = basePort+lockOffset;
583            if (lockPort<=0 || lockPort>0xFFFF) {
584                rethrow(new IOException("lockPort: " + lockPort));
585            }
586            try {
587                locked = new ServerSocket();
588                locked.bind(new InetSocketAddress(bindAddress, lockPort));
589                return lockPort-lockOffset;
590            } catch (IOException e) {
591                return lock(-basePort);
592            }
593        }
594
595        @Override
596        public synchronized void close() {
597            if (locked !=null) rethrow(locked::close);
598        }
599    }

601    /**
602     * <p>A convenient way to receive all port mappings in one shot, rather
603     * than getting them one by one. Cannot do validation.</p>
604     *
605     * <p>This exporter internally aggregates all observed port mappings and
606     * calls the publishing consumer once on successful allocation.</p>
607     *
608     * @see Exporter#batching(Consumer)
609     */
610    private static class BatchExporter implements Exporter, AutoCloseable {
611        private final Map<String, Integer> batched = new HashMap<>();
612        private boolean withBasePort;
613        private final Consumer<Map<String, Integer>> publisher;
614
615        /**
616         * Creates an aggregating exporter, calling {@code publisher} once
617         * all mappings are finalized.
618         *
619         * @param withBasePort if {@code true}, the map passed to the
620         *        {@code publisher} includes the {@code basePort} mapped to
621         *        an empty string.
622         *
623         * @param publisher typically a closure that does something with the
624         *        ports (i.e. creates config file, calls management APIs, etc.)
625         */
626        public BatchExporter(boolean withBasePort, Consumer<Map<String, Integer>> publisher) {
627            this.withBasePort = withBasePort;
628            this.publisher = publisher;
629        }
630
631        @Override
632        public void export(@NotNull String id, int port) {
633            if (!withBasePort && id.isEmpty()) return;
634            batched.put(id, port);
635        }
636
637        @Override
638        public void close() {
639            publisher.accept(batched);
640        }
641    }
